1 /**
  2  * @class
  3  * When performing a POI search using the {@link deCarta.Core.POISearch}'s
  4  * execute() method, the search is performed by specifying a center point and
  5  * a search radius.
  6  * This Radius class defines the structure for that search radius, and has 
  7  * the following structure:
  8  * <pre>
  9  *   deCarta.Core.Radius = function(distance, uom){
 10  *       distance: float, //Radius distance
 11  *       uom: string //Unit of measure. valid values are : KM (Kilometers), M (Meters), MI (Miles), FT (Feet)
 12  *   }
 13  * </pre>
 14  *
 15  * @description A radius with distance and unit-of-measure.
 16  *
 17  * @constructor
 18  * @param {float} distance Radius distance
 19  * @param {string} uom Unit of measure. valid values are : KM (Kilometers), M (Meters), MI (Miles), FT (Feet)
 20  */
 21 deCarta.Core.Radius = function(distance, uom){
 22 
 23     this.distance = distance;
 24     this.uom = uom;
 25 
 26 }
 27 
 28 /**
 29  * Returns a string representation of the radius, with both
 30  * distance and unit-of-measure.
 31  * @return {string} Space separated string in the format "distance uom"
 32  */
 33 deCarta.Core.Radius.prototype.toString = function(){
 34     return this.distance+" "+this.uom;
 35 };
 36 
 37 /**
 38  * Localizes a distance by converting it from its current units
 39  * to either Imperial or Metric units.
 40  * @param {deCarta.Core.Locale} locale The target locale. If the country for the
 41  * target locale is 'US' or 'UK', units are converted to Imperial units. Otherwise,
 42  * units are converted to metric.
 43  * @return {deCarta.Core.Radius} This radius object, modified to be localized to the target locale
 44  */
 45 deCarta.Core.Radius.prototype.localize = function(locale){
 46     if (!locale) deCarta.Core.Exception.raise('Locale is required to localize.');
 47     
 48     if (locale.country == 'US' || locale.country == 'UK'){
 49         this.toImperial();    
 50     } else {
 51         this.toMetric();
 52     }
 53     return this;
 54 };
 55 
 56 /**
 57  * Returns a string representation of the radius, localized to the target locale.
 58  * @param {deCarta.Core.Locale} locale The target locale. If the country for the
 59  * target locale is 'US' or 'UK', units are converted to Imperial units. Otherwise,
 60  * units are converted to metric.
 61  * @return {string} Space separated string in the format "distance uom", where the
 62  * distance and unit of measure have been converted based on locale
 63  */ 
 64 deCarta.Core.Radius.prototype.toLocalizedString = function(locale){    
 65     
 66     return new deCarta.Core.Radius(this.distance, this.uom).localize(locale).toString();
 67 };
 68 
 69 /**
 70  * Adjusts the Radius object to use a better distance / uom combination for the 
 71  * current values. For example, 1000 meters would be converted to 1 km. 
 72  * 0.1 miles would be converted to 175 yards. 
 73  * @return {deCarta.Core.Radius} The modified current Radius, replacing units with better units
 74  */
 75 deCarta.Core.Radius.prototype.autoFormat = function(){
 76     
 77     switch (this.uom){
 78         case 'KM':
 79             if (this.distance <= 0.5) {
 80                 this.distance = this.distance.toFixed(1);
 81                 this.uom = 'KM';
 82             } else {
 83                 this.distance = (this.distance * 1000).toFixed(0);
 84                 this.uom = 'M'
 85             }            
 86         break;
 87         case 'M':
 88             if (this.distance > 500) {
 89                 this.distance = (this.distance / 1000).toFixed(1);
 90                 this.uom = 'KM';
 91             } else {
 92                    this.distance = this.distance.toFixed(0);
 93                    this.uom = 'M';
 94             }                      
 95         break;
 96         case 'MI':
 97             if (this.distance < 0.3){
 98                 this.distance = (1760 * this.distance).toFixed(0);
 99                 this.uom = 'YDS';                
100             } else {
101                 this.distance = this.distance.toFixed(1);
102                 this.uom = 'MI';                
103             }
104         break;
105         case 'FT':
106             if (this.distance > 300){
107                 if (this.distance > 1500){
108                     this.distance = (this.distance * 0.000189393939).toFixed(1);
109                     this.uom = 'MI';                    
110                 } else {
111                     this.distance = (0.333333333 * this.distance).toFixed(0);
112                     this.uom = 'YDS';
113                 }                
114             } else {
115                 this.distance = this.distance.toFixed(0);
116                 this.uom = 'FT';                
117             }
118         break;
119     }
120     
121     return this;
122 }

126 /**
127  * Return the best zoom level to fit this radius
128  * good for use with geolocation API
129  * @param {deCarta.Core.Position} pos
130  * @param {int} x view port width
131  * @param {int} y view port height
132  * @return {int} zoom value
133  */
134 deCarta.Core.Radius.prototype.getZoom = function(pos, x, y){
135     var degs = this.getDegrees();
136     var min = new deCarta.Core.Position(pos.lat-degs,pos.lon-degs);
137     var max = new deCarta.Core.Position(pos.lat+degs,pos.lon+degs);
138     return new deCarta.Core.BoundingBox([min,max]).getIdealCenterAndZoom(x,y).zoom;
139 }

142 /**
143  * @private
144  */
145 deCarta.Core.Radius.prototype.getDegrees = function(){
146     if(this.uom=="KM"){
147         return this.distance/111.111;
148     }else if(this.uom=="M"){
149         return (this.distance/1000)/111.111;
150     }else if(this.uom=="MI"){
151         return (this.distance*1.609)/111.111;
152     }else if(this.uom=="FT"){
153         return ((this.distance/3.28)/1000)/111.111;
154     }
155     return 0;
156 }
157 
158 /**
159  * @private
160  */
161 deCarta.Core.Radius.prototype.toMetric = function(){
162     if (this.uom == 'KM' || this.uom == 'M') return;
163     if (this.uom == 'MI'){
164         this.distance *= 1.609344;
165         this.uom = 'KM';
166     }
167     if (this.uom == 'FT'){
168         this.distance *= 0.3048;
169         this.uom = 'M';
170     }
171     return this;
172 }
173 
174 /**
175  * @private
176  */
177 deCarta.Core.Radius.prototype.toImperial = function(){
178     if (this.uom == 'MI' || this.uom == 'FT') return;
179     if (this.uom == 'KM'){
180         this.distance *= 0.621371192;
181         this.uom = 'MI';
182     }
183     if (this.uom == 'M'){
184         this.distance *= 3.2808399;
185         this.uom = 'FT';
186     }
187     return this;
188 }

190 /**
191  * Return the radius distance in miles
192  * @return {float} Radius value, in miles
193  */
194 deCarta.Core.Radius.prototype.getMiles = function(){
195     switch (this.uom){
196         case 'KM':
197             return this.distance * 0.621371192;
198         break;
199         case 'M':
200             return (this.distance / 1000) * 0.621371192;
201         break;
202         case 'MI':
203             return this.distance;
204         break;
205         case 'FT':
206             return this.distance * 0.000189393939;
207         break;
208         default:
209             deCarta.Core.Exception.raise('Unknown UOM : ' + this.uom);
210         break;
211     }
212 }
213 /**
214  * Return the radius distance in KM
215  * @return {float} Radius value, in kilometers
216  */
217 deCarta.Core.Radius.prototype.getKm = function(){
218     switch (this.uom){
219         case 'KM':
220             return this.distance;
221         break;
222         case 'M':
223             return (this.distance / 1000);
224         break;
225         case 'MI':
226             return this.distance * 1.609344;
227         break;
228         case 'FT':
229             return this.distance * 0.0003048;
230         break;
231         default:
232             deCarta.Core.Exception.raise('Unknown UOM : ' + this.uom);
233         break;
234     }
235 }
