 68     withinDistance: function(p1,p2,dist){
 69         return Math.sqrt(Math.pow(p1.x - p2.x, 2) + Math.pow(p1.y - p2.y, 2)) <= dist;

 72     /**
 73      * Attach a listener to a DOM element.
 74      * @param {string} event the name of the event to listen for
 75      * @param {HTMLDOMElement} the element that will be observed
 76      * @param {function} listener the callback function that will be invoked when the event is triggered
 77      * @param {boolean} preventDefault if set to true, the default action for the event will be
 78      * prevented from happening (For example, setting it on an <a%gt; tag will prevent the link
 79      * from being clickable. <b>NOTE: </b> setting this flag on ANY listener will prevent the
 80      * default action for the element even if other listeners (attached to the same element / event)
 81      * don't explicitly prevent it. 
 82      *
 83      */
 84     attachListener: function(event, element, listener, preventDefault){
 85         //console.log('Attaching listener');
 86         var observed = this.getElement(element);
 87         observed.attachListener(event, listener, preventDefault);
 88     },
 89 
 90     /**
 91      * Remove a listener
 92      * @param {string} event
 93      * @param {HTMLDOMElement} element
 94      * @param {function} listener
 95      */
 96     removeListener: function(event, element, listener){
 97         var observed = this.getElement(element);
 98         observed.removeListener(event, listener);
 99     },
100 
101     getElement: function(element){
102         for (var i = 0; i < this.observedElements.length; i ++){
103             if (this.observedElements[i].element == element){
104                 return this.observedElements[i];
105             }
106         }
107 
108         var observed = new deCarta.Touch.ObservedElement(element);
109 
110         this.observedElements.push(observed);
111         return this.observedElements[i];
112     },
113 
114     stopEvent: function(e){
115         try {
116             if (e.preventDefault) e.preventDefault();
117             if (e.stopPropagation) e.stopPropagation();
118             e.returnValue = false;
119             e.cancelBubble = true;
120         } catch (e){
121 
122         }
123     }
124 }

126 /**
127  * @private
128  */
129 deCarta.Touch.ObservedElement = function(domElement){
130 
131     this.element = domElement;
132     this.preventDefault = false;
133     this.listeners = {};      
134 
135     var _Pt = function(x, y, t){
136         this.x = x;
137         this.y = y;
138         this.t = t;
139     }
140 
141     this.startPoint = null;
142     this.endPoint = null;
143     this.lastPoint = null;
144     //this.moveTrail = [];
145 
146     this.longTouchListener = null;
147 
148     this.lastTapTime = 0;
149     this.lastTapLocation = null;
150 
151 
152     this.start = function(e){
153         //console.log('Start', e.currentTarget);
154         if (this.preventDefault){
155                 if (e){
156                     if (e.preventDefault) e.preventDefault();                
157                     e.returnValue = false;
158                     e.cancelBubble = true;
159                 }
160         }
161 
162         var ev = this.normalizeEvent(e);
163 
164         // code from opera sessions, opera mobile does not set the 
165         // scale parameter, we calc from touch distance
166         if (e.touches && e.touches.length == 2){
167             this.startTouchDistance = deCarta.Utilities.pixelDistance({x:e.touches[0].pageX, y:e.touches[0].pageY},{x:e.touches[1].pageX, y:e.touches[1].pageY});
168         }
169 
170         this.trigger('touchstart', ev, e);
171 
172         this.startPoint = new _Pt(ev.pageX, ev.pageY, new Date().getTime());
173         this.lastPoint = this.startPoint;
174         //start a long touch listener
175         
176         this.longTouchListener = setTimeout(this.longTouch.bind(this, ev), deCarta.Touch.LONG_TAP_TIME);
177     }
178 
179     this.move = function(e){
180        
181         if (this.preventDefault){
182                 if (e){
183                     if (e.preventDefault) e.preventDefault();                
184                     e.returnValue = false;
185                     e.cancelBubble = true;
186                 }
187         }
188 
189         var ev = this.normalizeEvent(e);
190         var scale = (e.scale || 1);
191         ev.rotation = (e.rotation || 0);
192         
193         if (scale == 1 && e.touches && e.touches.length == 2 && (window.opera || window.PalmSystem)){
194             
195             var d = deCarta.Utilities.pixelDistance({x:e.touches[0].pageX, y:e.touches[0].pageY},{x:e.touches[1].pageX, y:e.touches[1].pageY});
196             scale = d / this.startTouchDistance;
197            // alert(scale + ' ' + d + ' ' + this.startTouchDistance);
198         }
199         
200         if (scale) {
201             
202             ev.scale = scale;
203             if (e.touches && e.touches.length > 1){
204                 this.preventDouble = true;
205                 
206                 var x = 0;
207                 var y = 0;
208                 for (var j = 0; j < e.touches.length; j++){
209                     x += e.touches[j].pageX;
210                     y += e.touches[j].pageY;
211                 }
212                 ev.centerX = x / e.touches.length;
213                 ev.centerY = y / e.touches.length;
214             }            
215         }
216 
217         if (deCarta.Window.isIe()){
218             ev.pageX = (ev.clientX + document.documentElement.scrollLeft);
219             ev.pageY = (ev.clientY + document.documentElement.scrollTop);
220         }
221 
222         this.trigger('touchmove', ev, e);
223 
224         var p = new _Pt(ev.pageX, ev.pageY, new Date().getTime());
225         this.lastPoint = p;        
226         //this.moveTrail.push(p);
227     }
228 
229     this.over = function(e){     
230         
231         var ev = this.normalizeEvent(e);
232         this.trigger('touchover', ev, e); 
233     }    
234 
235     this.out = function(e){     
236         
237         var ev = e;
238         var e = e.relatedTarget || e.toElement || e.originalTarget;
239         var legitimateChild = deCarta.Utilities.isChild(e, this.element);
240         
241         if (!legitimateChild){
242             var ev = this.normalizeEvent(e);
243             this.trigger('touchout', ev, e);            
244         }
245     }
246 
247     this.end = function(e){
248         //console.log('End', e.currentTarget);
249         if (this.preventDefault){
250                 if (e){
251                     if (e.preventDefault) e.preventDefault();                
252                     e.returnValue = false;
253                     e.cancelBubble = true;
254                 }
255         }
256         
257         var ev = this.normalizeEvent(e);
258         this.trigger('touchend', ev, e);
259      
260         this.endPoint = new _Pt(ev.pageX, ev.pageY, new Date().getTime());
261 
262         //it is a double tap.
263 
264         if (this.preventDouble){
265             setTimeout(function(){
266                 this.preventDouble = false;
267             }.bind(this), deCarta.Touch.DOUBLE_TAP_TIME);
268         }
269 
270         if (!this.preventDouble && this.lastTapLocation && (new Date().getTime()) - this.lastTapTime < deCarta.Touch.DOUBLE_TAP_TIME){
271 			
272             var d = deCarta.Utilities.pixelDistance(this.startPoint, this.endPoint);
273 
274             if (d < deCarta.Touch.DOUBLE_TAP_TOLERANCE) {
275                 this.lastTapLocation = this.startPoint;
276                 var elPagePos = deCarta.Utilities.getObjectPosition(this.element);
277                 if (this.longTouchListener){
278                     //remove it.
279                     clearInterval(this.longTouchListener);
280                     this.longTouchListener = null;
281                 }                
282 
283                 this.trigger('doubletap', {
284                     relativeX: this.startPoint.x - elPagePos.x,
285                     relativeY: this.startPoint.y - elPagePos.y,
286                     pageX: this.startPoint.x - elPagePos.x,
287                     pageY: this.startPoint.y - elPagePos.y
288                     },e);
289                 
290                 return;
291             }
292         }
293 
294         //it is a single tap
295         if (this.longTouchListener){
296             //remove it.
297             clearInterval(this.longTouchListener);
298             this.longTouchListener = null;
299 
300             //and fire tap.
301             this.lastTapLocation = this.startPoint;
302             this.lastTapTime = new Date().getTime();
303             if (deCarta.Touch.withinDistance(this.startPoint, this.lastPoint, 3)){
304                 if (e.button && e.button == 2){
305                     this.trigger('alttap', ev, e);
306                 } else {
307                     this.trigger('tap', ev, e);
308                 }
309             }                      
310             
311             return;
312         }
313 
314     }
315 
316     this.longTouch = function(e){
317         
318         //make sure lastPoint is in radius of startPoint
319         if (deCarta.Touch.withinDistance(this.startPoint, this.lastPoint, deCarta.Touch.MAX_TAP_DISTANCE)){
320 
321             if (this.longTouchListener != null){
322                 this.longTouchListener = null;
323                 var elPagePos = deCarta.Utilities.getObjectPosition(this.element);
324                 var o =  {
325                     relativeX: this.startPoint.x - elPagePos.x,
326                     relativeY: this.startPoint.y - elPagePos.y ,
327                     pageX: this.startPoint.x - elPagePos.x,
328                     pageY: this.startPoint.y - elPagePos.y
329                     };
330                 
331                 this.trigger('longTouch',o,e);
332             }            
333         } else {
334             this.longTouchListener = null;
335         }
336     }
337 
338     this.normalizeEvent = function (e){
339         if (!e) return;
340         var eObj = null;
341         try {
342             if (e.targetTouches && e.targetTouches[0]) {
343                 eObj = deCarta.Utilities.extendObject({}, e.targetTouches[0]);
344             } else {
345                 eObj = deCarta.Utilities.extendObject({} , e);
346             }
347         } catch (ex){
348             eObj = deCarta.Utilities.extendObject({}, e);
349         }
350 
351  	 if ( eObj.type == 'touchend'){
352 		//touchend does not provide coords. Use the last ones we stored	
353 		eObj.pageX = this.lastPoint.x;
354 		eObj.pageY = this.lastPoint.y;
355 	 }
356 
357         var elPagePos = deCarta.Utilities.getObjectPosition(this.element);
358         if (deCarta.Window.isIe()){ // TODO needs version testing
359             eObj.pageX = (eObj.clientX + document.documentElement.scrollLeft);
360             eObj.pageY = (eObj.clientY + document.documentElement.scrollTop);
361             eObj.relativeY = eObj.pageY - elPagePos.y;
362             eObj.relativeX = eObj.pageX - elPagePos.x;
363         } else {            
364             eObj.relativeY = eObj.pageY - elPagePos.y;
365             eObj.relativeX = eObj.pageX - elPagePos.x;
366         }
367         return deCarta.Utilities.extendObject({}, eObj);
368     }
369 
370     this.attachListener = function(event, listener, preventDefault){
371         
372         if (!deCarta.Touch.events[event.toLowerCase()]) {         
373             throw('Unknown event : ' + event + '. What are you trying to do? Think carefully!');
374         }
375         //de-alias
376         event = deCarta.Touch.events[event.toLowerCase()].toLowerCase();
377 
378         if (!this.preventDefault)
379             this.preventDefault = preventDefault;
380 
381         if (!this.listeners[event]) this.listeners[event] = [];
382         this.listeners[event].push({
383             listener: listener,
384             preventDefault: preventDefault
385         });
386     }
387 
388     this.removeListener = function(event, listener){
389         if (!this.listeners[event]) return;
390         for (var i =0; i< this.listeners[event].length; i++){
391             if (this.listeners[event][i].listener == listener){
392                 this.listeners[event].splice(i, 1);
393             }
394         }
395     }
396 	
397     this.trigger = function(eventName, event, originalEvent){
398         eventName = eventName.toLowerCase();        
399         if (this.listeners[eventName]){
400             for (var i = 0; i< this.listeners[eventName].length; i++){        
401                 this.listeners[eventName][i].listener(event, originalEvent);
402             }
403         }
404     }
405 
406     this.attach = function(ev, fn, pd){
407         if (deCarta.Window.isIe()){
408             this.element.attachEvent('on'+ev, fn);
409         } else {
410             this.element.addEventListener(ev, fn, pd);
411         }
412     }
413 
414     //init stuff
415     if (deCarta.Touch.touchable){
416         this.attach('touchstart', this.start.bind(this), false);
417         this.attach('touchend', this.end.bind(this), false);
418         this.attach('touchmove', this.move.bind(this), false);        
419     } else {
420         //this.element.addEventListener('click', function (e) {this.trigger('touch', e)}.bind(this), false);        
421         this.attach('mousedown', this.start.bind(this), false);
422         this.attach('mouseup', this.end.bind(this), false);
423         this.attach('mousemove', this.move.bind(this), false);        
424     }   
425     this.attach('mouseover', this.over.bind(this), false);        
426     this.attach('mouseout', this.out.bind(this), false);
