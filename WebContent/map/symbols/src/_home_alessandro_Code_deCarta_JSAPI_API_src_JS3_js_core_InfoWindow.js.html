1 /**
  2 *
  3 * @class
  4 * the InfoWindow class is used to display a popup on the map. 
  5 * Each Map object includes an instance of an InfoWindow. A reference to 
  6 * the map's InfoWindow can be obtained by calling
  7 * map.getInfoWindow().
  8 * @see deCarta.Core.Map
  9 *
 10 * Pins automatically manage showing / hiding the infoWindow based on their
 11 * constructor options. 
 12 * @see deCarta.Core.Pin
 13 *
 14 *
 15 * @constructor
 16 * @param {deCarta.Core.Map} map 
 17 */
 18 deCarta.Core.InfoWindow = function(map){
 19     this.map = map;
 20     this.visible = false;
 21     // Create a few elements
 22     this.domElement = document.createElement('div');
 23     this.baseCssName = this.domElement.className = 'deCarta-InfoWindow';
 24     this.background = document.createElement('div');
 25     this.background.className = 'deCarta-InfoWindow-Background';
 26     this.contentWrapper = document.createElement('div');
 27     this.contentWrapper.className = 'deCarta-InfoWindow-Content-External';
 28     this.contentElement = document.createElement('div');
 29     this.contentElement.className = 'deCarta-InfoWindow-Content';
 30     //add them 
 31     this.contentWrapper.appendChild(this.contentElement);
 32     this.domElement.appendChild(this.background);
 33     this.domElement.appendChild(this.contentWrapper);
 34     //prevent touch events from going thru the win
 35     deCarta.Touch.attachListener('touchstart', this.domElement, this.devNull.bind(this), true);
 36     deCarta.Touch.attachListener('touchend', this.domElement, this.devNull.bind(this), true);
 37     deCarta.Touch.attachListener('touchstart', this.domElement, this.devNull.bind(this), true);
 38     //and sroll wheel too
 39     /** DOMMouseScroll is for mozilla. */
 40     if(this.domElement.addEventListener)
 41         this.domElement.addEventListener('DOMMouseScroll', this.scrollWheel.bind(this), false);
 42     /** IE/Opera. */
 43     this.domElement.onmousewheel = this.scrollWheel.bind(this);    
 44 }
 45 
 46 deCarta.Core.InfoWindow.prototype = {
 47 
 48     onClose: null,
 49 
 50     /**
 51      * Show the infoWindow. 
 52      * @param {object} options 
 53      * The options parameter can contain: 
 54      * <ul>
 55      * <li>target: a Pin</li>
 56      * <li>onClose: a callback function </li>
 57      * <li>content: the content to display</li>
 58      * <li>autoReorient: the window will position itself automatically (def: true)</li>
 59      * </ul>
 60      */
 61     show: function(options){
 62         if (this.visible){
 63             this.hide();
 64         }
 65         if (typeof options.content == "string"){
 66             this.contentElement.innerHTML = options.content;
 67         } else {
 68             this.contentElement.innerHTML = '';
 69             this.contentElement.appendChild(options.content);
 70         }
 71         this.target = options.target;
 72         this.onClose = options.onClose;
 73         this.cssClass = options.cssClass;        
 74         this.autoReorient = (options.autoReorient || true);
 75 
 76         setTimeout(function(){this.position()}.bind(this), 0);
 77         this.visible = true;
 78         return this.domElement;
 79     },
 80 
 81     updateText: function(text){
 82         if (typeof text == "string"){
 83             this.contentElement.innerHTML = text;
 84         } else {
 85             this.contentElement.innerHTML = '';
 86             this.contentElement.appendChild(text);
 87         }
 88 
 89         setTimeout(function(){this.position()}.bind(this), 0);
 90     },
 91 
 92 
 93     /**
 94      * Hide the infoWindow. 
 95      * If a callback is registered, it will be invoked. 
 96      */
 97     hide: function(){
 98         this.domElement.style.display = 'none';
 99         this.visible = false;
100         if (this.onClose) this.onClose();
101     },
102 
103 
104     /**
105      * @private private
106      * undocumented
107      */
108     getElement: function(){
109         return this.domElement;
110     },
111 
112 
113     /**
114      * @private private
115      * undocumented
116      */
117     position: function(){
118         this.domElement.style.display = 'block';
119 
120         var bounds = this.target.getBounds();
121         var anchorPoints = this.target.getAnchorPositions();
122 
123         var mapWidth = this.map.width;
124         var mapHeight = this.map.height;
125 
126         var width = this.domElement.offsetWidth;  
127         var height = this.domElement.offsetHeight; 
128 
129         var dockY = 'bottom';
130         var dockX = 'center';
131 
132         if (this.autoReorient) {
133             //dock bottom only if not enough space on top (top and bottom mean the opposite of what you think)
134             if (height > bounds.top ) dockY = 'top';
135             //left and right, same thing
136             if (width / 2 > bounds.left) dockX = 'right';
137             else if (width / 2 > (mapWidth - bounds.right)) dockX = 'left';
138         }
139         //now set position
140         var y = (dockY == 'bottom') ? anchorPoints.top - height : anchorPoints.bottom;
141         this.domElement.style.top = y + 'px';
142 
143         var x = (dockX == 'center') ? (anchorPoints.right + anchorPoints.left - width)  / 2 : 
144                 (dockX == 'left') ? anchorPoints.right - (width) : anchorPoints.left;
145         this.domElement.style.left = x + 'px';
146         
147         this.domElement.className = 'deCarta-InfoWindow ' + dockY + ' ' + dockX + ' ' + this.cssClass;
148         //in case you are wondering about the next 3 lines,
149         //aside from the fact that it is surprising that you are reading them
150         //the reason for their existance 
151         //is that chrome has problems, and this fixes the problems. 
152         //the problems have to do with invalidating screen areas
153         //this forces a full redraw of the element 
154         //otherwise it could be clipped. 
155         this.domElement.style.display='none';
156         this.domElement.offsetHeight; // no need to store this anywhere, the reference is enough
157         this.domElement.style.display='block';
158     },
159 
160     /**
161     * @private private
162     * undocumented
163     */
164     scrollWheel: function(e){
165         if (e.stopPropagation) e.stopPropagation();
166         else e.cancelBubble = true;        
167     },
168 
169 
170     /**
171     * @private private
172     * undocumented
173     */
174     devNull: function(e,oe){
175         deCarta.Touch.stopEvent(e);
176     }
177 
178 }
