  3 /**
  4  * @class
  5  * Standard Event object that is passed to the listener callback from the EventManager.
  6  * @param opts: an object literal whose properties will be used to extend the event object. 
  7  */
  8 deCarta.Core.Event = function(opts){
  9 	/**	
 10 	 * Object upon which the event was fired, eg Map, Pin, Shape, etc.
 11 	 */
 12 	this.object;
 13 	/**	
 14 	 * Type of event fired, eg click, moveend etc.
 15 	 */
 16 	this.eventType;
 17 	/**	
 18 	 * Reference to the actual DOM event fired.
 19 	 */
 20 	this.DOMEvent;
 21 	/**	
 22 	 * Position (eg lat, lon) of the event if applicable.
 23 	 */
 24 	this.position;
 25 	/**
 26 	 * Current center of the map (if applicable)
 27 	 */
 28 	this.center;
 29 	/**
 30 	 * Current zoom of the map (if applicable);
 31 	 */
 32 	this.zoom;
 33 	/**
 34 	 * Current size of the map (if applicable);
 35 	 */
 36 	 this.size;
 37 
 38 	deCarta.Utilities.extendObject(this, opts);
 39 
 40 }
