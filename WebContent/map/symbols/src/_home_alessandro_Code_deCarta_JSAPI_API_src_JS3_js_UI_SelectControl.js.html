1 /**
  2  * @class
  3  * Add a Select control to the map.
  4  * This class inherits from {@link deCarta.UI.MapControl}.
  5  * 
  6  * @description Select Map Control
  7  *
  8  * The SelectControl allows the user to select an area of the map by drawing around it, 
  9  * and then zoom to it.
 10  * This class inherits from {@link deCarta.UI.MapControl}.
 11  * @param opt Options A list of options with which to initialize the ScaleControl.
 12  * Valid options are:
 13  * <ul>
 14  *   <li>(string) position: (Inherited from {@link deCarta.UI.MapControl}), which should
 15  *       be set to one of: 'topLeft', 'topRight', 'bottomLeft', 'bottomRight'
 16  *       (default='bottomRight')</li>
 17  * </ul>
 18  * <b>NOTE:</b> This control uses the HTML5 canvas element, and therefore does not work in IE8 - .
 19  * It will work if used with iecanvas.  
 20  * To style this control, edit the corresponding CSS file in the css/UI dir.
 21  * @see deCarta.UI.MapControl
 22  * @see deCarta.Core.Map
 23  */
 24 
 25 
 26 deCarta.UI.SelectControl = function(opt){
 27 
 28     this.options = {
 29         layout: function(width, height, existingControls){
 30             return {top: 10, left: 10}
 31         },
 32         cssClass : 'deCarta-control-select',
 33         markerColor: 'rgba(0,153,255,0.4)',
 34         markerWidth: 3
 35     }
 36     
 37     this.options = deCarta.Utilities.extendObject(this.options, opt);
 38 
 39     deCarta.UI.MapControl.call(this, this.options);
 40     this.active = false;
 41 	
 42 }

 46 deCarta.UI.SelectControl.prototype = {
 47     
 48     initElements: function(){
 49         
 50         // if (this.img) return; //exists already ...
 51         // this.domElement = document.createElement('div');
 52         // this.domElement.style.position = 'absolute';
 53         // this.domElement.style.display = "block";
 54         
 55         // this.domElement.className = 'deCarta-control-select';
 56         // if(this.options.cssClass !== this.domElement.className){
 57         //     this.domElement.className = this.domElement.className + ' ' + this.options.cssClass;
 58         // }
 59 
 60         this.domElement = document.createElement('div');
 61         this.contentElement = document.createElement('div');
 62         this.domElement.style.position = 'absolute';
 63         this.contentElement.className = 'deCarta-control-select deCarta-control-select-off';
 64         if(this.options.cssClass !== this.contentElement.className){
 65             this.contentElement.className = this.contentElement.className + ' ' + this.options.cssClass;
 66         }
 67         this.domElement.className = "domElement";
 68         this.mapCanvas = this.options.map.canvas;
 69 
 70         this.mapDomPos = deCarta.Utilities.domPosition(this.mapCanvas);
 71                 
 72         this.drawingBoard = document.createElement('div');
 73         this.drawingBoard.id = 'SelectControl-DrawingBoard';
 74         this.drawingBoard.style.display = 'none';
 75         this.drawingBoard.style.cursor = 'url("img/pen.cur"), auto !important';
 76         
 77         this.drawingCanvas = document.createElement('canvas');
 78         this.drawingCanvas.id = 'SelectControl-Drawing';
 79         this.drawingBoard.appendChild(this.drawingCanvas);
 80         
 81         this.mapCanvas.parentNode.insertBefore(this.drawingBoard, this.mapCanvas);
 82         
 83         deCarta.Touch.attachListener('touchstart', this.drawingBoard, this.startDrawing.bind(this), true);
 84         deCarta.Touch.attachListener('touchend', this.drawingBoard, this.endDrawing.bind(this), true);
 85         deCarta.Touch.attachListener('touchmove', this.drawingBoard, this.draw.bind(this), true);
 86         deCarta.Touch.attachListener('tap', this.domElement, this.handleTap.bind(this), true);
 87         this.ready = true;
 88     },
 89     
 90     handleTap: function(){
 91         // this code was not working
 92         // ...

 99         this.active = true;
100         
101         var base = this.contentElement.className;
102         base=base.substring(0,base.indexOf(" "));
103         this.contentElement.className =  base + ' deCarta-control-select-on';
104         try {

106         this.drawingBoard.style.width = this.mapCanvas.style.width;

108         this.drawingBoard.style.cursor = 'url("img/pen.cur"), auto';
109         this.drawingBoard.style.display = 'block';

111         this.drawingCanvas.width = parseInt(this.drawingBoard.style.width);

115         }

117     
118     startDrawing: function(e, ev){
119         
120         this.BB = null;
121         this.drawing = true;
122         this.ctx = this.drawingCanvas.getContext("2d"); 
123         this.ctx.beginPath();
124         this.points = [];
125         var px = e.pageX - this.mapDomPos.left;
126         var py = e.pageY - this.mapDomPos.top;
127         
128         this.lastPoint = {x: px, y: py};
129         this.ctx.moveTo(px, py);
130         this.ctx.strokeStyle = this.options.markerColor;
131         this.ctx.lineWidth = this.options.markerWidth;
132         this.points.push(this.lastPoint);
133 
134         return false;
135     },
136     
137     endDrawing: function (e, ev){
138         this.drawing = false;
139         
140         var px = e.pageX - this.mapDomPos.left;
141         var py = e.pageY - this.mapDomPos.top;
142 
143         var point = {x: px, y: py};
144         //this.points.push(point); //somehow this is 0,0 on ipad. 
145         
146         this.center = this.findCentroid();
147         this.angleSum = this.findAngleSum();
148         
149         if (this.angleSum > 300){
150             var center = this.findCenter();
151             
152             var zoom = this.findZoom();
153         
154             //hide and zoom the map.
155             this.drawingBoard.style.display = 'none';            
156             this.active = false; 
157             
158             var base = this.contentElement.className;
159             base = base.substring(0,base.indexOf(" "));
160             this.contentElement.className =  base + ' deCarta-control-select-off';
161 
162             this.options.map.zoomTo(zoom, null, true);
163             this.options.map.centerOn(center, {animated: false});
164             return;
165             
166         }
167         return false;
168     },
169     
170     draw: function( e, ev){
171         try{            
172             if (!this.drawing) return;
173             
174             var px = e.pageX - this.mapDomPos.left;
175             var py = e.pageY - this.mapDomPos.top;
176 
177             var point = {x: px, y: py};
178             
179             this.ctx.lineTo(point.x, point.y); 
180             this.ctx.stroke();
181             var dist = deCarta.Utilities.pixelDistance(this.lastPoint, point);
182             if (dist >=10){
183                 this.points.push(point);
184                 this.lastPoint = point;
185                 
186             }
187         }catch(e){
188             alert(e)
189         }            
190         
191         return false;
192     },
193 
194     getBB: function(){
195         
196         var minX = 999999999999999;
197         var minY = 999999999999999;
198         var maxX = 0;
199         var maxY = 0;
200         
201         for (var i = 1 ; i < this.points.length; i++){
202             if (this.points[i].x < minX) minX = this.points[i].x ;
203             if (this.points[i].x > maxX) maxX = this.points[i].x ;
204             if (this.points[i].y < minY) minY = this.points[i].y ;
205             if (this.points[i].y > maxY) maxY = this.points[i].y ;
206         }
207         
208         this.topLeftPoint = this.options.map.positionFromXY(minX, minY);
209         this.btmRightPoint = this.options.map.positionFromXY(maxX, maxY);
210         
211         this.BB = new deCarta.Core.BoundingBox(this.topLeftPoint, this.btmRightPoint);
212         
213     },
214     
215     findCentroid: function(){
216         var centerX = 0;
217         var centerY = 0;
218         for (var i = 0 ; i < this.points.length; i++){
219             centerX += this.points[i].x;
220             centerY += this.points[i].y;
221         }
222         centerX /= this.points.length;
223         centerY /= this.points.length;
224         return {x: centerX, y: centerY};
225     },
226     
227     findAngleSum: function(){
228         var sum = 0;
229         var lastAngle = deCarta.Utilities.getAnglePx(this.center,this.points[0], false);
230         for (var i = 1 ; i < this.points.length; i++){
231             var ang = deCarta.Utilities.getAnglePx( this.center,this.points[i], false);            
232             var delta = lastAngle - ang;
233             if (delta < 0) delta = 0;
234             sum +=  delta;
235             
236             lastAngle = ang;
237             
238         }
239         return sum;
240     },
241     
242     findCenter: function(){
243         
244         if (!this.BB) this.getBB();
245         return this.BB.getCenter();
246     },
247     
248     findZoom: function(){
249         if (!this.BB) this.getBB();
250         return this.BB.getIdealCenterAndZoom(this.options.map).zoom;
251     },
252     
253     render: function(container){
254         
255         // if (!this.ready) this.initElements();
256         // container.appendChild( this.domElement );
257 
258         if (!this.ready) this.initElements();
259         this.domElement.appendChild( this.contentElement );
260         container.appendChild( this.domElement );
261         this.width = this.domElement.offsetWidth;
262         this.height = this.domElement.offsetHeight;                
263     }
