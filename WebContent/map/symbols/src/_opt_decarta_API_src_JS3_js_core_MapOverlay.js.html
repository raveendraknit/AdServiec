1 /**
  2  * @class
  3  * MapOverlays are containers for objects that can be drawn over a map.
  4  * These include pins, shapes, polylines and so on.
  5  * You may instantiate several MapOverlays, with different names,
  6  * to group objects together and have the
  7  * ability to show or hide sets of objects with a single call.
  8  * MapOverlays index data internally into tiles, to speed up retreival of the
  9  * necessary objects for display.
 10  * A MapOverlay is used to create a container for a set of positioned
 11  * objects on a map. Objects that can be added to the overlay are:
 12  * <ul>
 13  *  <li>Pins {@link deCarta.Core.Pin}</li>
 14  *  <li>Polylines {@link deCarta.Core.Polyline}</li>
 15  *  <li>Circles {@link deCarta.Core.Circle}</li>
 16  *  <li>Images {@link deCarta.Core.Image}</li>
 17  *  <li>Polygons {@link deCarta.Core.Polygon}</li>
 18  * </ul>
 19  * <h3>Clustering</h3>
 20  * Pins can be set to "clustering" mode. In this mode, when pins are too close to
 21  * be reliably selected by the user, they will be displayed as a single pin
 22  * with a number overlay indicating how many pins have been merged in a group.
 23  * You can specify a callback function that will be invoked when the cluster is
 24  * clicked. The function will be passed a single parameter (array of Pin objects)
 25  * containing the pins that belong to the cluster. <br />
 26  *
 27  * @description Class for managing a map overlay layer of pins or other graphic features
 28  *
 29  * @constructor
 30  * @param {object} options The options object can contain one or more of the
 31  * following properties:
 32  * <ul>
 33  *  <li>name : The name that will be assigned to the layer - default name is 'Unnamed Layer'</li>
 34  *  <li>visible: Boolean value to set initial visibility value</li>
 35  *  <li>clustering: Boolean value, <em>true</em> enables pin clustering</li>
 36  *  <li>onClusterClick: callback function invoked when a cluster is tapped. </li>
 37  * </ul>
 38  */

 62 /*
 63  * Object collection organization
 64  *
 65  * Objects in collections are organized somehow.
 66  * Specifically, they are organized in an objects by zoom structure
 67  * which has a level of keys for zoom levels, and a second level
 68  * of keys which reference object handles in a spatial manner.
 69  *
 70  * For example :
 71  *
 72  * this.objects = [
 73  *     3: [
 74  *         'tile key': objectHandle
 75  *     ]
 76  * ]
 77  *
 78  * Objects that span more than 1 tile will have references in all the spanned 
 79  * tiles. When rendering, a TileGrid will be passed in and objects in that tile
 80  * grid will be added to a rendering queue. 
 81  * After deduplication, each object's renderer will be called and the generated 
 82  * elements added to the overlay.
 83  *
 84  *
 85  */

175     /**
176      * Adds an {@link deCarta.Core.OverlayObject} to the MapOverlay.
177 	 * Currently supported OverlayObjects are:
178 	 * {@link deCarta.Core.Pin}, {@link deCarta.Core.Polyline}, 
179 	 * {@link deCarta.Core.Circle}, and {@link deCarta.Core.Image}.
180 	 *
181      * @param object: the {@link deCarta.Core.OverlayObject} to add
182      * @return object handle
183      */
184     addObject: function(object){
185         
186         var type = object.type;
187 
188         object.setZIndex(deCarta.Core.Configuration.baseOvlZ + this.options.zIndex);
189 
190         var handle = this._genHandle();
191         object.handle = handle;
192         this.objects[handle] = object;
193 
194         switch (type.toLowerCase()){            
195             case 'pin':            
196                 this.indexObject(this.pins, object, handle);
197             break;
198             case 'image':
199             case 'shape':
200                 this.shapes.push(object);
201             break;
202             default:
203                 throw('Adding ' + type + ' to overlay is not yet supported. Also depending on what "' + type + '" is, it might never be');
204             break;
205         }
206         object.registerOwner(this);
207         this.owner.render();
208         return handle;
209     },

218     /**
219      * @private
220      * Adds an object to the spatial index for this overlay.
221      * @param index : the index we weant the object in
222      * @param object: the object we are adding
223      * @param handle: handle to the object.
224      */
225     indexObject: function(index, object, handle){        
226         var nZ = deCarta.Utilities.normalizeZoom( this.currentZoom );
227 
228         if (!index[nZ]){
229             index[nZ] = {};
230         }
231         if (this.options.clustering){            
232             //check if this fits in a cluster
233             //console.log('Clustering object :', object);
234             //oh, wow ... Making special special STRONG codes.
235             var added = false;
236             for (var j = 0; j < this.clusters.length; j ++){
237                 if (this.clusters[j].addPin(object)){
238                     added = true;
239                     //console.log('Added to cluster : ', j, this.clusters[j]);
240                     break;
241                 }
242             }
243             if (!added){
244                 var clusteringOptions = deCarta.Utilities.extendObject(this.options.clusteringOptions ,{
245                     onClick: this.options.onClusterClick,
246                     threshold: this.options.clusteringThreshold,
247                     zoom: deCarta.Utilities.normalizeZoom(this.currentZoom)
248                 });
249                 var cluster = new deCarta.Core.ClusteredPin(clusteringOptions);
250 
251                 
252 
253                 cluster.addPin(object);
254                 this.clusters.push(cluster);
255                     
256                 var tile = cluster.getPosition().getTileAtZoom(nZ);
257                 var key = deCarta.Utilities.normalizeKey(deCarta.Utilities.getTileKey(tile.E, tile.N, this.currentZoom));
258                 if (!index[nZ][key]) index[nZ][key] = [];
259                 
260                 var handle = this._genHandle();
261                 cluster.handle = handle;
262                 this.objects[handle] = cluster;
263 
264                 index[nZ][key].push(handle);
265 
266                 // register as the pins owner so it can notify us of any useful
267                 // thing that happens to it.
268                 //console.log('Created cluster');
269                 cluster.registerOwner(this);                
270             }
271 
272         } else {
273         
274             var tile = object.getPosition().getTileAtZoom(nZ);
275 
276             var key = deCarta.Utilities.normalizeKey(deCarta.Utilities.getTileKey(tile.E, tile.N, this.currentZoom));
277             if (!index[nZ][key]) index[nZ][key] = [];
278             index[nZ][key].push(handle);
279 
280             // register as the pins owner so it can notify us of any useful
281             // thing that happens to it.
282             object.registerOwner(this);
283         }
284     },

286     /**
287      * @private
288      * Private method used to rebuild the spatial index when zoom levels change.
289      *
290      */
291     refreshObjects: function(){        
292         /*
293          * Very crude. Completely rebuilds the indexes - called when a position changes.
294          * Considering the main bottleneck here is the rendering of the pins
295          * this is not the end of the world, but it would be better to change it
296          * into something *smart*.
297          */
298         this._deleteClusters(); //reset the clusters
299         this.pins = [];
300         this.shapes = [];
301         for (var handle in this.objects){
302             var object = this.objects[handle];
303             if (object.type == 'pin') {
304                 this.indexObject(this.pins, object, handle);
305             } else {
306                 this.shapes.push(object);
307             }
308         }
309

357     /**
358 	 * @private
359      * Iterates through elements and figures out which need rendering
360      * This should not be called directly, rendering of the overlay objects is 
361 	 * handled by the map object.
362      * @param {deCarta.Core.TileGrid} tileGrid The current tile grid that requires rendering
363      */
364     render: function(tileGrid){
365         
366         //if (this.currentZoom != deCarta.Utilities.normalizeZoom(this.currentZoom)) return;
367 
368         if (!this.visible) return;    
369 
370         //let's try a dom element removal 
371         if (this.options.useDOMElementRemoval) var restore = deCarta.Utilities.removeElementToReinsert(this.domElement);                  
372 
373         var grid = tileGrid.getGrid();
374         var tiles = grid.tiles;
375 
376         var gridX = grid.position.getX(this.currentZoom);
377         var gridY = grid.position.getY(this.currentZoom);
378 
379         var screenX = gridX - grid.x;
380         var screenY = gridY + grid.y;
381 
382         //var restore = deCarta.Utilities.removeElementToReinsert(this.domElement);
383 
384         //this.domElement.innerHTML = null;
385         this.renderShapes(screenX, screenY, tiles);
386         this.renderPins(screenX, screenY, tiles); 
387         if (this.options.useDOMElementRemoval) restore();
388 
389     },

391     /**
392      * @private
393      * Renders the various shapes (polylines)
394      * shapes are also keyed by tile
395      * */
396     renderShapes: function(screenX, screenY, tiles){
397         //ignore key for now, render all shapes.
398         var keepList = {};
399         /*this.boundary = new deCarta.Core.MapBoundary();
400         this.boundary.setPositions(this.owner.getVisibleRect());*/
401         var bb = new deCarta.Core.BoundingBox(this.owner.getVisibleRect());
402         //console.log(this.boundary);
403         //return this.boundary.checkPosition(pos);
404         for (var i = 0; i < this.shapes.length; i++){
405             try {
406                 var obj = this.shapes[i];
407 
408                 var pos = obj.getPosition();
409 
410                 var x = pos.getX(this.currentZoom) - screenX;
411                 var y = screenY - pos.getY(this.currentZoom);
412                 //clipping...
413                 if (bb.intersects(this.shapes[i].getBoundingBox())){
414                     var elem = this.shapes[i].render(x, y, this.currentZoom, tiles);
415                     if (elem) {
416                         if (!elem.parentNode){
417                             this.visibleShapes[elem.id] = elem;
418                             keepList[elem.id] = true;
419                             this.domElement.appendChild(elem);
420                             // force render
421                             this.shapes[i].render(x, y, this.currentZoom, tiles);
422                         } else {
423                             keepList[elem.id] = true;
424                         }
425                     }
426 
427                 }
428             }catch(e) {
429             
430                 console.log("Error rendering shape : ", obj, e.message);
431             }
432         }
433 
434         //remove old shapes
435         for (var shapeId in this.visibleShapes){
436             if (!keepList[shapeId]) {
437                 var shape = this.visibleShapes[shapeId];
438                 shape.parentNode.removeChild(shape);
439                 if (shape.unrender) shape.unrender();
440                 this.visibleShapes[shapeId] = null
441                 delete this.visibleShapes[shapeId];
442             }
443         }
444 
445     },

447     /**
448 	 * @private
449 	 */
450     renderPins: function(screenX, screenY, tiles){        
451         var keepList = {};
452         var renderList = [];
453 
454         for (var i = 0; i < tiles.length; i ++){
455             var key = deCarta.Utilities.getTileKey(tiles[i].E, tiles[i].N, tiles[i].Z);
456             renderList = renderList.concat(this.renderTilePins(key, screenX, screenY));            
457         }
458 
459         /* Sort the pins by y, then x*/
460         renderList.sort(function(a, b){
461             return (parseFloat(a.style.top) - parseFloat(b.style.top));
462         });       
463 
464         for (i = 0; i < renderList.length; i++){
465             if (!renderList[i].parentNode || !renderList[i].parentNode.id){
466                 this.visiblePins[renderList[i].id] = renderList[i];
467                 keepList[renderList[i].id] = true;
468                 this.domElement.appendChild(renderList[i]);            
469             } else {
470                 keepList[renderList[i].id] = true;                
471             }        
472         }
473 
474         //remove unused
475         for (var pinId in this.visiblePins){
476             if (!keepList[pinId]) {
477                 var pin = this.visiblePins[pinId];
478                 pin.parentNode.removeChild(pin);
479                 this.visiblePins[pinId] = null
480                 delete this.visiblePins[pinId];
481             }
482         }
483 
484     },

486     /**
487      * @private
488      * renders all pins that fit on the tile
489      */
490     renderTilePins: function(key, screenX, screenY){        
491         var count = 0;
492         var normalizedZoom = deCarta.Utilities.normalizeZoom(this.currentZoom);
493         var appendList = [];
494         var i = 0;
495         
496         var originalKey = key;
497         key = deCarta.Utilities.normalizeKey(key);        
498 
499         if (this.pins[normalizedZoom]){
500             if (this.pins[normalizedZoom][key]){
501                 for (i = 0; i < this.pins[normalizedZoom][key].length; i++){
502                     
503                     var elemKey = this.pins[normalizedZoom][key][i];
504                     var obj = this.objects[elemKey];
505 
506                     var pos = obj.getPosition();
507 
508                     var x = pos.getX(this.currentZoom) - screenX;
509                     var y = screenY - pos.getY(this.currentZoom);
510 
511                     var size = obj.getSize();
512                     
513                     if (x > this.owner.width || y > this.owner.height || x + size.width < 0 || y + size.height < 0) {
514                     
515                     } else {
516                         
517                         var elem = obj.render(x, y);                        
518                         appendList.push(elem);
519                         count ++;
520                     }
521                     
522                 }
523             }
524         }
525 
526 
527         return appendList;
528     },

605     /**
606      * @private
607      * Changes zoom levels, and switches indexes if necessary. 
608      */
609     _setZoom: function(ev){
610         
611         if (ev.object != this.owner) return;
612         this.currentZoom  = ev.zoom ;
613         var nZ = deCarta.Utilities.normalizeZoom( ev.zoom );
614         
615         
616         if (this.lastFullZoom != nZ){
617 
618             //
619             // Changing here to use refreshObjects instead of migrate since they do more or less the same thing
620             // Also this is more practical for clustering
621         
622             this.refreshObjects();
623             this.lastFullZoom = nZ;
624         }
625 
626 
627         for (var i = 0; i < this.shapes.length; i++){
628             this.shapes[i].setZoom(ev.zoom);
629         }        
630

686     callChildren: function(event, e, oe, alternateEvent){
687 
688         if (!this.visible) return;
689         //console.log('Passing event: ', event, ' to the kidz');
690         var mapPosition = deCarta.Utilities.domPosition(this.owner.owner.canvas);
691 
692         var x = (e ? e.pageX : 0) - mapPosition.left;
693         var y = (e ? e.pageY : 0) - mapPosition.top;            
694         var eventPosition = this.owner.owner.positionFromXY(x, y);
695 
696         for (var i = 0; i < this.shapes.length; i++){
697             //console.log(shapeId, this.objects[shapeId]);
698             if (this.shapes[i].visible){
699                 if (this.shapes[i].checkEvent(e, eventPosition)){
700                     this.shapes[i][event](e,oe);
701                 } else if (alternateEvent){                
702                     this.shapes[i][alternateEvent](e,oe);
703                 }
704             }
705         }
706     }
