1 /**
  2  * @class
  3  * A Pin is used to graphically represent a position on the map.
  4  * It can have a custom image, and when clicked it can display custom text.
  5  * Add the pin to a map overlay using the {@link deCarta.Core.MapOverlay}:addObject()
  6  * method.
  7  * Pin extends {@link deCarta.Core.OverlayObject}.
  8  * 
  9  * @description An OverlayObject used to display a pin on a MapOverlay
 10  *
 11  * @constructor
 12  * @param options Options. May contain one or more of the following:
 13  * <ul>
 14  *  <li>(string) text: text diplayed on the pin info window - optional, default=''</li>
 15  *  <li>{HTMLDOMElement} textElement: if this option is set, this will be an element that overrides the standard pin label</li>
 16  *  <li>{@link deCarta.Core.Position} position: the geographic position at which to place the pin, required</li>
 17  *  <li>(HTMLDOMElement) image: a DOM image element representing the pin</li>
 18  *  <li>(string) imageSrc: a string containing either the URI to an image or a base64 representation of the image</li>
 19  *  <li>(int) xOffset: Number of pixels to shift the pin to the left (set to half of the image width to center the image over the position on the map), optional, default=0</li>
 20  *  <li>(int) yOffset: Number of pixels to shift the pin up (set to the full height of the image to place the image above the position on the map), optional, default=0</li>
 21  *  <li>(function) onClick: a callback function that will be notified if the pin is clicked</li>
 22  *  <li>(function) onLabelClick: a callback that will be notified if the label is clicked</li>
 23  *  <li>(function) onShowText: this function will be invoked when the pin's infowindow is shown</li> 
 24  *  <li>(function) onHideText: this function will be invoked when the pin's infowindow is hidden</li> 
 25  *  <li>(function) onBeforeShowText: this function will be invoked before the pin's infowindow is shown - returning false will prevent the infowindow from being displayed.</li> 
 26  *  <li>(function) onBeforeHideText: this function will be invoked before the pin's infowindow is hidden - returning false will prevent the window from closing</li>  
 27  *  <li>(function) onImageLoad: invoked when the pin's image completes loading</li>  
 28  *  <li>(string) cssClass: this css class will be added to the pin. Can be used for styling</li>   
 29  *  <li>(deCarta.Core.InfoWindow) infoWindow: If you require the pin to have it's own infoWindow, instead of using the global object, you can pass an instance of {deCarta.Core.InfoWindow} in this property</li> 
 30  *  <li>(boolean) textVisible: Controls visibility of pin text. Default=false
 31  *      (click pin to toggle to true)</li>
 32  *  <li>(boolean) opaque: events will stop at the pin, and not propagate down to the map</li>
 33  * </ul>
 34  *
 35  * @see deCarta.Core.OverlayObject
 36  * @see deCarta.Core.MapOverlay
 37  */
 38 
 39 deCarta.Core.Pin = function(opts){
 40     deCarta.Core.OverlayObject.call(this, opts);
 41 
 42     /* Default option set b
 43      */
 44     this.options = {
 45         text: '',
 46         infoWindow: null,
 47         position: null,
 48         image: null,
 49         imageSrc: null,
 50         xOffset: 0,
 51         yOffset: 0,
 52         onClick: null,
 53         onLabelClick: null,
 54         onShowText: null,
 55         onHideText: null,
 56         onBeforeShowText: null,
 57         onBeforeHideText: null,
 58         onImageLoad: null,
 59         cssClass: '',
 60         label: '',
 61         opaque: false,
 62         infoWinOptions:{
 63             autoReorient: true
 64         }
 65     }
 66 
 67     this.zIndex = 1;
 68 
 69     /* This is because ExtendObject only goes ONE LEVEL DEEP */
 70     if (opts.defaultPinOptions){
 71         this.options.defaultPinOptions = deCarta.Utilities.extendObject(this.options.defaultPinOptions, opts.defaultPinOptions);
 72         delete opts.defaultPinOptions;
 73     }
 74     if (opts.infoWinOptions){
 75         this.options.infoWinOptions = deCarta.Utilities.extendObject(this.options.infoWinOptions, opts.infoWinOptions);
 76         delete opts.infoWinOptions;
 77     } 
 78 
 79     this.options = deCarta.Utilities.extendObject(this.options, opts);
 80 
 81     this.type = 'pin';
 82     
 83     this.domElement = null; 
 84 
 85     this.textEnabled = true;
 86 
 87     this.visible = true;
 88 }

 90 //Define methods to extend Pin
 91 deCarta.Core.Pin.prototype = {

 93     /**
 94      * Sets the image element that will be used for the pin
 95      * @param {Image} image the image
 96      * @param {int} xOffset Number of pixels to shift the image to the left of the pin position (use half the image width to center on position) (Optional, default=0)
 97      * @param {int} yOffset Number of pixels to shift the image upward (use the full image height to place image above position) (Optional, default=0)
 98      */
 99     setImage: function(image, xOffset, yOffset){
100         this.options.image = image;
101         if (xOffset) this.options.xOffset = xOffset;
102         if (yOffset) this.options.yOffset = yOffset;
103         if (this.domElement)
104             this._render(parseFloat(this.domElement.style.left), parseFloat(this.domElement.style.top), true );
105     },
106 
107     /**
108      * Sets a src value for the pin's image
109      * @param {HTMLDOMElement} imageSrc Source of the image (you can use imageSrc to convert from a URI or base64 string)
110      * @param {int} xOffset The NEW Number of pixels to shift the image to the left of the pin position (use half the image width to center on position) (Optional, default=0)
111      * @param {int} yOffset The NEW Number of pixels to shift the image upward (use the full image height to place image above position) (Optional, default=0)
112      */
113     setImageSrc: function(imageSrc, xOffset, yOffset){
114         this.options.imageSrc = imageSrc;
115         if (xOffset) this.options.xOffset = xOffset;
116         if (yOffset) this.options.yOffset = yOffset;
117         if (this.domElement)
118             this._render(parseFloat(this.domElement.style.left), parseFloat(this.domElement.style.top), true );
119     },

122      * Returns the current image for the pin
123      * @return {HTMLDOMElement} The HTML DOM Element of the image for the pin

125     getImage: function(){

138     /** Sets the rotation angle of the pin
139      * @param {float} angle Angle to rotate the pin, counter-clockwise
140      */
141     setRotation: function(angle){
142                 
143         this.pinImage.style.MozTransform = 'rotate(-'+Math.round(angle)+'deg)';
144         this.pinImage.style.webkitTransform = 'rotate(-'+Math.round(angle)+'deg)';
145         this.pinImage.style.WebkitTransform = 'rotate(-'+Math.round(angle)+'deg)';
146         this.pinImage.style.oTransform = 'rotate(-'+Math.round(angle)+'deg)';
147         this.pinImage.style.OTransform = 'rotate(-'+Math.round(angle)+'deg)';        
148     },
149 
150     hide: function(){
151         if (this.domElement)
152             this.domElement.style.display = 'none';
153 
154         this.visible = false;
155     },
156 
157     show: function(){
158         if (this.domElement)
159             this.domElement.style.display = 'block';
160 
161         this.visible = true;
162     },
163     /**
164      * @private
165      * Generates the necessary dom elements.
166      * Returns the main element to the caller for insertion in the DOM.
167      * @return HTMLDomElement
168      */
169     render: function(x, y){        
170         if (!this.domElement) this.domElement = this._render();
171         this.domPosition(x, y);
172 
173         if(this.textVisible)
174             this.showText();
175 
176         return this.domElement;
177     },
178 
179     /**
180      * @private
181      * Actual internal rendering. This *really* creates the elements. 
182      */
183     _render: function(x, y, skipListeners){        
184         if (typeof skipListeners === 'undefined') skipListeners = false;
185         //get the x and y
186         
187         //create main element
188         if (!this.domElement){
189             this.domElement = document.createElement('div');
190             this.domElement.style.position = 'absolute';
191             this.domElement.className = 'deCarta-Pin ' + this.options.cssClass;
192             this.domElement.style.zIndex = this.zIndex;
193             this.domElement.pin = this;
194             this.domElement.id = ("deCarta-pin-" + Math.random()).replace(".", "");
195         }
196 
197         
198         //create the image        
199         if (this.options.imageSrc){
200             if (this.pinImage) {
201                 deCarta.Utilities.domRemove(this.pinImage);
202                 delete this.pinImage;
203             }
204             
205             this.pinImage = new Image();
206             this.pinImage.style.display = 'none';
207             var self = this;
208             this.pinImage.onload = function(){
209                 this.style.display = 'block'
210                 if(!this.style.height)this.style.height=this.height+"px";
211                 if(!this.style.width)this.style.width=this.width+"px";
212                 if (self.options.onImageLoad) self.options.onImageLoad(self);
213             };
214             
215             this.pinImage.style.top = '-' + this.options.yOffset + 'px';
216             this.pinImage.style.left = '-' + this.options.xOffset + 'px';
217 
218             this.pinImage.src = this.options.imageSrc;
219 
220         } else {
221             if (this.options.image){                
222                 this.pinImage = this.options.image;
223                 this.pinImage.style.top = '-' + this.options.yOffset + 'px';
224                 this.pinImage.style.left = '-' + this.options.xOffset + 'px';
225                 this.imageLoaded = true;               
226             } else {
227                 var div = document.createElement("div"); 
228                 this.domElement.className="";
229                 div.className = 'deCarta-Pin-Default ' + this.options.cssClass;
230 
231                 div.innerHTML = this.options.label;
232                 this.pinImage = div;
233             }
234         }       
235         
236         this.pinImage.style.position = 'absolute';
237 
238         if (!skipListeners){            
239             this.addListeners(this.domElement/*, true*/); //WHY was this opaque? Is there a good reason?
240         }
241         //fixes missing onclick
242         this.onclick(function(){ if (this.options.onClick) this.options.onClick() }.bind(this))
243         //put image in main element,        
244         this.domElement.appendChild(this.pinImage);
245 
246         deCarta.Core.EventManager.listen('moveend', function(){            
247             if (this.textVisible 
248                 && this.getInfoWindow().domElement.style.display!='none') {
249                 this.getInfoWindow().position();
250             }
251         }.bind(this), this.owner.owner)
252         if (!this.visible) this.hide();
253         return this.domElement;    
254     },
255 
256     tap: function(e, oe){
257         
258         this.triggerEvent('click', e, oe);
259 
260         if (this.options.text){ // only open window if there is something to show                        
261             if (this.textVisible){
262                 if (this.options.onBeforeHideText) if (!this.options.onBeforeHideText(this)) return;
263                 this.hideText();
264             } else {
265                 if (this.options.onBeforeShowText) if (!this.options.onBeforeShowText(this)) return;
266                 this.showText();
267             }
268         }
269     },
270 
271     /**
272      * Returns the bounding coordinates for the pin (in map pixel coords)
273      **/
274     getBounds: function(){
275         var top = this.domY;
276         var left = this.domX;
277 
278         var styleTop = parseFloat(deCarta.Utilities.getStyle(this.pinImage, 'top')) || 0;
279         var styleLeft = parseFloat(deCarta.Utilities.getStyle(this.pinImage, 'left')) || 0;
280 
281         return {
282             top:    top + parseFloat(styleTop),
283             left:  left + parseFloat(styleLeft),
284             right: left + parseFloat(styleLeft) + this.pinImage.offsetWidth,
285             bottom: top + parseFloat(styleTop) + this.pinImage.offsetHeight,
286             offsetX: left,
287             offsetY: top
288         };
289     },
290     
291     getAnchorPositions: function(){
292         var top = this.domY;
293         var left = this.domX;
294 
295         var styleTop = parseFloat(deCarta.Utilities.getStyle(this.pinImage, 'top')) || 0;
296         var styleLeft = parseFloat(deCarta.Utilities.getStyle(this.pinImage, 'left')) || 0;
297 
298         return {
299             left: (parseFloat(styleLeft) + parseFloat(this.pinImage.offsetWidth) / 2),            
300             right: (parseFloat(styleLeft) + parseFloat(this.pinImage.offsetWidth) / 2),
301             top: parseFloat(styleTop),
302             bottom: parseFloat(styleTop) + parseFloat(this.pinImage.offsetHeight)            
303         }
304     },
305 
306     /**
307      * If the pin is clustered, this will return the cluster it belongs to. Otherwise returns null.
308      */
309     getCluster : function(){
310         return (this.cluster ? (this.cluster.getList().length>1 ? this.cluster : null) : null);
311     },
312 
313     /**
314      * Hides the text for this pin
315      */
316     hideText: function(){
317         
318         if (!this.textVisible) return;        
319         if (this.options.onHideText) this.options.onHideText(this);
320         this.textVisible = false;
321         this.domElement.style.zIndex = this.originalZ || 100;
322         this.getInfoWindow().hide();
323     },
324 
325     getInfoWindow: function(){
326         return this.options.infoWindow || this.owner.owner.infoWindow;
327     },
328 
329     /**
330      * Unhides the text for this pin
331      */
332     showText: function(){
333         
334         if  (!this.textEnabled) return;        
335 
336         if (this.textVisible || !this.domElement) return;
337         if (this.options.onShowText) this.options.onShowText(this);
338 
339         this.originalZ = this.domElement.style.zIndex;
340         this.domElement.style.zIndex = parseFloat(this.domElement.style.zIndex) + 100;       
341         
342         var el = this.getInfoWindow().getElement();
343         this.domElement.appendChild(el);
344 
345         this.getInfoWindow().show({
346             target: this,
347             content: this.options.text,
348             autoReorient: this.options.infoWinOptions.autoReorient,
349             cssClass: this.options.cssClass,
350             onClose: function(){            
351                 this.domElement.style.zIndex = this.originalZ || 100;
352                 if (this.options.onHideText) this.options.onHideText(this);
353                 this.textVisible = false;
354                 this.domElement.style.zIndex = this.originalZ || 100;
355             }.bind(this)
356         });
357         
358         if (this.options.onTextShown) this.options.onTextShown(this);
359         this.owner.owner.render();
360         this.textVisible = true;        
361     },
362 
363     /**
364      * Disables the text for this pin
365      * This both hides the text, and disables the text from appearing the 
366      * next time the pin is clicked.
367      */
368     disableText: function(){
369         this.hideText();
370         this.textEnabled = false;
371     },
372 
373     /**
374      * Enables the text for this pin
375      * If the text was disabled, it will not appear when the pin is clicked.
376      * Enabling the text re-enables the appear-on-click functionality.
377      */
378     enableText: function(){
379         this.textEnabled = true;
380     },
381 
382     /**
383      * Gets the center position for the pin
384      * @returns {deCarta.Core.Position} 
385      */
386     getCenter: function(){
387         return this.options.position;
388     }
