1 /* 
  2  * @private. Implements precaching for tiles. Useful for desktop onlehhh
  3  */
  4 
  5 
  6 deCarta.Core.TilePrecacher = function(map, tileStore){
  7     this.maxRequests = 4;
  8     this.map = map;
  9     this.tileStore = tileStore;
 10     this.grid = new deCarta.Core.TileGrid();
 11     this.init();
 12     
 13 }
 14 
 15 
 16 deCarta.Core.TilePrecacher.prototype = {
 17     
 18     init: function(){
 19         //Start a precache poll
 20         this.precachePoll();                
 21     },
 22     
 23     precachePoll: function(){
 24         
 25         var skip = false;
 26         for (var p in this.tileStore.loadRequests){
 27             skip = true;
 28             break;
 29         }
 30         if (skip){
 31             //console.log('Load requests active, not caching', this.tileStore.loadRequests);            
 32         } else {
 33             var rCount = 0;
 34             this.grid.resize(this.map.width, this.map.height);
 35             //look at tiles we'd like to have. Load them. New poll for later. 
 36             //start at zoom level 0. repeat, on centerpoint, for every zoom level. 
 37             for (var z = 1; z < 21 && rCount < this.maxRequests; z+=5){
 38                 //get a grid on the map center and at this zoom            
 39                 this.grid.prepare(this.map.center, z);
 40                 var grid = this.grid.getGrid();
 41 
 42                 for (var i = 0; i < grid.tiles.length; i++){
 43                     var tile = grid.tiles[i];
 44                     var key = deCarta.Utilities.getTileKey(tile.E,tile.N,tile.Z);
 45                     //do we have it? 
 46                     if (!this.tileStore.availableTiles[key]){
 47                         this.tileStore.getTile(tile.E,tile.N,tile.Z);
 48                         rCount ++;
 49                     }
 50                 }
 51             }        
 52         }
 53         setTimeout(this.precachePoll.bind(this), 500);
 54     }    
 55     
 56 }
