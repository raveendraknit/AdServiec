1 /**
  2  * @class ImagePack
  3  * Allows dynamic loading of different image sets depending on resolution.
  4  *
  5  *
  6  * The ImagePack allows you to load a base64 encoded set of images (a "pack").
  7  * These are generated server-side, either statically with a build tool or
  8  * dynamically. The content of an imagePack is an object with the following structure
  9  *<pre>
 10  * {
 11  *      "image name" : {
 12  *          img : "base64 data",
 13  *          scale: (float)
 14  *      }
 15  * }
 16  *</pre>
 17  * Each image will be loaded and rescaled to screen size (depending on the DPR of the
 18  * device) according to the scale paramter. This allows you to load the same pack
 19  * for different ratios. For example, you could load a pack
 20  * containing images with a scale of 2. These images will contain *twice as many pixels*
 21  * as required for display (hence the 2 factor). When displayed on a regular screen
 22  * the extra data will be discarded, but when displayed on a high res screen
 23  * they will utilize its full resolution. Of course this works the other way around
 24  * as well : you can load a pack of low res images with a scale setting of 1, and they
 25  * will be upscaled on high res screens to display the correct size.
 26  *
 27  * A Java tool to generate image packs is provided with the api. Please refer to the
 28  * documentation in the Tools section of the API distribution.
 29  * 
 30  * @description Tool for loading images
 31  */
 32 
 33 deCarta.UI.ImagePack = {
 34 
 35     /**
 36      * Load a pack.
 37      * @param pack (string) relative of absolute uri
 38      * @param onComplete a callback function that will be notified when the pack is ready for use
 39      *
 40      */
 41     load: function(pack, onComplete, forceScript){
 42 
 43         if (typeof onComplete !== 'function')
 44             deCarta.Core.Exception.raise("Image pack loading requires a callback.");
 45         
 46 
 47         if (!forceScript && (pack.indexOf('http://') == -1 && window.location.protocol != 'file:' && !deCarta.Window.isIe())) {
 48             this.loadXHR(pack, onComplete);
 49         } else {            
 50             this.loadScript(pack, onComplete);
 51         }
 52     },
 53 
 54     loadScript: function(pack, onComplete){        
 55         var sTag = document.createElement('script');
 56         var packName = pack.substring(pack.lastIndexOf('/') + 1, pack.lastIndexOf('.'));
 57         sTag.onload = function(){
 58             this.importLoadedPack(packName, onComplete);
 59             sTag.parentNode.removeChild(sTag);
 60         }.bind(this);
 61 
 62         sTag.onreadystatechange= function (sTag) {
 63             if (sTag.readyState == 'complete'){
 64                 this.importLoadedPack(packName, onComplete);
 65                 sTag.parentNode.removeChild(sTag);
 66             }
 67         }.bind(this, sTag);
 68 
 69         sTag.src = pack;
 70         document.body.appendChild(sTag);
 71     },
 72 
 73     loadXHR: function(pack, onComplete){
 74        /* try {
 75             netscape.security.PrivilegeManager.enablePrivilege("UniversalPreferencesWrite UniversalBrowserWrite UniversalPreferencesRead UniversalBrowserRead");
 76         } catch (e) {}*/
 77 
 78         var req = new XMLHttpRequest();
 79         req.open('GET', pack, true);
 80         req.onreadystatechange = function (aEvt) {
 81             if (req.readyState == 4) {
 82                 if(req.status == 200){
 83                     (new Function(req.responseText + '; window.imagePacks = imagePacks'))();
 84                     //window.imagePacks = deCarta.Utilities.extendObject(window.imagePacks, imagePacks);
 85                     var packName = pack.substring(pack.lastIndexOf('/') + 1, pack.lastIndexOf('.'));
 86                     deCarta.UI.ImagePack.importLoadedPack(packName, onComplete);
 87                 }
 88                 else
 89                     deCarta.Core.Exception.raise('Error loading resources');
 90 
 91             }
 92         };
 93         req.send();
 94     },
 95 
 96     /**
 97      * Get an image: returns a clone of the original image for use in the application
 98      * @param id is the name of the image in the pack
 99      */
100     get: function(id){
101         //console.log(id)
102         try {
103             return this[id].cloneNode(false);
104         } catch (e){
105             deCarta.App.streamlog('Failed Pin Load', {err: 'pin ' + id + ' is not here.'});
106             return false;
107         }
108     },
109 
110     /**
111      * @private. 
112      */
113     importLoadedPack: function(packName, onComplete){
114         if (!imagePacks[packName]) {
115             deCarta.Core.Exception.raise("Image pack " + packName + " contains no images, or is in the wrong format!");
116         }
117 
118         var image = null;
119 
120         var imageCount = 0;
121         for (image in imagePacks[packName]){
122             imageCount ++;
123         }
124 
125         for (image in imagePacks[packName]){
126 
127             this[image] = new Image();
128             this[image].onload = function(img, packImg) {
129 
130                 var scale = (packImg.scale) ? packImg.scale : 1;
131 
132 
133                 //using css pix because of opera
134                 img.style.width = (img.width / scale) + 'px';
135                 img.style.height = (img.height / scale) + 'px';
136 
137                 imageCount --;
138 
139                 if (imageCount == 0) onComplete();
140 
141             }.bind(this, this[image], imagePacks[packName][image]);
142             this[image].src = imagePacks[packName][image].img;
143         }
144         onComplete();
145     }
146 
147 
148 }
