1 /**
  2  * @private
  3  * @class Global object in the deCarta.Mobile namespace used to trigger exceptions
  4  * int the api.
  5  * @description Used in the deCarta Mobile Web api to trigger exceptions
  6  *
  7  */
  8 
  9 deCarta.Core.Exception = {
 10 
 11     /**
 12      *
 13      * Throws a new exception.
 14      * If vocal exceptions are set in the config, it will also show an alert box.
 15      */
 16     raise: function raise(msg){
 17         var caller = this.getFunctionName(arguments.callee.caller);
 18         var s = '[' + caller + '] - ' + msg;
 19         if (deCarta.Core.Configuration.vocalExceptions) alert('Exception: ' + s);
 20         throw(s);
 21     },
 22 
 23     /**
 24      * @private returns the name of a function.
 25      * @param {function} func the function of which we want to knwo the name
 26      */
 27     getFunctionName: function getFunctionName( func ) {
 28 
 29         // Sort out the caller of this function
 30         // For Mozilla        
 31         if( func.name ) {
 32             return func.name;
 33         }
 34 
 35         // try to parse the function name from the defintion for other browsers
 36         var definition = func.toString();
 37         var name = definition.substring(definition.indexOf('function') + 8, definition.indexOf('('));
 38         if ( name.replace(/ /g, "") ) {
 39             return name.replace(/ /g, "");
 40         }
 41 
 42         // We made it here if the function is anonymous
 43         return "anonymous";
 44     }
 45 
 46 }
