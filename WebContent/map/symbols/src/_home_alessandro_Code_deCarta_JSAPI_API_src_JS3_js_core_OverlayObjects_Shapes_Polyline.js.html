1  /**
  2  * @class
  3  * Polyline is a map overlay object used to draw a complex line
  4  * (such as a route) on the map.
  5  * It extends {@link deCarta.Core.OverlayObject}.
  6  * Use the {@link deCarta.Core.MapOverlay}:addObject() method to add 
  7  * this object to a map overlay.
  8  * (for example a route).
  9  *
 10  * @constructor
 11  * @param {object} opts Options. May contain one or more of the following
 12  * <ul>
 13  *  <li>lineGeometry: Array of deCarta.Mobile.Position</li>
 14  *  <li>fillColor: A polyline has thickness, and therefore has a fill color
 15  *      - optional, default '#000'</li>
 16  *  <li>strokeColor: Color hat will be used to stroke the line
 17  *      - optional, default '#0000FF'</li>
 18  *  <li>strokeSize: Size of the line, in pixels - optional, default 6</li>
 19  *  <li>strokeOpacity: Opacity of the line (1.0 = opaque, 0.0 = transparent)
 20  *      - optional, default 0.6</li>
 21  * </ul>
 22  *
 23  * @see deCarta.Core.OverlayObject
 24  * @see deCarta.Core.MapOverlay
 25  */
 26  /*
 27  * <p><b>POINT INDEXES AND THINNING</b></p>
 28  * <p>
 29  * When a line is rendered, all the points are indexed by the tile they
 30  * fall on  to, given the zoom level. On zoom level changes, a new
 31  * index is built.  
 32  *</p>
 33  *<p>
 34  * Creating the index is a linear operation, we look at each point and assign it
 35  * to a tile key. The index structure is an object with properties named as
 36  * tile keys which contain an array of indexes referring to the list of
 37  * positions that make up the polyline.
 38  *</p>
 39  * For example :
 40  *<pre>
 41  * var index = [
 42  *      10: {
 43  *          '10_-347_113' : [0, 34]
 44  *      }
 45  * ]
 46  *</pre>
 47  *<p>
 48  * <b>NOTE:</b> the indexes refer to the genGeom array, which contains (by zoom lev)
 49  * the generalized geometry (contiguous points). These indexes contain indexes into
 50  * the real geometry.
 51  *</p>
 52  *<p>
 53  * This means we have an index built for level 10, and tile 10, -347, 133 contains
 54  * points 0 and 34 of our positions array.
 55  *</p>
 56  * Points in the index have already undergone thinning (in fact it is part of
 57  * the same linear operation that builds the index) so they can simply be taken
 58  * and drawn on screen. 
 59  *
 60  * @description An OverlayObject used to display a polyline on a MapOverlay
 61  *
 62 */
 63 
 64 deCarta.Core.Polyline = function(options){
 65     opts = {
 66         lineGeometry: null,        
 67         strokeWidth: 6,
 68         granularity: 7 //not documented
 69     }
 70 
 71     options = deCarta.Utilities.extendObject(opts, options);
 72 
 73     deCarta.Core.Shape.call(this, options);
 74     //check if lineGeomtry is an array of pos
 75     if (!options.lineGeometry || options.lineGeometry.length == 0)
 76         deCarta.Core.Exception.raise('You need to provide a lineGeometry to instantiate a polyline');
 77 
 78 
 79     this.pointIndex = [];
 80     this.genGeom = []; //array of generalized geometry, per zoom level 
 81 
 82     this.positions = [];
 83 
 84     this.lastRenderedPointSet = null;
 85     
 86     
 87     if (typeof this.options.lineGeometry[0] == 'object' && this.options.lineGeometry[0].getLat){
 88         //these are positionS!!!
 89 
 90         this.positions = this.options.lineGeometry.slice();
 91         
 92     } else {
 93         for(var i = 0; i< this.options.lineGeometry.length; i++){
 94             var ll = this.options.lineGeometry[i].split(' ');
 95             var tPos = new deCarta.Core.Position(parseFloat(ll[0]), parseFloat(ll[1]));
 96             this.positions.push(tPos);
 97         }
 98     }
 99 
100 
101     this.type = 'shape';
102 
103     
104     this.padding = 5;
105 
106     this.findBounds();
107     
108     
109     this.boundingBox = new deCarta.Core.BoundingBox(this.positions);
110     
111 }

178

252     renderCanvas: function(x, y, z, tiles){
253         
254         if (!this.canvasElement){
255             this.findBounds();
256             this.canvasElement = document.createElement('canvas');
257 
258             this.canvasElement.setAttribute("id", ("deCarta-shape-" + Math.random()).replace(".", ""));
259             this.canvasElement.className = 'deCarta-polyline';        
260             this.canvasElement.style.zIndex = this.zIndex;
261             this.canvasElement.style.position = 'absolute';            
262            
263             return this.canvasElement;
264         }
265 
266         if (!this.canvasElement.getContext){
267             try {
268                 G_vmlCanvasManager.initElement(this.canvasElement);
269             } catch (e){
270                 deCarta.Core.Exception.raise('Error in drawing the shape - have you added exCanvas to your page?');
271             }
272         }        
273 
274         this.indexAndThinPoints(z);
275                 
276         var dim = this.getSize(z);
277         var topClip = 0;
278         var leftClip = 0;
279 
280         //clipping .. otherwise this is terrible.
281 
282         if (x < 0) {
283             leftClip = x;
284             x = 0;
285         }
286         if (y < 0) {
287             topClip = y;
288             y = 0;
289         }
290         if (x + dim.width > window.innerWidth) dim.width = window.innerWidth - x;
291         if (y + dim.height > window.innerHeight) dim.height = window.innerHeight - y;
292 
293         //this.canvasElement.setAttribute('style', "position: absolute;"); //why?
294         this.domPosition((x - this.padding), (y - this.padding), this.canvasElement);
295 
296         //maybe we can skip this rendering? depends.
297         if (z == this.lastZ &&
298             leftClip == this.lastLeftClip &&
299             topClip == this.lastTopClip &&
300             dim.width == this.lastWidth &&
301             dim.height == this.lastHeight){
302             
303             return this.canvasElement;
304 
305         }
306 
307         this.lastZ = z;
308         this.lastLeftClip = leftClip;
309         this.lastTopClip = topClip;
310         this.lastWidth = dim.width;
311         this.lastHeight = dim.height;
312 
313         this.canvasElement.width = dim.width + (this.padding*2);
314         this.canvasElement.height = dim.height+ (this.padding*2);
315 
316         var dX = this.topLeftPoint.getX(z) - leftClip;
317         var dY = this.topLeftPoint.getY(z) + topClip ;
318 
319 
320         var ctx = this.canvasElement.getContext('2d');
321         ctx.beginPath();        
322 
323         var points = this.preparePoints(z, dX, dY, tiles);
324 
325         var moved = false;
326         
327         for (var i = 0; i < points.length; i++){
328             p = points[i];
329             if (!moved) {
330                 ctx.moveTo(p.x, p.y);
331                 moved = true;
332             }
333             else ctx.lineTo(p.x, p.y);
334         }
335 
336         ctx.strokeStyle = this.options.strokeColor;
337         ctx.lineWidth = this.options.strokeWidth;
338         ctx.globalAlpha = this.options.strokeOpacity;
339         ctx.lineCap = 'round';
340 
341         ctx.stroke();
342         
343         return this.canvasElement;

347     /**
348      * @private
349      */
350     indexAndThinPoints: function(z){
351         //Check the point index
352         var nz = deCarta.Utilities.normalizeZoom(z);
353         if (!this.pointIndex[nz]){
354             
355             var tile = null;
356             var key = null;
357             var posLength = this.positions.length;
358             //use these to thin out teh points.
359             var lastPoint = null;
360             var tx = 0;
361             var ty = 0;
362             var distance = null;
363             var lastTileX = null;
364             var lastTileY = null;
365 
366             this.pointIndex[nz] = {};
367             this.genGeom[nz] = [];
368 
369             for (var i = 0; i < posLength; i++){
370                 var p = this.positions[i];
371 
372                 tx = p.getX(z);
373                 ty = p.getY(z);
374 
375                 if (lastPoint){
376                     distance = Math.sqrt(Math.pow(tx - lastPoint.x, 2) + Math.pow(ty - lastPoint.y, 2));
377                 }
378 
379                 if (distance > this.options.granularity || !lastPoint || i == (posLength - 1)){
380                     tile = p.getTileAtZoom(nz);
381                     key = deCarta.Utilities.getTileKey(tile.E, tile.N, nz);
382                     if (!this.pointIndex[nz][key]) this.pointIndex[nz][key] = [];
383 
384                     this.genGeom[nz].push(i);
385                     // keep only a ref to the index here
386                     // so this is an index into this.genGeom[nz] which is in turn
387                     // an index into the geometry. just so u know.. 
388                     this.pointIndex[nz][key].push( this.genGeom[nz].length - 1 );
389 
390                     // now this is important. What happens here is that :
391                     // 1 - if we have a previous point, check
392                     // 2 - if the tiles are different, add 
393                     // 3 - BOTH points to all tiles that connect the two
394                     if (lastTileX && lastTileY){
395                         if (tile.E != lastTileX || tile.N != lastTileY){
396 
397                             var x0 = Math.min(tile.E, lastTileX);
398                             var x1 = Math.max(tile.E, lastTileX)
399                             var y0 = Math.min(tile.N, lastTileY);
400                             var y1 = Math.max(tile.N, lastTileY);
401 
402                             var dx = Math.abs(x1-x0);
403                             var dy = Math.abs(y1-y0);
404                             var sx = (x0 < x1) ? 1 : -1;
405                             var sy = (y0 < y1) ? 1 : -1;
406                             var err = dx-dy;
407 
408                             while(true){
409                                 
410                                 var eK = deCarta.Utilities.getTileKey(x0, y0, nz);
411                                 if (!this.pointIndex[nz][eK]) this.pointIndex[nz][eK] = [];
412                                 this.pointIndex[nz][eK].push( this.genGeom[nz].length - 1 );
413                                 this.pointIndex[nz][eK].push( this.genGeom[nz].length - 2 );
414 
415                                 if ((x0==x1) && (y0==y1)) break;
416 
417                                 var e2 = 2*err;
418 
419                                 if (e2>-dy){
420                                     err -= dy;
421                                     x0  += sx;
422                                 }
423                                 if (e2 < dx){
424                                     err += dx;
425                                     y0  += sy;
426                                 }
427                             }
428                             
429                         }
430                     }
431                     lastTileX = tile.E;
432                     lastTileY = tile.N;
433 
434                     lastPoint = {x: tx, y: ty};
435                 }
436             }            
437         }
438     },

440     /**
441      * @private
442      */
443     preparePoints: function(z, dX, dY, tiles){
444         
445         var points = [];
446         var nz = deCarta.Utilities.normalizeZoom(z);
447         var p = null;
448         var px = null;
449         var py = null;
450         var addedPoints = {};
451 
452         for (var i = 0; i< tiles.length; i++){
453             var key = deCarta.Utilities.getTileKey(tiles[i].E, tiles[i].N, nz);
454             if ( this.pointIndex[nz][key])
455                 for (var j= 0; j< this.pointIndex[nz][key].length; j++){
456 
457                     p = this.positions[this.genGeom[nz][this.pointIndex[nz][key][j]]];
458                     px = p.getX(z) - dX + this.padding;                    
459                     py = dY - p.getY(z) + this.padding;
460 
461                     if (!addedPoints[p.toString()]){
462                         addedPoints[p.toString()] = true;                        
463                         points.push({x: px, y: py, i: this.pointIndex[nz][key][j]});
464                     }                    
465                 }
466         }
467 
468         points.sort(function(a,b){return a.i - b.i});        
469         return points;
470

473     /**
474      * @private
475      */
476     findBounds: function(){    
477         
478         this.topLeftPoint = null;
479         this.btmRightPoint = null;
480 
481         var minX = null;
482         var minY = null;
483         var maxX = null;
484         var maxY = null;
485 
486         for (var i = 0; i < this.positions.length; i++){
487 
488             var pt = this.positions[i];
489             
490             var ptX = pt.getX(18);
491             var ptY = pt.getY(18);
492 
493             if (ptX < minX || !minX) minX = ptX;
494             if (ptY < minY || !minY) minY = ptY;
495 
496             if (ptX > maxX || !maxX) maxX = ptX;
497             if (ptY > maxY || !maxY) maxY = ptY;
498 
499             //delete pt;
500         }
501 
502         this.topLeftPoint = new deCarta.Core.Position(0,0);
503         this.btmRightPoint = new deCarta.Core.Position(0,0);
504 
505         this.topLeftPoint.setXY(minX, maxY, 18);
506         this.btmRightPoint.setXY(maxX, minY, 18);
507 
508         this.boundingBox = new deCarta.Core.BoundingBox([this.topLeftPoint, this.btmRightPoint]);
509     },
