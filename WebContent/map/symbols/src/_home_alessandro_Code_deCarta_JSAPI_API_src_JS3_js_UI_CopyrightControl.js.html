1 /**
  2  * @class
  3  * Add a copyright string to the map.
  4  * This class inherits from {@link deCarta.UI.MapControl}.
  5  *
  6  * @description Copyright Map Control
  7  *
  8  * @constructor
  9  * @param opt Options A list of options with which to initialize the CopyrightControl.
 10  * Valid options are:
 11  * <ul>
 12  *   <li>(string) position: (Inherited from {@link deCarta.UI.MapControl}), which should
 13  *       be set to one of: 'topLeft', 'topRight', 'bottomLeft', 'bottomRight' (default='topLeft')</li>
 14  * </ul>
 15  * To style this control, edit the corresponding CSS file in the css/UI dir.
 16  * @see deCarta.UI.MapControl
 17  * @see deCarta.Core.Map
 18  */
 19 
 20 deCarta.UI.CopyrightControl = function(opt){
 21     this.options = {
 22         text:"deCarta 2012", /* default */
 23         layout: function(width, height, existingControls){
 24             return {top: height - this.height, left:  0};
 25         }
 26     }
 27     
 28     this.options = deCarta.Utilities.extendObject(this.options, opt);
 29     
 30     deCarta.UI.MapControl.call(this, this.options);  
 31 }
 32 
 33 
 34 
 35 //Define methods to extend CopyrightControl
 36 deCarta.UI.CopyrightControl.prototype = {
 37 	
 38     /**
 39 	 * @private
 40 	 */
 41     ready: false,
 42 
 43     /**
 44 	 * @private
 45      */
 46     initElements: function(){   
 47 
 48         this.domElement = document.createElement('div');
 49         this.contentElement = document.createElement('div');
 50         this.domElement.style.position = 'absolute';
 51         this.contentElement.className = 'deCarta-control-copyright';
 52         if(this.options.cssClass !== this.contentElement.className){
 53             this.contentElement.className = this.contentElement.className + ' ' + this.options.cssClass;
 54         }
 55         this.domElement.className = "domElement";
 56 
 57         this.contentElement.innerHTML = this.options.text;
 58         this.ready = true; 
 59         this.height = (this.options.height || 20);
 60     },
 61 
 62     /**
 63 	 * This render method implements the render method from the 
 64 	 * {@link deCarta.UI.MapControl} base class.
 65 	 * It is responsible for rendering this control on the map,
 66 	 * and produces a single HTML Dom Element containing the whole
 67 	 * GUI for the control.
 68 	 * @param {string} container The DOM element within which the control is rendered. 
 69 	 */
 70     render: function(container){
 71 
 72         if (!this.ready) this.initElements();
 73         this.domElement.appendChild( this.contentElement );
 74         container.appendChild( this.domElement );
 75         this.width = this.domElement.offsetWidth;
 76         this.height = this.domElement.offsetHeight;        
 77 
 78     }
 79 
 80 }; //end CopyrightControl prototype
 81 
 82 //Extend the MapControl with the additional methods for CopyrightControl
 83 deCarta.UI.CopyrightControl.prototype = deCarta.Utilities.inherit(deCarta.UI.CopyrightControl.prototype, deCarta.UI.MapControl.prototype);
