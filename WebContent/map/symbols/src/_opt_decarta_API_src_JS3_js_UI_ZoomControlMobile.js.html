1 /**
  2  * @class ZoomControlMobile - Add a zoom control to the map to present the user
  3  * with zoom in / zoom out buttons.
  4  * This class inherits from {@link deCarta.UI.MapControl}.
  5  *
  6  * @description Zoom Map Control
  7  *
  8  * @constructor
  9  * @param opt Options A list of options with which to initialize the ZoomControlMobile.
 10  * Valid options are:
 11  * <ul>
 12  *   <li>(string) position: (Inherited from {@link deCarta.UI.MapControl}), which should
 13  *       be set to one of: 'topLeft', 'topRight', 'bottomLeft', 'bottomRight' (default='topLeft')</li>
 14  * </ul>
 15  * To replace the images used in this control, place them in the correct (std / hires)
 16  * pack in the resources directory. Read more about image packs in the docs index.
 17  *
 18  * @see deCarta.UI.MapControl
 19  * @see deCarta.Core.Map
 20  */
 21 
 22 deCarta.UI.ZoomControlMobile = function(opt){
 23     this.options = {
 24         autoResize: true,        
 25         cssClass : 'deCarta-control-zoom-mobile',
 26         layout: function(width, height, existingControls, allControls){
 27             
 28             var top = 10;
 29             if (allControls)
 30                 for (var i = 0; i < allControls.length; i++){
 31                     var c = allControls[i];
 32                     if (c instanceof deCarta.UI.PanControl){
 33                         var p = c.options.layout(width, height, existingControls);                        
 34                         if (p.top == 10 && p.left == 10){                        
 35                             var top = 70;
 36                         }
 37                     }
 38                 }
 39             
 40             return {top: top, left: 20}
 41         }
 42     }
 43     this.options = deCarta.Utilities.extendObject(this.options, opt);    
 44     deCarta.UI.MapControl.call(this, this.options);    
 45 }
 46 
 47 
 48 //Define methods to extend ZoomControlMobile
 49 deCarta.UI.ZoomControlMobile.prototype = {
 50 
 51     /**
 52 	 * @private
 53 	 */
 54     ready: false,
 55 	
 56     /**
 57 	 * @private
 58 	 */
 59     zoomLevels: 21,
 60 	
 61     /**
 62 	 * @private
 63 	 */
 64     zoomOfs: 0,
 65 
 66     /**
 67 	 * @private
 68 	 */
 69     initElements: function(){
 70 
 71         if (this.domElement) return;
 72 
 73         this.domElement = document.createElement('div');
 74         this.contentElement = document.createElement('div');
 75         this.domElement.style.position = 'absolute';
 76         this.contentElement.className = 'deCarta-control-zoom-mobile';
 77         if(this.options.cssClass !== this.contentElement.className){
 78             this.contentElement.className = this.contentElement.className + ' ' + this.options.cssClass;
 79         }
 80         this.domElement.className = "domElement";
 81 
 82 
 83         this.zoomIn = document.createElement('div');
 84         this.zoomOut = document.createElement('div');
 85         
 86         this.zoomIn.className = "deCarta-control-zoomInBtn-mobile";
 87         this.zoomOut.className = "deCarta-control-zoomOutBtn-mobile";
 88 
 89 
 90         deCarta.Touch.attachListener('tap', this.zoomIn, function(ev){
 91              this.options.map.zoomIn();
 92         }.bind(this), false);
 93 
 94         deCarta.Touch.attachListener('tap', this.zoomOut, function(ev){
 95              this.options.map.zoomOut();
 96         }.bind(this), false);
 97         
 98         this.contentElement.appendChild(this.zoomIn);
 99         this.contentElement.appendChild(this.zoomOut);        
100         this.ready = true; 
101     },
102 
103     /**
104     * This render method implements the render method from the 
105     * {@link deCarta.UI.MapControl} base class.
106     * It is responsible for rendering this control on the map,
107     * and produces a single HTML Dom Element containing the whole
108     * GUI for the control.
109     * @param {string} container The DOM element within which the control is rendered. 
110     */
111     render: function(container){
112         if (!this.ready) this.initElements();
113         this.domElement.appendChild( this.contentElement );
114         container.appendChild( this.domElement );
115         this.width = this.domElement.offsetWidth;
116         this.height = this.domElement.offsetHeight;
117     }
118 }; //end ZoomControlMobile prototype
119 
120 //Extend the MapControl with the additional methods for ZoomControlMobile
121 deCarta.UI.ZoomControlMobile.prototype = deCarta.Utilities.inherit(deCarta.UI.ZoomControlMobile.prototype, deCarta.UI.MapControl.prototype);
