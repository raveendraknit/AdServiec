1 /**
  2  * @private
  3  * Used by the MapOverlay to cluster pins together. Not an end-user class. 
  4  */
  5 deCarta.Core.ClusteredPin = function(opts){
  6 
  7     deCarta.Core.OverlayObject.call(this, opts);
  8 
  9     this.options = {
 10         threshold: 50,
 11         zoom: null,
 12         image: null,
 13         imageSrc: null,
 14         xOffset: 0,
 15         yOffset :0,
 16         onClick: function(){}
 17     }
 18 
 19     this.options = deCarta.Utilities.extendObject(this.options, opts);
 20 
 21     if (!this.options.zoom)
 22         deCarta.Core.Exception.raise('You cannot instantiate a ClusteredPin without specifying the zoom');
 23 
 24     this.pin = null;
 25     this.cluster = null;
 26     this.point = null;
 27     this.threshold = this.options.threshold;
 28     this.domElement = null;
 29 
 30 }

 33 
 34     /**
 35      * Attempts to add a pin to the cluster.
 36      * returns true if successful, false if pin is outside cluster
 37      */
 38     addPin: function(pin){
 39         if (!this.cluster) {
 40             this.cluster = [];
 41             this.cluster.push(pin);
 42             this.options.position = pin.getPosition();
 43             this.centerPoint = this.options.position.getPixelPoint(this.options.zoom);            
 44             pin.cluster = this;
 45             return true;
 46         } else {
 47             var newPoint = pin.getPosition().getPixelPoint(this.options.zoom);
 48             if (deCarta.Utilities.pixelDistance(this.centerPoint, newPoint) < this.options.threshold){
 49                 this.cluster.push(pin);
 50                 pin.cluster = this;
 51                 return true;
 52             } else {
 53                 return false;
 54             }
 55         }
 56     },
 57 
 58     getCount: function(){
 59         //console.log('get count for ', this);
 60         var count = 0;
 61         for (var i = 0 ;i < this.cluster.length; i++){
 62             if (this.cluster[i] && this.cluster[i].visible){
 63                 count ++;
 64             }
 65         }
 66         //console.log('count', count);
 67         return count;
 68     },
 69     /**
 70      * if you have a pin that you would like to see, but it is inside a cluster, this function will 
 71      * take the pin and zoom the map down as far as needed to break the pin out of the cluster.
 72      * from the pin you can get the cluster with: <code>pin.getCluster()</code>.
 73      * @param pin {Pin}
 74      * @param {Function} callback called when zooming is complete
 75      * @param {Function} unbreakable called if pin cannot be broken out of cluster
 76      */
 77     zoomDownToBreakPinFromCluster : function(pin, callback, unbreakable){
 78 
 79         var nearest=null;
 80         var minDist = parseFloat("Infinity");
 81 
 82         for(var i=0;i<this.cluster.length;i++){
 83             if(pin.objectId!==this.cluster[i].objectId){
 84                 var d = deCarta.Utilities.positionPixelDistance(pin.getPosition(),this.cluster[i].getPosition(),19);
 85                 if (d<minDist){
 86                     nearest=this.cluster[i];
 87                     minDist = d;
 88                 }
 89             }
 90         }
 91         if(minDist<=this.options.threshold){
 92             unbreakable();
 93             return;
 94         }
 95         var currZoom = pin.owner.owner.zoom;
 96         for(var z = currZoom; z<=19; z++){
 97             var d = deCarta.Utilities.positionPixelDistance(pin.getPosition(),nearest.getPosition(),z);
 98             if(d>this.options.threshold){
 99                 var zEnd = function (e){
100                     deCarta.Core.EventManager.stopListening('zoomend', this.handley); 
101                     setTimeout(callback,100)
102                 }.bind(this)    
103                 this.handley = deCarta.Core.EventManager.listen('zoomend', zEnd, pin.owner.owner);   
104                 pin.owner.owner.centerOn(pin.getPosition(), {animated: false});
105                 pin.owner.owner.zoomTo(z);
106                 return;
107             }
108         }
109     },
110 
111     getSingle: function(x,y){
112         for (var i = 0; i < this.cluster.length; i++){
113             if (this.cluster[i].visible) break;
114         }
115         return this.cluster[i].render(x, y);
116     },
117 
118     render: function(x, y){
119         var count = this.getCount();
120 
121         if (count == 1) return this.getSingle(x,y);
122 
123         var zIndex = this.cluster[0].zIndex;
124 
125         if (!this.domElement){
126             this.domElement = document.createElement('div');
127             this.domElement.className = 'deCarta-Pin-Cluster';
128             this.domElement.style.zIndex = zIndex;
129             this.domElement.id = ("deCarta-pin-" + Math.random()).replace(".", "");
130             
131             this.domElement.style.position = 'absolute';
132             this.imgElement = this.chooseIcon();
133             this.imgElement.style.position = 'absolute';            
134 
135             this.labelElement = document.createElement('div');
136             this.labelElement.className = 'deCarta-Pin-Cluster-label';
137             
138 
139 
140 
141             this.domElement.appendChild(this.imgElement);
142             this.domElement.appendChild(this.labelElement);
143 
144             deCarta.Touch.attachListener('tap', this.domElement, this.onClick.bind(this), false);
145         }
146 
147         /*var img = deCarta.UI.ImagePack.get('pinDefaultImage');
148 
149         var iHeight = parseFloat(img.style.height);
150         var iWidth = parseFloat(img.style.width);*/
151 
152         this.domPosition(x - this.options.xOffset, y - this.options.yOffset);
153 
154         /*this.domElement.style.top  = y - iHeight + 'px';
155         this.domElement.style.left = x - (iWidth / 2)+ 'px';*/
156 
157         this.labelElement.innerHTML = count;
158         
159         this.domElement.style.display = (count > 0) ? 'block' : 'none';
160 
161         return this.domElement;
162     },

167     genericClusterIcon : function(pin){
168 
169         var div = document.createElement("div"); 
170         div.className = 'deCarta-Pin-Cluster-Default';
171         return div;

174     chooseIcon: function(){
175         
176         if (this.options.image){
177             return this.options.image;
178         }
179         
180         if (this.options.imageSrc){
181             var img = new Image();
182             img.src = this.options.imageSrc;
183             return img;            
184         }
185 
186         var candidate = this.cluster[0].getImage();
187         if(candidate){ // image based pin
188             for (var i = 1 ;i < this.cluster.length; i++){
189                 if (typeof candidate !== 'string') {
190                     if (this.cluster[i].getImage().src != candidate.src) {
191                         return this.genericClusterIcon(this.cluster[i]);
192                     }
193                 } else {
194                     if (this.cluster[i].getImage() != candidate) {
195                         return this.genericClusterIcon(this.cluster[i]);
196                     }
197                 }
198             }
199 
200             var img = new Image();
201             if (typeof candidate === 'string') {
202                 img.src = candidate;            
203             }
204             else {
205                 img.src = candidate.src;
206                 img.style.width = candidate.style.width;
207                 img.style.height = candidate.style.height;
208             }
209             this.options.xOffset = this.cluster[0].options.xOffset;
210             this.options.yOffset = this.cluster[0].options.yOffset;
211 
212         } else { // non image based pin
213            return this.genericClusterIcon(this.cluster[i]);
214         }
215         return img;
216     },

226     //infowindow stuf for the cluster. 
227     getBounds: function(){
228         var top = this.domY;
229         var left = this.domX;
230 
231         var styleTop = parseFloat(deCarta.Utilities.getStyle(this.imgElement, 'top')) || 0;
232         var styleLeft = parseFloat(deCarta.Utilities.getStyle(this.imgElement, 'left')) || 0;
233 
234         return {
235             top: top + parseFloat(styleTop),
236             left:  left + parseFloat(styleLeft),
237             right: left + parseFloat(styleLeft) + this.imgElement.offsetWidth,
238             bottom: top + parseFloat(styleTop) + this.imgElement.offsetHeight,
239             offsetX: left,
240             offsetY: top
241         };
242     },
243     
244     getAnchorPositions: function(){
245         var top = this.domY;
246         var left = this.domX;
247 
248         var styleTop = parseFloat(deCarta.Utilities.getStyle(this.imgElement, 'top')) || 0;
249         var styleLeft = parseFloat(deCarta.Utilities.getStyle(this.imgElement, 'left')) || 0;
250 
251         return {
252             left: (parseFloat(styleLeft) + parseFloat(this.imgElement.offsetWidth) / 2),            
253             right: (parseFloat(styleLeft) + parseFloat(this.imgElement.offsetWidth) / 2),
254             top: parseFloat(styleTop),
255             bottom: parseFloat(styleTop) + parseFloat(this.imgElement.offsetHeight)            
256         }
257     },
258 
259     showText: function(content, onClose, options){
260 
261         this.originalZ = this.domElement.style.zIndex;
262         this.domElement.style.zIndex = parseFloat(this.domElement.style.zIndex) + 100;
263 
264         var el = this.owner.owner.infoWindow.getElement();
265         this.domElement.appendChild(el);
266         this.owner.owner.infoWindow.show({
267             target: this,
268             content: content,
269             autoReorient: true,
270             onClose: function(){                
271                 this.domElement.style.zIndex = this.originalZ || 100;
272                 if (onClose) onClose(this);
273                 this.textVisible = false;
274                 this.domElement.style.zIndex = this.originalZ || 100;
275             }.bind(this)
276         });
277     },
278 
279     hideText: function(){
280         this.domElement.style.zIndex = this.originalZ || 100;
281         this.owner.owner.infoWindow.hide();
282     }
