 12     /**
 13      * @private
 14      */
 15     getXML: function(criteria){
 16         /* Example XML
 17          *
 18          *
 19         <?xml version="1.0"?>
 20         <xls:XLS xmlns:xls="http://www.opengis.net/xls" xmlns:gml="http://www.opengis.net/gml" version="1" xls:lang="en" rel="4.5.1.sp02">
 21           <xls:RequestHeader clientName="client_name" sessionID="8285826" clientPassword="client_password" configuration="global-decarta"/>
 22           <xls:Request maximumResponses="20" version="1.0" requestID="9132342" methodName="DirectoryRequest">
 23             <xls:DirectoryRequest database="search:decarta:tmo" sortDirection="Ascending" rankCriteria="Score">
 24               <xls:POILocation>
 25                 <xls:WithinDistance>
 26                   <xls:POI ID="1">
 27                     <gml:Point>
 28                       <gml:pos>37.774053 -122.421398</gml:pos>
 29                     </gml:Point>
 30                   </xls:POI>
 31                   <xls:MinimumDistance value="0" uom="KM"/>
 32                   <xls:MaximumDistance value="1.9613978800000003" uom="KM"/>
 33                 </xls:WithinDistance>
 34               </xls:POILocation>
 35               <xls:POIProperties>
 36                 <xls:POIProperty value="foobar" name="POIName"/>
 37               </xls:POIProperties>
 38             </xls:DirectoryRequest>
 39           </xls:Request>
 40         </xls:XLS>
 41          **/
 42 
 43 
 44         var database = 'database="' + criteria.database + '"';
 45         var sortDirection = (criteria.sortDirection) ? 'sortDirection="' + criteria.sortDirection + '"' : '';
 46         var rankCriteria = (criteria.rankCriteria) ? 'rankCriteria="' + criteria.rankCriteria + '"' : '';
 47 
 48         var searchXml = '';
 49 
 50         // backwards compatibility mode - if routeID is on top level (as was in v2)
 51         // push it into the new corridorParams holder
 52         if(criteria.routeId){
 53           criteria.corridorParams.routeId=criteria.routeId;
 54         }
 55 
 56         if (criteria.corridorParams.routeId){
 57             //corridor search
 58             searchXml += "<xls:NearRoute>";
 59             if(criteria.corridorParams.measure==="distance"){
 60                 searchXml += "<xls:distance value='"+criteria.corridorParams.measureValue+"' uom='M'/>";
 61             } else if(criteria.corridorParams.measure==="euclideanDistance"){
 62                 searchXml += "<xls:euclideanDistance value='"+criteria.corridorParams.measureValue+"' uom='M'/>";
 63             } else {
 64                 // TODO need to build in better
 65                 searchXml += "<xls:drivetime>P0DT"+Math.floor(criteria.corridorParams.measureValue/60)+"H"+(criteria.corridorParams.measureValue%60)+"M0S</xls:drivetime>";
 66             }
 67             searchXml += "<xls:RouteID>"+criteria.corridorParams.routeId+"</xls:RouteID>";
 68             searchXml += "</xls:NearRoute>";
 69         } else {
 70             //regular search
 71             searchXml = '<xls:WithinDistance>' +
 72               '<xls:POI ID="1">' +
 73                 '<gml:Point>' +
 74                   '<gml:pos>' + criteria.position.lat+ ' ' +criteria.position.lon + '</gml:pos>' +
 75                 '</gml:Point>' +
 76               '</xls:POI>' +
 77               '<xls:MinimumDistance value="0" uom="KM"/>' +
 78               '<xls:MaximumDistance value="' + criteria.radius.distance + '" uom="' + criteria.radius.uom + '"/>' +
 79             '</xls:WithinDistance>'
 80         }
 81         
 82         var top =
 83               '<xls:Request maximumResponses=\'' + criteria.maximumResponses+ '\' version="1.0" requestID="0123456" methodName="DirectoryRequest">' +
 84                   '<xls:DirectoryRequest ' + database + ' ' +  sortDirection + ' ' + rankCriteria + '>' +
 85                   '<xls:POILocation>' +
 86                     searchXml+
 87                   '</xls:POILocation>' +
 88                   '<xls:POIProperties>';
 89        var bottom = 
 90                    '</xls:POIProperties>' +
 91                 '</xls:DirectoryRequest>' +
 92               '</xls:Request>';
 93         var xml='';
 94 
 95         var crits = deCarta.Utilities.makeArrayFix(criteria.properties);
 96         for(var i=0;i<crits.length;i++){
 97             for (var name in crits[i]){                
 98                 if (typeof crits[i][name] == 'object'){
 99                     var s = ' ';
100                     for (var prop in crits[i][name]){
101                         s += prop + '="' + crits[i][name][prop] + '" ';                        
102                     }
103                     xml += top + '<xls:POIProperty name="' + name + '" '+ s +' />'+bottom;
104                 } else {
105                     xml+= top+'<xls:POIProperty value="' + crits[i][name] + '" name="' + name + '"/>'+bottom;
106                 }
107             }
108         }
109 
110         return xml;
111     },

113     /**
114      * Executes a POI search with the requested criteria
115      * @param {deCarta.Core.SearchCriteria} Search criteria to be used for the
116 	 * search. This can be either an instance of a {@link deCarta.Core.SearchCriteria},
117 	 * or an inline object with the same structure, as follows:
118      * <pre>
119      *  SearchCriteriaObject: {
120      *      queryString: null, //[Not yet defined]
121      *      maximumResponses: int, //Max number of responses that will be returned
122      *      database: string, //Database that will be queried for this search
123      *      sortDirection: string, //Sort order for results ["Ascending", "descending"]
124      *      sortCriteria: string, //Property results will be sorted by ["Distance", "Score"]
125      *      rankCriteria: string, //Property by which results will be ranked [Distance, Score]
126      *      allowAggregates: bool, //Allow Aggregates
127      *      retrieveAll: bool, //retreive all values
128      *      properties: { //Object describing search properties, contains ONE OF:
129      *        CATEGORY: string, //If present, this will perform a category search on this value
130      *      	KEYWORD: string, //If present, this will perform a Keyword search on this value
131      *      	POIName: string //If present, this will perform a freeform POI Search on this value
132      *      }
133      *      position: deCarta.Core.Position, //Position where the search is centered
134      *      radius: deCarta.Core.Radius, //Radius of the search
135      *      routeId: int, //if routeId is set, the search will be performed along the route. 
136      *      corridorType: //[Not yet defined]
137      *      map: deCarta.Core.Map //Optionally pass Map instead of position+radius
138      * </pre>
139      * @param {function} callback Function that will be called when the POI search
140 	 *   response is received from the DDS Web Service.
141 
142      * @param {function} callback A user defined function that will receive
143      * the respose to the query from the DDS Web Service. <br />
144      * Function signature: <br />
145      * function poiSearchResponseCallback(resultArray)<br />
146 	 * The callback function is passed a single object, which is an array of 
147 	 * objects, with each object in the array having the following
148 	 * structure:
149      * <pre>
150      *     
151   {
152     "ID":"search:decarta:nt:1001042664-US",
153     "name":"Extreme Pizza",
154     "position":deCarta.Core.Position,
155     "address":{
156       "locale":deCarta.Core.Locale,
157       "buildingNumber": int,
158       "landmark": string,
159       "street": string,
160       "streetNameAndNumber": string,
161       "speedLimit": string,
162       "countryCode": string,
163       "countrySubdivision": string,
164       "countrySecondarySubdivision": string,
165       "countryTertiarySubdivision": string,
166       "municipality": string,
167       "postalCode": string,
168       "municipalitySubdivision": string,
169       "type": string,
170       "freeFormAddress": string
171     },
172     "phoneNumber": string,
173     "distance":deCarta.Core.Distance,
174     "categories":[
175       "restaurant",
176       "pizza"
177     ]
178   }
179      * </pre>
180      * @return {int} requestId which can be used to canceling long running requests
181      */
182     execute: function(criteria, callback, getRaw){
183         
184         var defaults = new deCarta.Core.SearchCriteria();
185         criteria = deCarta.Utilities.extendObject(defaults, criteria);
186 
187         if(!callback) deCarta.Core.Exception.raise('You need to provide a search callback, or you will never know what was found');
188         if (criteria.map){
189             criteria.position = criteria.map.getCenter();
190             if(!criteria.radiusOverride)
191                 criteria.radius = criteria.map.getRadius();
192 //            delete criteria.map;
193         }
194 
195         if (! (criteria.position || criteria.corridorParams.routeId) ){
196             deCarta.Core.Exception.raise('A position or routeId is required to search. Please provide one in the search criteria');
197         }
198 //        if (!criteria.properties.POIName && !criteria.properties.CATEGORY){
199 //            deCarta.Core.Exception.raise('Can\'t search : Provide either a search term or a category');
200 //        }
201 
202         return deCarta.Core.JSRequest.send(this.getXML(criteria), this.handleResponse.bind(this,callback), this.handleFailure.bind(this, callback), getRaw);
203 
204     },

209     handleResponse: function(callback, response){        
210         var poiContextObject;
211         // multi
212         try {
213             if (deCarta.Utilities.isArray(response)){
214                 callback(response);
215                 return; 
216             }
217             if(deCarta.Utilities.isArray(response.XLS.Response)){
218 
219                 var pois = [];
220                 for(var i=0;i<response.XLS.Response.length;i++){
221                     poiContextObject = response.XLS.Response[i].DirectoryResponse.POIContext;
222                     if(poiContextObject){
223                         var poiContext = deCarta.Utilities.makeArrayFix(poiContextObject);
224                         pois = pois.concat(poiContext);
225                     }
226                 }
227                 callback(pois);
228             } else { // single
229                 poiContextObject = response.XLS.Response.DirectoryResponse.POIContext;
230                 if(poiContextObject){
231                     callback( deCarta.Utilities.makeArrayFix(response.XLS.Response.DirectoryResponse.POIContext));
232                 } else {
233                     callback([]);
234                 }
235             }            
236             callback(response);
237         } catch (e){
238             
239             callback([], {message: e.message, param: response});
240             //deCarta.Core.Exception.raise('Error executing search: ' + e.message);
241         }
242     },

244     /**
245 	 * @private
246 	 */
247     handleFailure: function(callback, param){
248         //deCarta.Core.Exception.raise('Error executing search: Timeout');
249         callback([], {param: param, message: 'Error executing search'});
250     },

252     responseInterceptor: function(response){
253         
254         var pois = [];
255 
256         if(deCarta.Utilities.isArray(response.XLS.Response)){                
257                 for(var i=0;i<response.XLS.Response.length;i++){
258                     poiContextObject = response.XLS.Response[i].DirectoryResponse.POIContext;
259                     if(poiContextObject){
260                         var poiContext = deCarta.Utilities.makeArrayFix(poiContextObject);
261                         pois = pois.concat(poiContext);
262                     }
263                 }                        
264         } else { // single
265                 poiContextObject = response.XLS.Response.DirectoryResponse.POIContext;
266                 if(poiContextObject){
267                     pois = deCarta.Utilities.makeArrayFix(response.XLS.Response.DirectoryResponse.POIContext);
268                 } else {
269                     pois = [];
270                 }
271         }
272 
273         var ret = [];
274         //Found all POIs. Now format them...
275         for (var i = 0; i < pois.length; i ++){
276                 var obj = {};
277                 var poi = pois[i];                                        
278                 obj.ID = poi.POI.ID;
279                 obj.name = poi.POI.POIName;
280                 obj.position = new deCarta.Core.Position(poi.POI.Point.pos);
281                 obj.address = {};//poi.POI.Address;
282                 obj.phoneNumber = poi.POI.phoneNumber;
283                 if(poi.Distance){
284                   if(poi.Distance.length){
285                     obj.distance = {
286                       onRoute : new deCarta.Core.Distance(poi.Distance[0].value, poi.Distance[0].uom),
287                       offRoute : new deCarta.Core.Distance(poi.Distance[1].value, poi.Distance[1].uom)
288                     }                    
289                   } else {
290                     obj.distance = new deCarta.Core.Distance(poi.Distance.value, poi.Distance.uom);
291                   }
292 
293                 }else if(poi.Duration){
294                     obj.duration = {
295                       onRoute : {time : poi.Duration[0].content},
296                       offRoute : {time : poi.Duration[1].content}
297                     }
298                 }                
299                 var addr = new deCarta.Core.StructuredAddress();
300                 addr.fromWSResponse(poi.POI.Address);
301                 obj.address = addr;
302                 //now the categories ...
303                 for (var j = 0; j < poi.POI.POIAttributeList.POIInfoList.POIInfo.length; j++){
304                         var name = poi.POI.POIAttributeList.POIInfoList.POIInfo[j].name;
305                         var value = poi.POI.POIAttributeList.POIInfoList.POIInfo[j].value;
306                         var test = name.split('-');
307                         if (test[0] == 'category'){
308                                 if (!obj.categories) obj.categories = [];
309                                 if (obj.categories.indexOf(value.toLowerCase())) obj.categories.push(value.toLowerCase());
310                         } 
311                 }
312                 ret.push(obj);
313         }                
314 
315         return ret;
316     }
