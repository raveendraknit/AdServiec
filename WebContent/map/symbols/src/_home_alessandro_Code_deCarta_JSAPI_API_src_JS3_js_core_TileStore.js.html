 52 deCarta.Core.TileStore.prototype  = {
 53 
 54     /**
 55 	 * @private
 56 	 */
 57     getKey : function(x,y,z){
 58 
 59         // added to key as uniquifier: config (image setting) and whether or not it is a traffic tile
 60 
 61         return deCarta.Utilities.getTileKey(x,y,z) + '-' + deCarta.Core.Configuration.defaultConfig+ 
 62         '-TDATA=' + deCarta.Core.Configuration.showTraffic;
 63     },
 64 
 65     /**
 66 	 * @private
 67 	 */
 68     getTile: function(x, y, z){       
 69    
 70         z = Math.floor(z);
 71 
 72         var key = this.getKey(x,y,z);
 73         /* This is when the tile is cached as a full dom element. we just pass that*/
 74         if (this.availableTiles[key] && this.availableTiles[key].img && this.availableTiles[key].img.src) {
 75             
 76             return this.availableTiles[key];
 77         }
 78         /* This is when the dom element has expired, but we have seen the tile before. 
 79          * In this case we reconstruct the image ( also add it to the availabletiles)*/
 80         /*if (this.browserCacheTiles[key]){
 81             var drawTile = {};
 82             var img = new Image();
 83             //img.crossOrigin = "anonymous";
 84             img.src = this.getTileUrl(x,y,z);             
 85             
 86             drawTile.name = key;
 87             drawTile.img = img;
 88             drawTile.scale = 1;
 89             drawTile.ofsX = 0;
 90             drawTile.ofsY = 0;
 91             
 92             this.availableTiles[key] = drawTile;
 93 
 94             return drawTile;                        
 95         }*/
 96         
 97         if (this.loadingTiles[key]) {
 98             return this.getNextBestTile(x,y,z);
 99         }
100         this.loadingTiles[key] = true;
101 
102         var tile = new Image();
103         //tile.crossOrigin = "anonymous"; //crashes the map if not in webgl mode, why?
104         tile.key = key;        
105         tile.tileStoreId = this.tileStoreId;
106         tile.timeout = setTimeout(this.purgeRequest.bind(this, key), this.tileTimeout);
107         tile.onload = function(t){
108             if (this.tileStoreId != t.tileStoreId) {
109                 return;
110             }
111             clearTimeout(t.timeout);
112             var now = new Date().getTime();
113             t.loadTime = now;
114             //this.availableTiles[key] = t;
115             
116             var drawTile = {};
117             drawTile.name = key;
118             drawTile.img = t;//this.availableTiles[key];
119             drawTile.scale = 1;
120             drawTile.ofsX = 0;
121             drawTile.ofsY = 0;
122             
123             this.availableTiles[key] = drawTile;
124             
125             
126             this.keyStack.push(key);
127             this.tileCount ++;
128             delete this.loadingTiles[key];
129             delete this.loadRequests[key];
130             this.removeBlankTile(key);
131             t.onload = null;
132             if (this.tileCount > this.tileCacheSize){
133                 var k = this.keyStack.shift(key);
134                 //this.browserCacheTiles[k] = true;
135                 delete this.availableTiles[k];
136                 this.tileCount --;                
137             }	
138             deCarta.Core.EventManager.trigger('tileLoad', t);            
139 //            deCarta.Core.EventManager.trigger('tileLoad', {tileObj : t}, t);
140         }.bind(this, tile);
141         tile.onerror = function(e){            
142             this.purgeRequest(this.key);           
143         }.bind(this)
144         this.loadRequests[key] = tile;        
145         
146         tile.src = this.getTileUrl(x,y,z);        
147         
148         return this.getNextBestTile(x,y,z);
149        
150     },
151 
152     /**
153      * @private
154      * purges the load queue based on a center and a zoom level
155      * basically goes through the load list and purges tiles that are
156      * on different zoom or that are far away from the position.
157      * radius is IN TILES
158      */
159     purgeLoadRequests: function(centerTile, zoom, radius){
160         
161         zoom = deCarta.Utilities.normalizeZoom(zoom);        
162         for (var key in this.loadRequests){
163             var tInfo = deCarta.Utilities.splitTileKey(key);
164             if (tInfo.z != zoom){
165                 continue;
166              /*   console.log('Purging : ' + key + ' since zoom is ' + zoom);
167                 this.purgeRequest(key);*/
168             }
169             //now check the distance of the tile
170             var d = Math.floor(Math.sqrt(Math.pow(centerTile.E - tInfo.x,2) + Math.pow(centerTile.N - tInfo.y,2)));            
171             if (d > (radius / 2) +1){
172                // console.log('Purging: ' + key + ' its out of radius. ', d, radius);
173                 this.purgeRequest(key);
174             }
175         }
176     },
177 
178     /**
179 	 * @private
180 	 */
181     purgeRequest: function(key){        
182         if (this.loadRequests[key]) {
183             this.loadRequests[key].onload = null;
184             this.loadRequests[key].src = null;
185             delete this.loadRequests[key];
186         }
187         if (this.loadingTiles[key]) delete this.loadingTiles[key];        
188     },
189 
190     /**
191 	 * @private
192 	 */
193     zoomEnd: function(ev){
194         
195         this.currentZoom = ev.zoom;
196         //return; //this is buggy, fix it then renable it.
197         for (var k in this.loadRequests){
198             var info = deCarta.Utilities.splitTileKey(k);
199             if (info.z != this.currentZoom){
200                 if (!this.availableTiles[k]){
201                     this.purgeRequest(k);
202                 }
203             }
204         }
205     },

214     /**
215 	 * @private
216 	 */
217     getNextBestTile: function(x,y,z){
218         //looks for the tile at a lower zoom level, returns it
219 
220         z = Math.round(z);
221         for (var i = 1; i < 6; i++){
222             var nz = z - i;
223 
224             var nx = x / Math.pow(2, i);
225             var ny = y / Math.pow(2, i);
226 
227             if (nx < 0) nx = Math.floor(nx); else nx = Math.floor(nx);
228             if (ny < 0) ny = Math.floor(ny); else ny = Math.floor(ny);
229 
230             var tx = nx * Math.pow(2, i);
231             var ty = ny * Math.pow(2, i);
232 
233             var ox = tx - x;
234             var oy = ty - y;
235 
236             var key = this.getKey (nx,ny,nz);
237          
238             if (this.availableTiles[key]) {
239 
240                 var nTiles = Math.pow(2, i);
241                 oy = Math.abs(oy) - (nTiles - 1);// - oy;
242 
243                 var drawTile = {};
244                 drawTile.name = key;
245                 drawTile.img = this.availableTiles[key].img;
246                 drawTile.scale = Math.pow(2, i); //if we went up 2 zoom levels, tile is x4 each side, and so on.
247                 drawTile.ofsX = ox; //likely wrong.
248                 drawTile.ofsY = oy;
249 
250                 return drawTile;
251             }
252         }
253 
254         var key = this.getKey (x,y,z);
255         drawTile = {};
256         drawTile.name = 'empty_' + key;
257         drawTile.img = this.getBlankTile(key);
258         drawTile.img.empty = true;
259         drawTile.scale = 1;
260         drawTile.ofsX = 0;
261         drawTile.ofsY = 0;
262 
263         return drawTile;
264     },
265 
266     /**
267 	 * @private
268 	 */
269     getBlankTile: function(k){
270         if (this.blankTiles[k]) return this.blankTiles[k];
271         this.blankTiles[k] = this.blankTileImg.cloneNode(true);
272         this.blankTiles[k].id = 'empty_' + k;
273         this.blankTiles[k].key = 'empty_' + k;
274         return this.blankTiles[k];
275     },
276 
277     /**
278 	 * @private
279 	 */
280     removeBlankTile: function(k){
281         if (this.blankTiles[k]){
282             delete this.blankTiles[k];
283         }
284     },

291     startPrecaching: function(){
292         //console.log('Starting to precache', this.opts.precache);
293         if (!deCarta.Core.Configuration.usePrecaching) return;
294         if (!this.opts.precache) return;
295         if (!this.precacher){               
296             this.precacher = new deCarta.Core.TilePrecacher(this.owner, this);            
297         }
