1 /**
  2  * @class
  3  * the EventableObject class is a base class which can be extended to 
  4  * create objects which respond to user input. 
  5  * All OverlayObjects and the Map objet extend this class.
  6  * This class installs all necessary event listeners, and also creates all shortcut
  7  * event methods on the object (ex, object.onclick())
  8  * The class provides a set of standard methods for event handling, which
  9  * the subclass will have to override if it needs to provide specific
 10  * functionality.
 11  * 
 12  * @see deCarta.Core.OverlayObject
 13  * @see deCarta.Core.Map
 14  */
 15 deCarta.Core.EventableObject = function(){
 16     this.objectId = 'dCObject-' + (Math.random() + "").replace(".", "");
 17     this.hovered = false;
 18     this.createEventHooks();
 19 }
 20 
 21 deCarta.Core.EventableObject.prototype = {
 22 	
 23 	createEventHooks: function(){
 24 		var events = deCarta.Core.EventManager.eventNames;
 25         for (var event in events){
 26 
 27             var mapProp = 'on' + event;
 28             
 29             this[mapProp] = function(event, callback){
 30                 var listenerId = deCarta.Core.EventManager.listen(event, callback, this);
 31                 return function(){
 32                     deCarta.Core.EventManager.stopListeningByIdx(event, listenerId);
 33                 }
 34             }.bind(this, event);
 35         }
 36 	},
 37 
 38 	on: function(eventList, callback){
 39 		var events = eventList.split(' ');
 40 		for (var i = 0; i < events.length; i++){
 41 
 42 			var mapProp = 'on' + events[i];
 43 
 44 			if (typeof this[mapProp] === 'function'){
 45 				this[mapProp](callback);
 46 			}
 47 		}
 48 	},
 49 
 50 	trigger: function(event, params){
 51 		deCarta.Core.EventManager.trigger(event, params, this);
 52 	},
 53 
 54 	addListeners: function(element, opaque){
 55 		
 56 	    deCarta.Touch.attachListener('touchstart', element, this.touchStart.bind(this), opaque);
 57         deCarta.Touch.attachListener('touchmove', this.owner.owner.touchLayer, this.touchMove.bind(this), opaque);
 58         deCarta.Touch.attachListener('touchend', this.owner.owner.touchLayer, this.touchEnd.bind(this), opaque);        
 59         deCarta.Touch.attachListener('touchout', element, this.touchOut.bind(this), opaque);
 60         deCarta.Touch.attachListener('touchover', element, this.touchOver.bind(this), opaque);
 61         deCarta.Touch.attachListener('doubleTap', element, this.doubleTap.bind(this), opaque);
 62         deCarta.Touch.attachListener('longTouch', element, this.longTouch.bind(this), opaque);
 63         deCarta.Touch.attachListener('tap', element, this.tap.bind(this), opaque);
 64         deCarta.Touch.attachListener('altTap', element, this.altTap.bind(this), opaque);
 65 	},
 66 
 67 	touchStart: function(e, oe){
 68 		
 69 	},
 70 
 71 	touchMove: function(e, oe){		
 72 		
 73 	},
 74 
 75 	touchEnd: function(e, oe){
 76 		
 77 	},
 78 
 79 	touchOut: function(e, oe){				
 80 		if (!this.hovered || this.dragging) return;		
 81 		this.triggerEvent('hoveroff', e, oe, true);
 82 		this.hovered = false;
 83 	},
 84 
 85 	touchOver: function(e, oe){	
 86 		if (this.hovered || this.dragging) return;
 87 		this.hovered = true;	
 88 		this.triggerEvent('hoveron', e, oe);
 89 	},
 90 
 91 	shouldProcessEvent: function(position){
 92 		return true;
 93 	},
 94         
 95 	doubleTap: function(e, oe){		                
 96         this.triggerEvent('doubleclick', e, oe);
 97 	},
 98 
 99 	longTouch: function(e, oe){			
100 		this.triggerEvent('longtouch', e, oe);
101 	},
102 
103 	tap: function(e, oe){			
104 		this.triggerEvent('click', e, oe);
105 	},
106 
107 	altTap: function(e, oe){		
108 		this.triggerEvent('rightclick', e, oe);
109 	},
110 
111     checkEvent: function(e, pos){
112     	
113     	if (this.dragging) return true;
114     	
115         if (!e) return false;
116         
117         var x = e.pageX - this.owner.owner.domX - this.domX;
118         var y = e.pageY - this.owner.owner.domY - this.domY;  
119 		
120         return this.shouldProcessEvent(pos, x, y);
121     },
122 
123     triggerEvent: function(which, e, oe, skipPositionCheck){ 
124     	
125     	var mapPosition = deCarta.Utilities.domPosition(this.owner.owner.canvas);
126         var x = e.pageX - mapPosition.left;
127         var y = e.pageY - mapPosition.top;      	
128 		var eventPosition = this.owner.owner.positionFromXY(x, y);
129 
130         if (!skipPositionCheck) if (!this.checkEvent(e, eventPosition)) return;
131         var event = new deCarta.Core.Event({
132         	DOMevent: oe,
133         	touchEvent: e,
134         	position: eventPosition
135         });        
136 		deCarta.Core.EventManager.trigger(which, event, this);
137     }
138 
139 }
