1  /**
  2  * @class
  3  * Polyline is a map overlay object used to draw a complex line
  4  * (such as a route) on the map.
  5  * It extends {@link deCarta.Core.OverlayObject}.
  6  * Use the {@link deCarta.Core.MapOverlay}:addObject() method to add 
  7  * this object to a map overlay.
  8  * (for example a route).
  9  *
 10  * @constructor
 11  * @param {object} opts Options. May contain one or more of the following
 12  * <ul>
 13  *  <li>lineGeometry: Array of deCarta.Mobile.Position</li>
 14  *  <li>fillColor: A polyline has thickness, and therefore has a fill color
 15  *      - optional, default '#000'</li>
 16  *  <li>strokeColor: Color hat will be used to stroke the line
 17  *      - optional, default '#0000FF'</li>
 18  *  <li>strokeSize: Size of the line, in pixels - optional, default 6</li>
 19  *  <li>strokeOpacity: Opacity of the line (1.0 = opaque, 0.0 = transparent)
 20  *      - optional, default 0.6</li>
 21  *  <li>strokeLineCap: <b>NOTE: SVG Rendering only</b> Style for the line endings. Can be one of : "butt", "round", "square" (see http://www.w3.org/TR/SVGTiny12/painting.html)</li>
 22  *  <li>strokeLineJoin: <b>NOTE: SVG Rendering only</b> Shape which shall be used at the corners of shapes when they are stroked. Can be one of: "miter", "round, "bevel". (see http://www.w3.org/TR/SVGTiny12/painting.html)</li> 
 23  * </ul>
 24  *
 25  * @see deCarta.Core.OverlayObject
 26  * @see deCarta.Core.MapOverlay
 27  */

 30 deCarta.Core.Polyline = function(options){
 31     opts = {
 32         lineGeometry: null,        
 33         strokeWidth: 6,
 34         strokeLineCap: "round",
 35         strokeLineJoin: "round",
 36         strokeColor: "#0000FF",
 37         strokeOpacity: 0.6,    
 38         granularity: 7 //not documented
 39     }
 40 
 41     options = deCarta.Utilities.extendObject(opts, options);
 42 
 43     deCarta.Core.Shape.call(this, options);
 44     //check if lineGeomtry is an array of pos
 45     if (!options.lineGeometry || options.lineGeometry.length == 0)
 46         deCarta.Core.Exception.raise('You need to provide a lineGeometry to instantiate a polyline');
 47 
 48     this.pointIndex = [];
 49     this.genGeom = []; //array of generalized geometry, per zoom level 
 50 
 51 
 52     this.segmentIndex = {};
 53 
 54     this.positions = [];
 55 
 56     this.lastRenderedPointSet = null;
 57     
 58     if (typeof this.options.lineGeometry[0] == 'object' && this.options.lineGeometry[0].getLat){
 59         //these are positionS!!!
 60 
 61         this.positions = this.options.lineGeometry.slice();
 62         
 63     } else {
 64         for(var i = 0; i< this.options.lineGeometry.length; i++){
 65             var ll = this.options.lineGeometry[i].split(' ');
 66             var tPos = new deCarta.Core.Position(parseFloat(ll[0]), parseFloat(ll[1]));
 67             this.positions.push(tPos);
 68         }
 69     }
 70 
 71     this.type = 'shape';
 72 
 73     this.padding = 5;
 74 
 75     this.findBounds();
 76         
 77     this.boundingBox = new deCarta.Core.BoundingBox(this.positions);
 78     
 79 }

345      * @private

347     indexAndThinPoints: function(z){
348     try {
349         //Check the point index
350         var nz = deCarta.Utilities.normalizeZoom(z);
351         var posLength = this.positions.length;
352         var lastPoint = null;
353         var tx = 0;
354         var ty = 0;
355         var distance = null;
356         if (!this.segmentIndex[nz]){   
357             this.segmentIndex[nz] = {};
358            /*
359              the new point index!
360 
361              using edges this time. 
362              each edge is indexed based on all tiles that it crosses
363              when preparing points, we get the affected edges from the edge index,
364              and create a list of instructions - move to, line to depending on if they are connected or not. 
365              this is passed to the render function which then, yes, renders it. 
366 
367            */
368             var segCount = 0;
369             for (var i = 0; i < posLength; i++){
370 
371                 var p = this.positions[i];
372 
373                 tx = p.getX(z);
374                 ty = p.getY(z);
375 
376                 if (lastPoint){
377                     distance = Math.sqrt(Math.pow(tx - lastPoint.x, 2) + Math.pow(ty - lastPoint.y, 2));                                
378                 }
379 
380                 if (distance > this.options.granularity){
381                     var lastPoint = {x: tx, y: ty};
382                     
383                     var segment = new deCarta.Core.Segment(lastPosition, p);
384                     //now index this segment as needed
385                     var tiles = segment.getCrossedTiles(nz);
386                     segment.idx = segCount;
387                     segCount ++;
388                     for (var j = 0; j < tiles.length; j++){
389                         if (!this.segmentIndex[nz][tiles[j]]) this.segmentIndex[nz][tiles[j]] = []
390                         this.segmentIndex[nz][tiles[j]].push(segment);
391                     }
392                     var lastPosition = p;
393                 }
394 
395                 if (!lastPoint) {
396                      var lastPoint = {x: tx, y: ty};
397                      var lastPosition = p;
398                 }
399             }
400         }
401     } catch (e){
402         console.log(e.message);
403     }

409     preparePoints: function(z, dX, dY, tiles){            
410 
411         var points = [];
412         var nz = deCarta.Utilities.normalizeZoom(z);
413 
414         var p = null;
415         var px = null;
416         var py = null;
417         var addedPoints = {};
418 
419         //lets start by returning the right stuff like that
420         //then we split them up and return structs
421 
422         var segs = [];
423         var uniques = {};
424         var throwed =0;
425 
426        // debugger;
427 
428     
429 
430         for (var i = 0; i< tiles.length; i++){
431             var key = deCarta.Utilities.getTileKey(tiles[i].N, tiles[i].E, nz);
432             
433             if (this.segmentIndex[nz][key]){   
434                 //console.log(key, this.segmentIndex[nz][key])             
435                 for (var j = 0; j < this.segmentIndex[nz][key].length; j++){
436                     var seg = this.segmentIndex[nz][key][j];                    
437                     if (!uniques[seg.idx]){
438                         segs.push(seg);
439                         uniques[seg.idx] = true;
440                     }    else {
441                         throwed ++
442                     }
443                 }
444             }
445         }
446 
447 
448         segs.sort(function (a, b){ return a.idx - b.idx });
449 
450         //console.log(segs.length, throwed);
451 
452         //now for the points 
453         var last = null;
454         var points = [];
455         var lastIdx = -100;
456         for (var i = 0; i < segs.length; i ++){
457             var px=0;var py=0
458             var seg = segs[i];        
459                         
460             px = seg.a.getX(z) - dX + this.padding;                    
461             py = dY - seg.a.getY(z) + this.padding;
462 
463             if (lastIdx + 1 < seg.idx){
464                 points.push({x: px, y: py, i: seg.idx, op: 'm'})
465             }
466             
467             points.push({x: px, y: py, i: seg.idx, op: 'l'});
468         
469 
470             px = seg.b.getX(z) - dX + this.padding;                    
471             py = dY - seg.b.getY(z) + this.padding;
472             
473             points.push({x: px, y: py, i: seg.idx, op: 'l'}); 
474 
475             last = seg.b;
476             lastIdx = seg.idx;
477         }
478 
479         //console.log(points);
480     
481         return points;

563     //get the tiles it crosses by zoom level
564     getCrossedTiles: function(z){
565 
566         z = deCarta.Utilities.normalizeZoom(z);
567 
568         var aTile = this.a.getTileAtZoom(z);
569         var bTile = this.b.getTileAtZoom(z);
570 
571         var aKey = deCarta.Utilities.getTileKey(aTile.N, aTile.E, z);
572         var bKey = deCarta.Utilities.getTileKey(bTile.N, bTile.E, z);
573 
574         if (aKey == bKey) return [aKey];
575 
576         //if we are here, well. Figure out the affected tiles and return them.
577 
578         var x0 = Math.min(aTile.E, bTile.E);
579         var x1 = Math.max(aTile.E, bTile.E)
580 
581         if (x0 == aTile.E){
582             y0 = aTile.N;
583             y1 = bTile.N;
584         } else {
585             y1 = aTile.N;
586             y0 = bTile.N;
587         }
588         /*var y0 = Math.min(aTile.N, bTile.N);
589         var y1 = Math.max(aTile.N, bTile.N);*/
590 
591         var dx = Math.abs(x1-x0);
592         var dy = Math.abs(y1-y0);
593         var sx = (x0 < x1) ? 1 : -1;
594         var sy = (y0 < y1) ? 1 : -1;
595         var err = dx-dy;
596 
597         var tl = {};
598         var ret = [];
599 
600         while(true){
601             
602             var eK = deCarta.Utilities.getTileKey(y0, x0, z);            
603             if (!tl[eK]){
604                 tl[eK] = true;
605                 ret.push(eK);
606             }
607 
608             if ((x0==x1) && (y0==y1)) break;
609 
610             var e2 = 2*err;
611 
612             if (e2>-dy){
613                 err -= dy;
614                 x0  += sx;
615             } else 
616             if (e2 < dx){
617                 err += dx;
618                 y0  += sy;
619             }
620         }
621                 
622         return ret;
