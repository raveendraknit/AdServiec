1 /**
  2  * @class
  3  *
  4  * Polygon is a map overlay object used to draw a polygon
  5  * on the map.
  6  * It extends {@link deCarta.Core.OverlayObject}.
  7  * Use the {@link deCarta.Core.MapOverlay}:addObject() method to add 
  8  * this object to a map overlay.
  9  *
 10  * @description An OverlayObject used to display a polygon on a MapOverlay
 11  *
 12  * @constructor
 13  * @param {object} options Options. May contain one or more of the following:
 14  * <ul>
 15  *  <li>(array of {@link deCarta.Core.Position}) vertices: Array of {@link deCarta.Core.Position}s comprising the vertices of the polygon</li>
 16  *  <li>(string) strokeColor: Color that will be used to stroke the line
 17  *      - optional, default '#0077D2'</li>
 18  *  <li>(int) strokeWidth: Size of the polygon border, in pixels - optional, default 2</li>
 19  *  <li>(string) fillColor: The fill color for the interior of the polygon
 20  *      - optional, default '#000'</li>
 21  *  <li>(float) opacity: Opacity of the shape (1.0 = opaque, 0.0 = transparent)
 22  *      - optional, default 0.6</li> 
 23  * </ul>
 24  *
 25  * @see deCarta.Core.OverlayObject
 26  * @see deCarta.Core.MapOverlay
 27  */
 28 
 29 deCarta.Core.Polygon = function(options){
 30     deCarta.Core.Shape.call(this, options);
 31     
 32     opts = {
 33         draggable: false,
 34         scroll: null
 35     }
 36 
 37     this.options = deCarta.Utilities.extendObject(opts, this.options);
 38 
 39     this.lastRenderedZ = 0;    
 40 
 41     if (this.options.vertices.length < 3)
 42         deCarta.Core.Exception.raise('Instantiating a polygon with too few vertices. I believe you need at least 3.');
 43     
 44     this.boundingBox = new deCarta.Core.BoundingBox(this.options.vertices);    
 45     
 46     this.getPosition();
 47 }
 48

 50 //Define methods to extend Polygon
 51 deCarta.Core.Polygon.prototype = {
 52     
 53     /**
 54      * @private 
 55      */
 56     getPosition: function(){
 57         //if (this.position) return this.position;
 58         
 59         var lat = null;
 60         var lon = null;
 61         
 62         for (var i = 0; i < this.options.vertices.length; i++){
 63             
 64             var v = this.options.vertices[i];
 65             if (!lat || v.lat > lat) lat = v.lat;
 66             if (!lon || v.lon < lon) lon = v.lon;
 67             
 68         }
 69         
 70         this.position = new deCarta.Core.Position(lat,lon);
 71         
 72         
 73         return this.position;
 74     },
 75 
 76     /**
 77     * Returns a GML description of the shape. 
 78     * 
 79     *
 80     */    
 81     getGML: function(){
 82         var GML = '';
 83         GML += "<gml:Polygon><gml:exterior><gml:LinearRing>";
 84         for ( i = 0; i < this.options.vertices.length; i++ ) {
 85             GML += "<gml:pos>"+this.options.vertices[i]+"</gml:pos>"
 86         }
 87         GML += "</gml:LinearRing></gml:exterior></gml:Polygon>";
 88         return GML;
 89     },
 90     
 91     /**
 92      * @private 
 93      */
 94     setPosition: function(p){
 95         
 96         var oPx = this.position.getPixelPoint();
 97         var nPx = p.getPixelPoint();
 98         
 99         var deltaX = oPx.x - nPx.x;
100         var deltaY = oPx.y - nPx.y;               
101         
102         for (var i = 0; i < this.options.vertices.length; i++){
103             var v = this.options.vertices[i];
104             var vPx = v.getPixelPoint();
105             vPx.x = vPx.x - deltaX;
106             vPx.y = vPx.y - deltaY;
107             v.setXY(vPx.x, vPx.y);
108         }
109         
110         if (this.boundary){
111             this.boundary.fromPoly(this);
112         }
113 
114         this.position = p;        
115     },
116     
117     /**
118      * @private 
119      */
120     getCenter: function(){
121         
122         var pos = this.getPosition().getPixelPoint();
123         var size = this.getSize(21);
124         var x = pos.x + (size.width / 2);
125         var y = pos.y - (size.height / 2);
126         var center = new deCarta.Core.Position(0,0);
127         center.setXY(x,y,21);
128         return center;        
129     },
130     
131     /**
132      * @private 
133      */
134     getSize: function(z){        
135         return this.boundingBox.getSize(z);
136     },    
137 
138     getBoundingBox: function(){
139         return this.boundingBox;
140     },
141     
142     /**
143      * @private 
144      */
145     setVertices: function(verts){
146         this.options.vertices = verts;
147         this.boundingBox = new deCarta.Core.BoundingBox(verts);
148     },
149 
150 
151     /**
152     * Returns true if the position passed to the function is within the polygon. 
153     * @param pos: the position to check. 
154     * NOTE: You can also pass an OverlayObject to this function, and the position will be extracted automatically
155     */
156     contains: function(pos){
157         //console.log('Contains:', this.objectIid)
158         if (!pos.getLon) {
159             if (pos.getCenter){
160                 pos = pos.getCenter();
161             } else {
162                 throw new deCarta.Core.Exception('The position passed to the "contains" function is not a position.');
163             }
164         }
165         if (!this.boundary){
166             this.boundary = new deCarta.Core.MapBoundary();
167             this.boundary.fromPoly(this);
168         }
169         return this.boundary.checkPosition(pos);
170     },
171 
172     shouldProcessEvent: function(position){
173         return this.contains(position);
174     },
175     
176     /**
177      * @private 
178      */
179     renderCanvas: function(x, y, z, tiles){                
180         
181         
182         if (!this.canvasElement){            
183             this.canvasElement = document.createElement('canvas');
184 
185             this.canvasElement.setAttribute("id", ("deCarta-shape-" + Math.random()).replace(".", ""));
186             this.canvasElement.className = 'deCarta-polygon';
187             this.canvasElement.style.zIndex = this.zIndex;
188             this.canvasElement.style.position = 'absolute';            
189 
190             return this.canvasElement;
191         }
192 
193         if (!this.canvasElement.getContext){
194             try {
195                 G_vmlCanvasManager.initElement(this.canvasElement);
196             } catch (e){
197                 deCarta.Core.Exception.raise('Error in drawing the shape - have you added exCanvas to your page?');
198             }
199         }        
200 
201  
202         var size = this.getSize(z);
203         
204         this.domPosition((x - this.padding), (y - this.padding), this.canvasElement);
205 
206         if (z == this.lastRenderedZ) {
207             //no need to rerender, just reposition - for some reason this breaks should check it
208            // return this.canvasElement;
209         }
210 
211         /* Now render the stuff whatever */
212         
213         var newW = size.width + (this.padding * 2);
214         var newH = size.height + (this.padding * 2);
215         
216         if (newW != this.canvasElement.width || newH != this.canvasElement.height) {
217 
218             this.canvasElement.width = newW;
219             this.canvasElement.height = newH;  
220 
221             var ctx = this.canvasElement.getContext("2d"); 
222 
223             ctx.strokeStyle = deCarta.Utilities.makeRGBA(this.options.strokeColor, this.options.strokeOpacity);
224             ctx.lineWidth = this.options.strokeWidth;
225             ctx.fillStyle = deCarta.Utilities.makeRGBA(this.options.fillColor, this.options.fillOpacity);            
226 
227             ctx.beginPath();
228 
229 
230             var absPos = this.getPosition().getPixelPoint(z);
231 
232             for (var i = 0; i < this.options.vertices.length; i++){
233                 var v = this.options.vertices[i];
234                 var pos = v.getPixelPoint(z);
235 
236                 var dx = pos.x - absPos.x  + this.padding;
237                 var dy =  absPos.y - pos.y + this.padding;
238 
239                 if (i == 0) ctx.moveTo(dx, dy);
240                 else ctx.lineTo(dx, dy);            
241 
242             }
243             ctx.closePath();
244             ctx.fill();
245             ctx.stroke();
246         }
247         this.lastRenderedZ = z;
248         return this.canvasElement;
249     },
250     
251     renderVML: function(x, y, z){
252         if (!this.container){
253             this.container = document.createElement('div');
254             this.container.id = ("deCarta-polygon-container-" + Math.random()).replace(".", "");            
255                        
256             this.container.style.zIndex = this.zIndex;            
257             this.container.style.position = 'absolute';
258             
259             return this.container;
260         }
261         
262         var absPos = this.getPosition().getPixelPoint(z);
263         var size = this.getSize(z);
264                         
265         var newW = size.width + (this.padding * 2);
266         var newH = size.height + (this.padding * 2);                        
267         
268         var vStr = 'm ';
269         
270         for (var i = 0; i < this.options.vertices.length; i++){
271             var v = this.options.vertices[i];
272             var pos = v.getPixelPoint(z);
273 
274             var dx = pos.x - absPos.x  + this.padding;
275             var dy =  absPos.y - pos.y + this.padding;
276                         
277             if (i == 1 ) vStr += 'l ';
278             vStr += Math.round(dx) + ',' + Math.round(dy) + ' ';
279         }
280         vStr += 'x e';           
281         
282         var fillStr = '<v:fill opacity="' + Math.round(this.options.fillOpacity * 100) + '%" color="'+this.options.fillColor+'"/>'
283         var strokeStr = '<v:stroke opacity="' + Math.round(this.options.strokeOpacity * 100) + '%" weight="'+ this.options.strokeWidth+'px" color="'+this.options.strokeColor+'" />';
284         var vmldata = '<v:shape coordsize="'+ newW +',' + newH + '" style="position:absolute;width:'+ newW + 'px;height:' + newH + 'px;" path="' + vStr + '"> ' + fillStr + strokeStr + '</v:shape>';
285 
286         this.domPosition((x - this.padding), (y - this.padding), this.container);
287         this.container.innerHTML = vmldata;
288         
289         return this.container;
290     },
291     
292     /**
293      * @private 
294      */
295     renderSVG: function(x, y, z, tiles){
296                 
297         
298         if (!this.svgElement) {        
299             
300             this.container = document.createElement('div');
301             this.container.id = ("deCarta-polygon-container-" + Math.random()).replace(".", "");            
302                         
303             this.svgElement = document.createElementNS("http://www.w3.org/2000/svg", "svg");
304             this.svgElement.setAttribute("version", "1.1");                
305 
306             this.svgElement.setAttribute("id", ("deCarta-polygon-svg-" + Math.random()).replace(".", ""));
307            
308             this.container.style.zIndex = this.zIndex;            
309             this.container.style.position = 'absolute';
310             
311             this.container.appendChild(this.svgElement); 
312             return this.container;
313         }        
314         
315         if (!this.polyElement) {
316             
317             this.polyElement = document.createElementNS("http://www.w3.org/2000/svg", "polygon");            
318             this.svgElement.appendChild(this.polyElement);            
319         }        
320         var absPos = this.getPosition().getPixelPoint(z);
321         
322         var size = this.getSize(z);
323                         
324         var newW = size.width + (this.padding * 2);
325         var newH = size.height + (this.padding * 2);                        
326         
327         var pStr = '';
328         
329         
330         for (var i = 0; i < this.options.vertices.length; i++){
331             var v = this.options.vertices[i];
332             var pos = v.getPixelPoint(z);
333 
334             var dx = pos.x - absPos.x  + this.padding;
335             var dy =  absPos.y - pos.y + this.padding;
336             
337             pStr += dx + ',' + dy + ' ';
338         }
339         
340         this.polyElement.setAttribute('points', pStr);
341         this.polyElement.setAttribute('fill-opacity', this.options.fillOpacity);
342         this.polyElement.setAttribute('stroke-opacity', this.options.strokeOpacity);        
343         
344         this.polyElement.setAttribute('style', "fill:"+this.options.fillColor+";stroke:"+this.options.strokeColor+";stroke-width:"+this.options.strokeWidth+";position: absolute;");
345         
346         //this.domPosition((x - 5),(y - 5), this.svgElement);
347         this.container.style.width = newW + 'px';
348         this.container.style.height = newH + 'px';
349         this.domPosition((x - this.padding), (y - this.padding), this.container);
350 
351         return this.container;      
352     }
