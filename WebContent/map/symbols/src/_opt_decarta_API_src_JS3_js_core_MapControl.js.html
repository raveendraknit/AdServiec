1 /**
  2  * @class
  3  * A MapControl is a base class upon which to build custom map controls
  4  * This base class should be extended to provide the desired functionality
  5  * in your control.
  6  *
  7  * Some the following for some already-defined map controls:
  8  * {@link deCarta.UI.CopyrightControl}
  9  * {@link deCarta.UI.ZoomControl}
 10  *
 11  * @description Base Class for Map Controls
 12  *
 13  * @constructor
 14  * @param {object} options Options A list of options for the control, which will vary 
 15  * depending on the functionality the control offers. However, there is a set
 16  * of standard options every control should implement:
 17  * <ul>
 18  *  <li>(string) position: specifies how the control will be positioned on the map. can
 19  *  be one of 'topLeft', 'topRight', 'bottomLeft', 'bottomRight' or any other
 20  *  <li>(function) layout: You can also specify a layout function, given a map size (width, height) 
 21  * and a list of existing map controls returns 
 22  * an object in the format {top: int, left: int}, specifying where the control should be placed. 
 23  * If specified, the result of the layout function overrides the position attribute. </li>
 24  * </ul>
 25  *
 26  * @see deCarta.UI.CopyrightControl
 27  * @see deCarta.UI.ZoomControl
 28  * @see deCarta.Core.Map
 29  **/
 30 
 31 deCarta.UI.MapControl = function(options){
 32 
 33     this.options = {
 34         position: null, //position on the map : topLeft, topRight, bottomLeft, bottomRight, center, custom.
 35         marginX: 0, //margins from edge of map
 36         marginY: 0,
 37         layout: function(){                                    
 38             return {top: 0, left: 0}
 39         }
 40     }
 41 
 42     this.options = deCarta.Utilities.extendObject(this.options, options);
 43     
 44     this.id = 'deCarta-control-' + Math.random().toString().replace('.','');
 45     
 46     this.domElement = null;
 47 }
 48

 50 deCarta.UI.MapControl.prototype = {
 51     /**
 52 	 * You need to implement the render method in every map control so that
 53      * it produces a single HTML Dom Element containing the whole GUI for the control.
 54      * The render function should return this DOM Element to the caller so that it can
 55      * be placed on the map.
 56      *
 57      * The dom element should be created with the following css attributes :
 58      * <ul>
 59      *  <li>Position: absolute</li>
 60      *  <li>top (bottom): appropriate for the position specified in the options</li>
 61      *  <li>left (right): appropriate for the position specified in the options</li>
 62      * </ul>
 63      *
 64      * @return HTMLDomElement
 65      */
 66     render: function(){
 67         throw('You are rendering a base MapControl. It does NOTHING!');
 68     },

 70     resize: function(){
 71         return;

 74     show: function(){

 76         this.domElement.style.display = 'block';
 77     },

 79     hide: function(){

 81         this.domElement.style.display = 'none';
 82     },
 83     remove: function(){
 84         this.domElement.parent.removeChild(this.domElement);
 85     },
 86     /**
 87      * Controls the positioning of the control over the map.
 88 	 * You might need to implement the position function in controls inherited from the base
 89 	 * MapControl.
 90      */
 91     position: function(existingControls, allControls){
 92         
 93         this.domElement.style.position = 'absolute';
 94         //this.domElement.style.display = "block";
 95         
 96         if (this.options.position) return this.quickPosition(existingControls);
 97         
 98         if (this.options.layout) var placement = this.options.layout.bind(this)(this.options.map.width, this.options.map.height, existingControls, allControls);
 99         
100         this.domElement.style.left = placement.left + 'px';        
101         this.domElement.style.top = placement.top + 'px';
102         
103     },
104     
105     /**
106      *
107      * @private 
108      */
109     quickPosition: function(existingControls){
110         
111         //you give this function a set of controls, positions you care about,
112         //a property to check (frm the getPlacement() func) the value you 
113         //check against and if it's a > or < check. it will get the 
114         //value you need for you like a good function should. 
115         function findControlLimit(controls, goodPositions, property, currentValue, bigger){
116             //    console.log(controls.length)
117             var prop = currentValue;
118             for (var i = 0; i < controls.length; i++){
119                 var c  = controls[i];
120                 // if(c && c.domElement)
121                 //     console.log(c.domElement.style.display)
122                 // if(c.domElement.style.display=="none") {
123                 //     continue;
124                 // }
125                 if (goodPositions.indexOf(c.options.position) != -1){
126                     var place = c.getPlacement();
127                     
128                     if (bigger){
129                         if (place[property] > prop) prop = place[property];
130                     } else {
131                         if (place[property] < prop) prop = place[property];
132                     }
133                 }
134             }
135             return prop;
136         }
137         
138         switch (this.options.position){
139             
140             case 'topRight':                
141                 var right = findControlLimit(existingControls, ['topRight', 'rightTop'], 'left', this.options.map.width, false);
142                 
143                 this.domElement.style.left = right - this.width - this.options.marginX + 'px';                
144                 this.domElement.style.top = this.options.marginY + 'px';
145                 break;
146                 
147             case 'rightTop':                
148                 var top = findControlLimit(existingControls, ['topRight', 'rightTop'], 'bottom', 0, true);
149                 
150                 this.domElement.style.top = top + this.options.marginY + 'px';
151                 this.domElement.style.left = this.options.map.width - this.width - this.options.marginX+ 'px';                
152             break;
153             
154             case 'bottomLeft':
155                 var left = findControlLimit(existingControls, ['bottomLeft', 'leftBottom'], 'right', 0, true);
156                 
157                 this.domElement.style.left = left + this.options.marginX + 'px';
158                 this.domElement.style.top = this.options.map.height - this.height - this.options.marginY + 'px';          
159                 break;
160                 
161             case 'leftBottom':                
162                 var top = findControlLimit(existingControls, ['bottomLeft', 'leftBottom'], 'top', this.options.map.height, false);
163                 
164                 this.domElement.style.left = this.options.marginX + 'px';                
165                 this.domElement.style.top = top - this.height -  this.options.marginY + 'px';      
166                 break;
167                 
168             case 'bottomRight':
169                 var right = findControlLimit(existingControls, ['bottomRight', 'rightBottom'], 'left', this.options.map.width, false);
170                 
171                 this.domElement.style.left = right - this.width - this.options.marginX + 'px';                
172                 this.domElement.style.top = this.options.map.height - this.height - this.options.marginY + 'px';
173                 break;
174                 
175             case 'rightBottom':                
176                 var top = findControlLimit(existingControls, ['bottomRight', 'rightBottom'], 'top',  this.options.map.height, false);
177                 
178                 this.domElement.style.top = top - this.height - this.options.marginY + 'px';
179                 this.domElement.style.left = this.options.map.width - this.width - this.options.marginX + 'px';                
180             break;
181             //CONTINUE
182             case 'topLeft':
183             default:
184                 var left = findControlLimit(existingControls, ['topLeft', 'leftTop'], 'right', 0, true);
185                 
186                 this.domElement.style.top = this.options.marginY + 'px';
187                 this.domElement.style.left = left + this.options.marginX + 'px';
188                 break;
189            case 'leftTop':            
190                 var top = findControlLimit(existingControls, ['topLeft', 'leftTop'], 'bottom', 0, true);
191                 
192                 this.domElement.style.top = top + this.options.marginY + 'px';
193                 this.domElement.style.left = this.options.marginX + 'px';
194                 break;
195         }
196     },

230     getPlacement: function(){
231         return {
232             top: parseFloat(this.domElement.style.top),
233             left: parseFloat(this.domElement.style.left),
234             width: this.width,
235             height: this.height,
236             bottom: parseFloat(this.domElement.style.top) + this.height,
237             right: parseFloat(this.domElement.style.left) + this.width
238         };              
239     }
