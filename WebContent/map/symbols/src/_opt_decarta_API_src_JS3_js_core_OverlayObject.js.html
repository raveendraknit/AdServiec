1 /**
  2  * @class
  3  * OverlayObject is a base class that can be extended to create objects
  4  * suitable for display on a {@link deCarta.Core.MapOverlay}.
  5  * Pins {@link deCarta.Core.Pin}, Polylines {@link deCarta.Core.Polyline},
  6  * Circles {@link deCarta.Core.Circle}, Images {@link deCarta.Core.Image}, Polygons {@link deCarta.Core.Polygon} all extend this class.
  7  *
  8  * @description Base class for Map Overlay Objects
  9  *
 10  * @constructor
 11  * @param {options} opts Options are to be defined by the subclass.
 12  *
 13  * @see deCarta.Core.MapOverlay
 14  * @see deCarta.Core.Pin
 15  * @see deCarta.Core.Polyline
 16  * @see deCarta.Core.Circle
 17  * @see deCarta.Core.Image
 18  * @see deCarta.Core.Polygon
 19  */
 20 
 21 deCarta.Core.OverlayObject = function(opts){
 22     
 23     deCarta.Core.EventableObject.call(this, opts);
 24 
 25     this.useCSSTransforms = (deCarta.Core.Configuration.useHardwareAcceleration && deCarta.Window.hasCSSTransforms());
 26     this.renderingMode = 'none';
 27 
 28     this.zIndex = 100;
 29     
 30     var doc = document;
 31     // create xmlns
 32     
 33     //let's see if we can drop svg support in 3.0.9
 34     //why? canvas works, and we have less maint. it was chrome only anyways
 35     if (deCarta.Window.hasSVG() && !deCarta.Window.isFirefox()){ //firefox is teh suck with SVG
 36         this.renderingMode = 'SVG'; 
 37     } else if (deCarta.Window.hasCanvas()){ //so canvas is our second choice
 38         this.renderingMode = 'canvas';
 39     } else if (deCarta.Window.hasVML()){ //and lastly, it will be VML woho
 40         this.renderingMode = 'VML';
 41     } 
 42 
 43     if (this.renderingMode == 'VML' && doc.namespaces){
 44         if (!doc.namespaces['v']) {
 45             doc.namespaces.add('v', 'urn:schemas-microsoft-com:vml', "#default#VML");
 46         }
 47 
 48         if (!doc.styleSheets['deCarta_VML_Shapes']) {
 49             var ss = doc.createStyleSheet();
 50             ss.owningElement.id = 'deCarta_VML_Shapes';
 51             ss.cssText = 'v:\\* { behavior:url(#default#VML); display:inline-block }' 
 52         }
 53     }
 54 }

 57 deCarta.Core.OverlayObject.prototype = {
 58 
 59     /**
 60      * Returns the object's position
 61      * @return deCarta.Core.Position
 62      */
 63     getPosition: function(){
 64         return this.options.position;
 65     },
 66 
 67     /**
 68      * Sets the css z-index of this overlay object. This allows the user to place
 69      * page elements above or below this overlay.
 70      * @param {int} z The css z-index
 71      */
 72     setZIndex: function(z){
 73         this.zIndex = z;
 74     },
 75 
 76     /**
 77      * Returns the size of the pin
 78      * @return object <pre>{width: int, height: int}</pre>
 79      */
 80     getSize: function(z){
 81         if (this.boundingBox) return this.boundingBox.getSize(z);
 82         if (!this.domElement) return {width: 0, height: 0};        
 83         return {width: parseFloat(this.domElement.style.width), height: parseFloat(this.domElement.style.height)};
 84     },
 85 
 86     elementSpecs: function(x,y,z){
 87 
 88         var size = this.getSize(z);
 89         var x = (x - this.padding);
 90         var y = (y - this.padding);
 91 
 92         var xOfs = 0;
 93         var yOfs = 0;
 94         //shrink it 
 95         if (x < 0){
 96             xOfs = -x;
 97             size.width += x;
 98             x = 0;
 99         }
100 
101         if (y < 0){
102             yOfs = -y;
103             size.height += y;
104             y = 0;
105         }
106 
107         if (x + size.width > this.owner.owner.width){
108             size.width = this.owner.owner.width  - x;
109         }
110         if (y + size.height > this.owner.owner.height){
111             size.height = this.owner.owner.height  - y;
112         }   
113 
114         size.width += this.padding * 2;
115         size.height += this.padding * 2;
116 
117         return {
118             x: x,
119             y: y,
120             width: size.width,
121             height: size.height,
122             xOfs: xOfs,            
123             yOfs: yOfs
124         }
125     },    
126     
127     getIdealCenterAndZoom: function(map){
128         if (!this.boundingBox) deCarta.Core.Exception.raise('No bounding box!');
129         return this.boundingBox.getIdealCenterAndZoom(map);
130     },
131 
132     getBoundingBox: function(){
133         if (!this.boundingBox) deCarta.Core.Exception.raise('No bounding box!');
134         return this.boundingBox;
135     },
136 
137     /**
138      * Changes the position of the object. If the OverlayObject (or Pin)
139 	 * is registered with an overlay
140      * it will also take care of having the MapOverlay update its spatial
141      * indexes
142      * @param {deCarta.Core.Position} position The new position.
143      */
144     setPosition: function(position){
145         this.options.position = new deCarta.Core.Position(position.lat, position.lon);
146         if (this.setBB) this.setBB();
147         if (this.owner && this.owner.refreshObjects) this.owner.refreshObjects();
148         if(this.owner && this.owner.owner) this.owner.owner.render();
149     },
150 
151     /**
152      * Called by the subclass. Takes the domElement and positions it
153      * either with regular css or css3 transforms based on the settings.
154      */
155     domPosition: function(x, y, element){   
156         this.domX = x;
157         this.domY = y;
158         if (!element) element = this.domElement;
159         x = Math.floor(x);
160         y = Math.floor(y);
161 
162         if (!this.useCSSTransforms){
163             element.style.top = y  + 'px';
164             element.style.left = x  + 'px';
165         } else {
166             element.style.top = '0px';
167             element.style.left = '0px';
168             element.style.webkitTransformOrigin = " 0 0 ";
169             element.style.MozTransformOrigin = " 0 0 ";
170             element.style.OTransformOrigin = " 0 0 ";            
171             element.style.msTransformOrigin = " 0 0 ";         
172             element.style.TransformOrigin = " 0 0 ";
173             element.style.webkitTransform = "translate3d("+x+"px, "+y+"px, 0)";
174             element.style.msTransform = "translate3d("+x+"px, "+y+"px, 0)";
175             element.style.MozTransform = "translate3d("+x+"px, "+y+"px, 0)";
176             element.style.OTransform = "translate3d("+x+"px, "+y+"px, 0)";
177             element.style.Transform = "translate3d("+x+"px, "+y+"px, 0)";
178         }
179     },
180     
181     /**
182      * Called by the containing MapOverlay when a pin is added to it
183      * Allows the overlay to be notified when a pin's position is changed
184      * so it can rebuild its indexes.
185      */
186     registerOwner: function(owner){
187         this.owner = owner;
188     },

221     /**
222      * @private 
223      */
224     touchStart: function(ev, oe){
225 
226         if (this.options.opaque){            
227             deCarta.Touch.stopEvent(oe);
228         }
229 
230         if (!this.options.draggable) return;
231         this.dragging = true;        
232         this.dragged = false;
233 
234         this.startDragX = ev.screenX;
235         this.startDragY = ev.screenY;
236         
237         if (this.options.onDrag) this.options.onDrag.call(this, this.getCenter());
238         var eventPosition = this.owner.owner.positionFromXY(ev.pageX, ev.pageY);        
239         deCarta.Touch.stopEvent(oe);
240     },
241     
242     /**
243      * @private 
244      */
245     touchEnd: function(ev, oe){
246 
247         if (this.options.opaque){            
248             deCarta.Touch.stopEvent(oe);
249         }
250 
251         if (!this.options.draggable) return;
252         if (!this.dragging) return;
253 
254         this.dragging = false;
255         if (this.options.onDrop) this.options.onDrop.call(this, this.getCenter());
256         var eventPosition = this.owner.owner.positionFromXY(ev.screenX, ev.screenY);
257         
258         if (this.dragged)
259             deCarta.Core.EventManager.trigger('moveend',{object: this, event: ev, originalEvent: oe, eventPosition: eventPosition}, this);        
260         
261         deCarta.Touch.stopEvent(oe);
262     },
263     
264     /**
265      * @private 
266      */
267     touchMove: function(ev, oe){
268 
269         
270         if (!this.dragging) return;
271         if (!this.dragged) {
272             deCarta.Core.EventManager.trigger('movestart',{object: this, event: ev, originalEvent: oe, eventPosition: eventPosition}, this);
273             this.dragged = true;
274         }
275         
276         var deltaX = ev.screenX - this.startDragX;
277         var deltaY = ev.screenY - this.startDragY;
278 
279         this.startDragX = ev.screenX;
280         this.startDragY = ev.screenY;        
281 
282         
283         var posPx = this.getPosition().getPixelPoint(this.owner.owner.zoom)
284         //find new position 
285 
286         posPx.x += deltaX;
287         posPx.y -= deltaY;
288         var p = new deCarta.Core.Position(0,0);
289         p.setXY(posPx.x, posPx.y, this.owner.owner.zoom);
290 
291         this.setPosition(p);
292         this.owner.owner.render();
293         //this.startDragPos = this.getPosition();
294         var eventPosition = this.owner.owner.positionFromXY(ev.pageX, ev.pageY);
295         deCarta.Core.EventManager.trigger('move',{object: this, event: ev, originalEvent: oe, eventPosition: eventPosition}, this);                    
296         deCarta.Touch.stopEvent(oe);
297         
298     }
299 }
