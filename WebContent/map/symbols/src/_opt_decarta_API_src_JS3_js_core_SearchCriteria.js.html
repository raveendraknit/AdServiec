1 /**
  2  *  @class
  3  *  Data structure for POI request parameters, used with the
  4  *  {@link deCarta.Core.POISearch}:execute() method. The structure of the
  5  *  SearchCriteria class is as follows:
  6  * <pre>
  7  *  SearchCriteriaObject: {
  8  *      queryString: null, //[Not yet defined]
  9  *      maximumResponses: int, //Max number of responses that will be returned
 10  *      database: string, //Database that will be queried for this search
 11  *      sortDirection: string, //Sort order for results ["Ascending", "descending"]
 12  *      sortCriteria: string, //Property results will be sorted by ["Distance", "Score"]
 13  *      rankCriteria: string, //Property by which results will be ranked [Distance, Score]
 14  *      allowAggregates: bool, //Allow Aggregates
 15  *      retrieveAll: bool, //retreive all values
 16  *      properties: { //Object describing search properties, contains ONE OF:
 17  *          CATEGORY: string, //If present, this will perform a category search on this value
 18  *          KEYWORD: string, //If present, this will perform a Keyword search on this value
 19  *          POIName: string //If present, this will perform a freeform POI Search on this value
 20  *      }
 21  *      position: deCarta.Core.Position, //Position where the search is centered
 22  *      radius: deCarta.Core.Radius, //Radius of the search
 23  *      routeId: int, //if routeId is set, the search will be performed along the route. 
 24  *      corridorType: //[Not yet defined]
 25  *      map: deCarta.Core.Map //Optionally pass Map instead of position+radius
 26  * </pre>
 27  *
 28  * @description Data structure for POI request parameters.
 29  *
 30  * @see deCarta.Core.POISearch
 31  */
 32 
 33 deCarta.Core.SearchCriteria=function(){}
 34 deCarta.Core.SearchCriteria.prototype={
 35     /**
 36      *
 37      */
 38     queryString: null,
 39     /**
 40      * (int) Max number of responses that will be returned (default=10)
 41      */
 42     maximumResponses: 10,
 43     /**
 44      * (string) Database that will be queried for this search
 45 	 * (default='search:decarta:poi')
 46      */
 47     database: 'search:decarta:poi',
 48     /**
 49      * (string) Sort order for results ["Ascending" (default), "descending"]
 50      */
 51     sortDirection: "Ascending",
 52     /**
 53      * (string) Property results will be sorted by ["Distance" (default), "Score"]
 54      */
 55     sortCriteria: "Distance",
 56     /**
 57      * (string) Property by which results will be ranked [Distance, Score (default)]
 58      */
 59     rankCriteria: "Score",
 60     /**
 61      * (bool) Allow Aggregates (default=false)
 62      */
 63     allowAggregates: false,
 64     /**
 65      * (bool) retreive all values (default=false)
 66      */
 67     retrieveAll: false,
 68     /**
 69      * Object describing search properties
 70      * Can contain one of the following:
 71      * <ul>
 72      *  <li>CATEGORY : "string". <br />This will perform a category search.</li>
 73      *  <li>KEYWORD : "string". <br />This will perform a Keyword search.</li>
 74      *  <li>POIName : "string". <br />This will perform a freeform POI Search.</li>
 75      * </ul>
 76      */
 77     properties: {},
 78     /**
 79      * {@link deCarta.Core.Position} Position where the search is centered
 80      */
 81     position: null,
 82     /**
 83      * {@link deCarta.Core.Radius} Radius of the search
 84      */
 85     radius: new deCarta.Core.Radius(5, 'km'),
 86     /**
 87      * if corridorParams are set the search will be performed around the route.
 88      *  routeId - returned from @See deCarta.Core.Routing
 89      * measure : 'distance' or 'time' or 'euclideanDistance'
 90      * measureValue : int representing either meters or minutes
 91      */
 92     corridorParams: {
 93         routeId: null,
 94         measure : 'distance', 
 95         measureValue : 10
 96     },
 97     /**
 98      * {@link deCarta.Core.Map} Instead of passing position and radius, you can
 99 	 * pass a reference to your map and Position and Radius will be set according
100 	 * to the currently visible area on the map.
101      */
102     map: null
103 
104 }
