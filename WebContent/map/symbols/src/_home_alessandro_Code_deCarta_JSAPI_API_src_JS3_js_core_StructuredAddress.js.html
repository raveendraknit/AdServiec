1 /**
  2  * @class The StructuredAddress class holds structured address information.
  3  * Structured address information is pre-parsed and assigned an appropriate field
  4  * within the class; there is no confusion over whether the number is a postal
  5  * code or a block address number. This class can be used to pass an address
  6  * to a {@link deCarta.Core.Geocoder}:geocode() request. It can also be used to
  7  * parse a response from a {@link deCarta.Core.Geocoder}:reverseGeocode() request,
  8  * by passing the return object from that request to the {@link #fromWSResponse}
  9  * method of this class.
 10  *
 11  * For the purposes of the {@link deCarta.Core.Geocoder}:geocode() function,
 12  * you can either create an instance of a StructuredAddress, or you can create an
 13  * inline object with the following structure:
 14  * <pre>
 15  *   object: {
 16  *       buildingNumber: string, //(optional) Address number for this location.
 17  *       countryCode: string, //(optional) Country, formatted as ISO 3166-1 alpha-2 (2 digit) country code.
 18  *                            //See http://developer.decarta.com/docs/read/bada_api/Features
 19  *                            //for supported countries.
 20  *       countrySecondarySubdivision: string, //(optional) County (or equivalent) for this location.
 21  *       countrySubdivision: string, //(optional) Sub-country administrative
 22  *       division (ie the state, province, or region) for this location.
 23  *       countryTertiarySubdivision: string, //(optional) Additional subdivision (when relevant)
 24  *       landmark: string, //(optional) 
 25  *       locale: string, //(optional) Language, formatted as ISO 639 Language code.
 26  *                       //See http://developer.decarta.com/docs/read/bada_api/Features
 27  *                       //for supported languages.
 28  *       municipality: string, //(optional) City, town, village, or equivalent for this location.
 29  *       municipalitySubdivision: string, //(optional) Recognized neighborhood, borough, or equivalent for this location.
 30  *       postalCode: string, //(optional) Postal code, postcode, ZIP code, or equivalent numerical code for this location.
 31  *       speedLimit: string, //(optional) [Not Yet Defined]
 32  *       street: string, //(optional) Name and designation of the street (ie Main St) for this location.
 33  *       streetNameAndNumber: string, //(optional) [Not Yet Defined]
 34  *       type: string //(required) Must be set to 'structured' to create an inline structured address object
 35  *   }
 36  * </pre>
 37  *
 38  * @description The StructuredAddress class holds structured address information for the Geocoder
 39 
 40  * @constructor
 41  * @param {Object} addressObj
 42  * An object containing any of the following fields:
 43  * <ul>
 44  *   <li>buildingNumber: (optional) Address number for this location.</li>
 45  *   <li>countryCode: Country, formatted as ISO 3166-1 alpha-2 (2 digit) country code.
 46  *       See http://developer.decarta.com/docs/read/bada_api/Features for supported countries.</li>
 47  *   <li>countrySecondarySubdivision: (optional) County (or equivalent)
 48  *       for this location.</li>
 49  *   <li>countrySubdivision: (optional) Sub-country administrative
 50  *       division (ie the state, province, or region) for this location.</li>
 51  *   <li>countryTertiarySubdivision: (optional) Additional subdivision (when relevant)</li>
 52  *   <li>landmark</li>
 53  *   <li>locale: Language, formatted as ISO 639 Language code.
 54  *       See http://developer.decarta.com/docs/read/bada_api/Features
 55  *       for supported languages.</li>
 56  *   <li>municipality: (optional) City, town, village, or equivalent for
 57  *       this location.</li>
 58  *   <li>municipalitySubdivision: (optional) Recognized neighborhood,
 59  *       borough, or equivalent for this location.</li>
 60  *   <li>postalCode: (optional) Postal code, postcode, ZIP code, or
 61  *        equivalent numerical code for this location.</li>
 62  *   <li>speedLimit: [Not Yet Defined]</li>
 63  *   <li>street: (optional) Name and designation of the street (ie Main
 64  *       St) for this location.</li>
 65  *   <li>streetNameAndNumber: [Not Yet Defined]</li>
 66  * </ul>
 67  * @param {deCarta.Core.Locale} locale optional What parsing rules should be used to improve
 68  * the address match rate. Default is equivalent to a Locale of "en" (English)
 69  * language and "US" (United States) country parsing.
 70  *
 71  * @see deCarta.Core.Geocoder
 72  * @see deCarta.Core.FreeFormAddress
 73  */
 74 deCarta.Core.StructuredAddress = function(addressObj, locale){
 75     if (!addressObj) addressObj = {};
 76     //this.setAddress(addressObj);
 77     this.locale = locale || new deCarta.Core.Locale("en","US");
 78 
 79 
 80     // The building number could be 0, I guess
 81     if ( addressObj.buildingNumber || addressObj.buildingNumber == 0 ) {
 82         this.buildingNumber = addressObj.buildingNumber;
 83     } else {
 84         this.buildingNumber = "";
 85     }
 86     this.landmark = "";
 87     this.street = addressObj.street || "";
 88     this.streetNameAndNumber = "";
 89     this.speedLimit = "";
 90     this.countryCode = "";
 91     this.countrySubdivision = addressObj.countrySubdivision || "";
 92     this.countrySecondarySubdivision = addressObj.countrySecondarySubdivision || "";
 93     this.countryTertiarySubdivision = "";
 94     this.municipality = addressObj.municipality || "";
 95     this.postalCode = addressObj.postalCode || "";
 96     this.municipalitySubdivision = addressObj.municipalitySubdivision || "";
 97 
 98     this.type = 'structured';
 99 
100 }

102 deCarta.Core.StructuredAddress.prototype = {
103 
104     /**
105 	 * Provides the structured address as a single string, formatted ready for
106 	 * display to the user
107 	 * @return {string} String containing the address in a single string
108 	 */
109     toString: function(){
110 
111 
112         var retval = "";
113         if ( this.buildingNumber != "" && typeof this.buildingNumber != 'object')
114             retval += this.buildingNumber + " ";
115         if ( this.street != "" && typeof this.street === 'string')
116             retval += this.street + " ";
117         if ((deCarta.Core.Configuration.country==='CH') && this.municipalitySubdivision != "" && typeof this.municipalitySubdivision === 'string'){
118             retval += this.municipalitySubdivision + " ";
119         }
120         if ( this.municipality != "" && typeof this.municipality === 'string'){
121             retval += this.municipality + " ";
122         } else {
123             if ( this.municipalitySubdivision != "" && typeof this.municipalitySubdivision === 'string'){
124                 retval += this.municipalitySubdivision + " ";  
125             }
126         }
127         if ( this.countrySubdivision != "" && typeof this.countrySubdivision === 'string')
128             retval += this.countrySubdivision + " ";
129         if ( this.postalCode != "" && typeof this.postalCode === 'string')
130             retval += this.postalCode;
131         return retval;
132     },
133 
134     /**
135 	 * @private
136 	 * Initializes the properties of this StructuredAddress class using
137 	 * the values returned from a {@link deCarta.Core.Geocoder}:reverseGeocode()
138 	 * function call in the result data structure passed to the callback function.
139 	 * @param {reverseGeocodeResult} r Value passed into the callback function
140 	 * of a {@link deCarta.Core.Geocoder}:reverseGeocode() request.
141 	 */
142     fromWSResponse: function(r){
143         if (r.StreetAddress){
144             if (r.StreetAddress.Building){
145                 if(r.StreetAddress.Building.number.length && r.StreetAddress.Building.number.indexOf("-")>-1){
146                     var spl = r.StreetAddress.Building.number.split(" ");
147                     r.StreetAddress.Building.number = spl[0]+"-"+spl[spl.length-1];
148                 }
149                 this.buildingNumber = r.StreetAddress.Building.number;
150             }
151             if (r.StreetAddress.Street) this.street = r.StreetAddress.Street;
152         }
153         if (r.freeFormAddress) this.freeFormAddress = r.freeFormAddress;
154         if (r.countryCode) {
155             this.locale.country = r.countryCode;
156             this.countryCode = r.countryCode;
157         }
158         if (r.language) this.locale.language = r.language;
159         if (r.Place){
160             r.Place = deCarta.Utilities.makeArrayFix(r.Place);
161             for (var i =0; i < r.Place.length; i++){
162                 var prop = r.Place[i].type.substring(0,1).toLowerCase() + r.Place[i].type.substring(1);
163                 this[prop] = r.Place[i].content;
164             }
165         }
166         if (r.StreetIntersection){
167             this.streetIntersection = r.StreetIntersection;
168         }
169         if (r.PostalCode){
170             this.postalCode = r.PostalCode.replace(/\"/g, "");
171         }        
172         
173     },
174     
175 	/**
176 	 * @private
177 	 */
178     fromWSAddressResponse: function(r){
179         for (var prop in this){
180             if (this.hasOwnProperty(prop) && r[prop])
181             this[prop] = r[prop];
182         }
183     },
184 
185     /**
186      * @private
187      */
188     toXML: function(){
189 
190     },
191 
192     /**
193 	 * Generates a FreeFormAddress from this structured address
194 	 * @return {deCarta.Core.FreeFormAddress} A FreeFormAddress object containing the address in a single string
195 	 */
196     toFreeForm: function(){
197         return new deCarta.Core.FreeFormAddress(this.toString(), new deCarta.Core.Locale('EN', 'US'));
198     }
