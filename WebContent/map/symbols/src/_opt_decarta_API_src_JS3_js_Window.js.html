 65         var undef,

 68             all = div.getElementsByTagName('i');
 69 
 70         while (
 71             div.innerHTML = '<!--[if gt IE ' + (++v) + ']><i></i><![endif]-->',
 72             all[0]
 73         );
 74

139     /**
140      * Returns the reported devicePixelRatio, or if the device does not report it
141      * For example, Opera Mobile) it returns a best guesstimate
142      */
143     getDpr: function(){
144         
145         if (deCarta.Window.dpr) return deCarta.Window.dpr;
146         
147         if (window.devicePixelRatio){            
148             deCarta.Window.dpr = window.devicePixelRatio
149             return deCarta.Window.dpr;
150         }
151 
152         if (this.isIe() || !this.isMobile()) {
153             deCarta.Window.dpr = 1;
154             return deCarta.Window.dpr;
155         }
156 
157         if (/bada/i.test(navigator.userAgent || navigator.vendor || window.opera)){            
158             deCarta.Window.dpr = window.outerWidth / window.innerWidth;
159             return deCarta.Window.dpr;
160         }
161         
162         if (document.documentElement.clientWidth != window.innerWidth){ 
163             deCarta.Window.dpr = Math.floor((window.innerWidth / document.documentElement.clientWidth) * 100) / 100;
164             return deCarta.Window.dpr;
165         }
166         
167         if (document.documentElement.clientWidth < 480) deCarta.Window.dpr = 1;
168         deCarta.Window.dpr = 1;
169         return deCarta.Window.dpr;
170     },
171 
172     /**
173      * Get dimensions of the available viewport, in CSS pixels. 
174      *
175      */
176     getViewport: function(){
177 
178         if (this.isIe()){
179             return {width: document.body.clientWidth, height: document.body.clientHeight};
180         }
181 
182         var dpr = this.getDpr();
183 
184         if (window.opera){
185             return {width: window.innerWidth, height: Math.ceil(window.outerHeight / dpr)};
186         }
187 
188         if (/bada/i.test(navigator.userAgent || navigator.vendor)){
189              return {width: window.innerWidth, height: window.innerHeight};
190         }
191 
192        /* if (/android/i.test(navigator.userAgent || navigator.vendor || window.opera)){
193              return {width: window.outerWidth / dpr, height: (window.outerHeight / dpr) + 10};
194         }*/
195 
196         //return {width: document.documentElement.clientWidth , height: document.documentElement.clientHeight};
197         if (window.devicePixelRatio) return {width: window.innerWidth, height: window.innerHeight};
198         return {width: window.innerWidth / dpr, height: window.innerHeight / dpr};
199     }
