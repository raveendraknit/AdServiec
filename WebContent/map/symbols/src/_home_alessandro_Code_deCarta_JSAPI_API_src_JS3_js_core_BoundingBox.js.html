 10 deCarta.Core.BoundingBox = function(){
 11     
 12     var args = Array.prototype.slice.call(arguments);
 13     if (args.length == 1){
 14         var points = args[0];
 15     } else {
 16         var points = args.slice(0);
 17     }
 18     
 19     this.setPoints(points);
 20 }
 21 
 22 deCarta.Core.BoundingBox.prototype = {
 23     
 24     /**
 25     * Top left position of the bounding box.
 26     */
 27     topLeftPoint: null,
 28 
 29     /**
 30     * Bottom right position in the bounding box 
 31     */
 32     btmRightPoint: null,
 33     
 34     setPoints: function(points){
 35         
 36         this.points = [];
 37 
 38         for (var i = 0; i < points.length; i ++){
 39             
 40             if ((typeof points[i]).toLowerCase() === 'string'){
 41                 this.points[i] = new deCarta.Core.Position(points[i]);
 42             } else {
 43                 this.points[i] = points[i];
 44             }
 45         }
 46         
 47         this.findBounds();
 48         
 49         this.points = null; 
 50     },
 51     
 52     /**
 53      * Returns a {deCarta.Core.Position} with the center of the bounding box
 54      */
 55     getCenter: function(){
 56         var x = (this.topLeftPoint.getX(21) + this.btmRightPoint.getX(21)) / 2;
 57         var y = (this.topLeftPoint.getY(21) + this.btmRightPoint.getY(21)) / 2;
 58 
 59         var center = new deCarta.Core.Position();
 60         center.setXY(x, y, 21);
 61 
 62         return center;
 63     },
 64     
 65     /**
 66      * 
 67      * Does the bounding box contain this position? 
 68      * @param {deCarta.Core.Position} position
 69      */
 70     contains: function(pos){
 71         var minX = this.topLeftPoint.getX(21);
 72         var maxX = this.btmRightPoint.getX(21);
 73         var maxY = this.topLeftPoint.getY(21);
 74         var minY = this.btmRightPoint.getY(21);
 75 
 76         var x = pos.getX(21);
 77         var y = pos.getY(21);
 78 
 79         return (x >= minX && x <= maxX && y >=minY && y <= maxY);
 80     },
 81     
 82     /**
 83      * 
 84      * Returns a bounding box which is the intersection of this box and the 
 85      * one passed as a parameter
 86      */
 87     intersects: function(boundingBox){
 88 
 89         var left = this.topLeftPoint.getX(21);
 90         var right = this.btmRightPoint.getX(21);
 91         var bottom = this.topLeftPoint.getY(21);
 92         var top = this.btmRightPoint.getY(21);
 93 
 94         var left1 = boundingBox.topLeftPoint.getX(21);
 95         var right1 = boundingBox.btmRightPoint.getX(21);
 96         var bottom1 = boundingBox.topLeftPoint.getY(21);
 97         var top1 = boundingBox.btmRightPoint.getY(21);
 98 
 99         return !(left1 > right || right1 < left || top1 > bottom || bottom1 < top);
100     },

102     /**
103      * find the right zoom level to fit this bounding box
104      * the function needs to know the physical size of the 
105      * viewport. You can either pass in the map or the height 
106      * and width or even the DOM id for the map.  We will 
107      * take care of figuring out the right zoom regardless.
108      * getIdealCenterAndZoom(map)
109      * getIdealCenterAndZoom(width, height)
110      * getIdealCenterAndZoom(domID)
111      * getIdealCenterAndZoom({width:w, height: h})
112      */
113     getIdealCenterAndZoom: function(map){
114         //params :
115         // - map
116         // - width , height
117         // - 1 string : dom ID
118         // - object {w,h}
119         var args = Array.prototype.slice.call(arguments);
120         
121         var width = 0;
122         var height = 0;
123         
124         if (args.length == 2){
125             //are they numberos?
126             width = parseFloat(args[0]);
127             height = parseFloat(args[1]);
128             
129         } else if (args.length == 1) {
130             
131             if (typeof args[0] == 'object'){
132                 width = args[0].width;
133                 height = args[0].height;
134             } else if (typeof args[0] == 'string'){
135                 //dom id 
136                 var e = document.getElementById(args[0]);
137                 width = e.offsetWidth;
138                 height = e.offsetHeight;
139             } else {
140                 deCarta.Core.Exception.raise ('I have no idea what you might be trying to do. BoundingBox.getIdealCenterAndZoom can accept: aString, and Object {width: N, height: N}, a Map, or a DOM id. You are passing something weird.');
141             }
142             
143         } else {
144             deCarta.Core.Exception.raise('There are too many arguments in this call to the BoundingBox getIdealGenterAndZoom.')
145         }
146         
147         var res = {
148             zoom: null,
149             center: null
150         };
151         //i use the bounds i already have.
152         //i suppose. ... .. .. ..
153         //i will get the pixel dx and dy for the box
154         //then i will go through the zoom levels and get
155         //the pixel bounding box for the map at that level
156         //compare, find the right one - have zoom
157         //if i worked at facebook, i would get my own blog post,
158         //and maybe a badge or even a small flag 
159         
160         for (var z = 20; z > 0; z --){
161             
162             var x1 = this.topLeftPoint.getX(z);
163             var y1 = this.topLeftPoint.getY(z);
164             var x2 = this.btmRightPoint.getX(z);
165             var y2 = this.btmRightPoint.getY(z);
166             
167             var dX = Math.abs(x2 - x1);
168             var dY = Math.abs(y2 - y1);
169 
170             if (dX <= width && dY <= height){
171                 res.zoom = z; 
172                 break;
173             }
174         }
175         //then find the center point for the box
176         //convert to latlong, have center.
177         var x = (this.topLeftPoint.getX(res.zoom) + this.btmRightPoint.getX(res.zoom)) / 2;
178         var y = (this.topLeftPoint.getY(res.zoom) + this.btmRightPoint.getY(res.zoom)) / 2;
179 
180         var center = new deCarta.Core.Position();
181         center.setXY(x, y, res.zoom);
182         res.center = center;        
183         
184         return res;
185     },

187     /**
188     * Get the size (in pixels) of the bounding box, for a given zoom
189     * @param {float} zoom
190     *
191     */
192     getSize: function(zoom){
193         var top = this.topLeftPoint.getY(zoom);
194         var left = this.topLeftPoint.getX(zoom);
195 
196         var bottom = this.btmRightPoint.getY(zoom);
197         var right = this.btmRightPoint.getX(zoom);
198 
199         return {width: right - left ,height: top - bottom}
200     },
201     
202     /**
203      * @private
204      *
205      */
206     findBounds: function(){
207 
208         this.topLeftPoint = null;
209         this.btmRightPoint = null;
210 
211         var minX = null;
212         var minY = null;
213         var maxX = null;
214         var maxY = null;
215 
216         for (var i = 0; i < this.points.length; i++){
217             var pt = this.points[i];
218             
219             var ptX = pt.getX(18);
220             var ptY = pt.getY(18);
221 
222             if (ptX < minX || !minX) minX = ptX;
223             if (ptY < minY || !minY) minY = ptY;
224 
225             if (ptX > maxX || !maxX) maxX = ptX;
226             if (ptY > maxY || !maxY) maxY = ptY;
227 
228             delete pt;
229         }
230 
231         this.topLeftPoint = new deCarta.Core.Position(0,0);
232         this.btmRightPoint = new deCarta.Core.Position(0,0);
233 
234         this.topLeftPoint.setXY(minX, maxY, 18);
235         this.btmRightPoint.setXY(maxX, minY, 18);
236     }
