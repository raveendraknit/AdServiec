1 /**
  2  *  @class
  3  *  Data structure for route request parameters, used with the
  4  *  {@link deCarta.Core.Routing}:execute() method. The structure of the
  5  *  RouteCriteria class is as follows:
  6  * <pre>
  7  *   object: {
  8  *       distanceUnit: string, //Unit for route distance measurement. Valid values are: 'KM' (Kilometers), 'M' (Meters), 'MI' (Miles), 'FT' (Feet).
  9  *       instructionProvidePoint: bool, //[Not yet defined]
 10  *       instructionRules: string, //[Not yet defined]
 11  *       provideRouteHandle: bool, //Provide a route Id for further operations on route
 12  *       returnIdOnly: //<em>true</em> will cause route query to return a server route ID
 13  *       routePreference: string, //"AvoidFreeways", "Easy", "Fastest", "MoreFreeways", "NoFreeways", "Pedestrian", "Shortest"
 14  *       routeQueryType: string, //Type of DDS Query, 'RMAN' or 'RTXT' [Need to define valid values and what they indicate.]
 15  *       waypoints: //Must include at least start and end points
 16  *           0: {deCarta.Core.Position}
 17  *           ...
 18  *           N: {deCarta.Core.Position}
 19  * </pre>
 20  *
 21  * @description Data structure for setting route request parameters.
 22  *
 23  * @see deCarta.Core.Routing
 24  */
 25 deCarta.Core.RouteCriteria = function(){};
 26 deCarta.Core.RouteCriteria.prototype = {
 27     /**
 28      * Unit for route distance measurement. Valid values are: 'KM' (Kilometers), 'M' (Meters), 'MI' (Miles), 'FT' (Feet)
 29      */
 30     distanceUnit : 'M',
 31     /**
 32      * Type of DDS Query, 'RMAN' or 'RTXT' [Need to define valid values and what they indicate.]
 33      */
 34     routeQueryType: 'RMAN',
 35     /**
 36      * Provide a route Id for further operations on route @See deCarta.Core.SearchCriteria for CorridorSearch
 37      */
 38     provideRouteHandle: 'false',
 39     /**
 40      * Route preferences.
 41      * <p>The various different routing styles are summarized below.</p>
 42      * <ul>
 43      * <li>"AvoidFreeways" = Return a vehicular route that avoids limited access
 44      * roads (e.g. freeways) as much as possible.</li>
 45      * <li>"Easy" = Return a vehicular route that attempts to make as few turns,
 46      * balancing this constraint with the shortest travel time.</li>
 47      * <li>"Fastest" = (DEFAULT) Return a vehicular route with the smallest, calculated
 48      * travel time. This is the standard, and default routing style.</li>
 49      * <li>"MoreFreeways" = Return a vehicular route that will attempt to use as
 50      * many limited access roads (e.g. freeways) as possible.</li>
 51      * <li>"NoFreeways" = Return a vehicular route that avoids limited access
 52      * roads (e.g. freeways) entirely. This route will take only surface street
 53      * and arterial roads to the destination.</li>
 54      * <li>"Pedestrian" = Return a route fit for Pedestrian traffic. Routes will
 55      * avoid limited access roads, ignore vehicular signage restrictions, obey
 56      * impassible physical restrictions (like grade separations), take the most
 57      * direct path possible, and utilize pedestrian only foot-traffic paths.</li>
 58      * <li>"Shortest" = Return a vehicular route with the shortest total distance
 59      * traveled.</li>
 60      * </ul>
 61      */
 62     routePreference: 'Fastest',
 63 	
 64     /**
 65      * An array of {@link deCarta.Core.Position}s, from start-to-end of the
 66 	 * route. Must include at least 2 (start and end).
 67      */
 68     waypoints: [],
 69 	
 70     /**
 71      * [Not yet defined]
 72      */
 73     instructionProvidePoint: 'true',
 74 
 75     /**
 76      * The type of vehicle type to use in calculating the route.
 77      * Valid values are  "PassengerCar", "4WD", "HOV", "Emergency", "Taxi",
 78      * "PublicBus", "DeliveryTruck", "TransportTruck", "Motorcycle" and
 79      * "Pedestrian". 
 80      * It can either be a single value (eg PassengerCar) or a comma separated list
 81      * (eg PassengerCar,4WD) or an array (eg ["PassengerCar", "4WD"])
 82      * Defaults to PassengerCar
 83      */
 84     vehicleTypes: "PassengerCar",
 85 	
 86     /**
 87      * [Not yet defined]
 88      */
 89     instructionRules: 'maneuver-rules',
 90 	
 91     /**
 92      * If set to <em>true</em>, the route calculation will only return a route ID.
 93 	 * Default=false
 94      */
 95     returnIdOnly: 'false',
 96 
 97     /**
 98      * If set to <em>true</em>, the route calculation will take into consideration live traffic.
 99      * Default=false
100      * @note the server must be running the deCarta Traffic Manager
101      */
102     trafficEnabled: 'false',
103 
104     /**
105      * Array of any combination of:     
106      * Polygon, Circle or Polyline objects
107      */
108     avoidAreas: [],
109 
110     /***
111      * Array of features to avoid. Features are string and can be :
112      * Ferry
113      * Toll
114      * Tunnels
115      * Bridges
116      *
117      */
118     avoidList: [],
119 
120     /**
121     * Integer. 
122     * If this is set to N, the routing query will return N alternate routes 
123     * in addition to the regular response. 
124     */
125     alternateRoutes: 0,
126 
127     /**
128     * Boolean. 
129     * If the route has moltiple waypoints, setting this to true will fine the best 
130     * order in which to visit the waypoints. If false, the waypoints will be 
131     * visited in their original order. 
132     */
133     optimized: false,
134     /**
135     * maneuverMaps is a flag to trigger each RouteInstruction to include
136     * a URL to a small image representing the maneuver in a top up
137     * orientation
138     * @type String URL
139     * @default false
140     */
141     maneuverMaps : false,
142     /**
143     * configures the height and width of the maneuver maps
144     * @type Object
145     * @default {height:200,width:200}
146     */
147     maneuverMapConfig : {
148         format:"GIF",
149         orientation:"heading-up",
150         height:200,
151         width:200
152     }    
153
