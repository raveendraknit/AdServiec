  2         F=F+"";
  3         var K="0123456789ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz-_.~";
  4         var C="!*'();:@&=+$,/?%#[]";
  5         var I=K+C;
  6         var H="0123456789ABCDEFabcdef";
  7         function J(L){
  8             return "%"+H.charAt(L>>4)+H.charAt(L&15);
  9         }
 10         var D="";
 11         for(var E=0;E<F.length;E++){
 12             var B=F.charAt(E);
 13             if(K.indexOf(B)!=-1){
 14                 D=D+B;
 15             }else{
 16                 var G=F.charCodeAt(E);if(G<128){
 17                     D=D+J(G);
 18                 }if(G>127&&G<2048){
 19                     D=D+J((G>>6)|192);D=D+J((G&63)|128);
 20                 }if(G>2047&&G<65536){
 21                     D=D+J((G>>12)|224);D=D+J(((G>>6)&63)|128);D=D+J((G&63)|128);
 22                 }if(G>65535){
 23                     D=D+J((G>>18)|240);D=D+J(((G>>12)&63)|128);D=D+J(((G>>6)&63)|128);D=D+J((G&63)|128);
 24                 }
 25             }

 30 /**
 31  * @private
 32  * Performs JS Request to DDW Web Services APIs.
 33  *
 34  * This class is for advanced users, and is used to communicate directly with deCarta DDS Web Services
 35  * through XML requests. Just use .send(xml, onSuccess, onFail) with the desired XML 
 36  * and success / fail callbacks. 
 37  *
 38  * Requests are performed using Dynamic Script Tag mode.
 39  * @description
 40  * The deCarta JavaScript API communicates with the DDS Web Services via
 41  * XML-over-HTTP requests. Browser "domain of origin" security prevents any
 42  * JavaScript library (including the deCartaMobile.js library) from being
 43  * downloaded from one domain and used to send queries to a different domain.
 44  * Since most AJAX applications require accessing services from more than one
 45  * domain, there are two main methods used to work around the domain of origin
 46  * security. One method requires implementation of a proxy on the server to
 47  * redirect queries from the client to the DDS Web Services. The second, method
 48  * involves wrapping the XML sent to the DDS Web Services in a set of dynamic
 49  * script tags, and sending the XML over as a JSON request.
 50  *
 51  * The second method is simpler to use out of the box because it is directly
 52  * supported via the JSRequest object.
 53  *
 54  * @description Performs JS Request to DDW Web Services APIs.
 55  *
 56  */
 57 deCarta.Core.JSRequest = {
 58 
 59     callbacks: {},
 60     timeouts: {},    
 61     interceptors: {
 62         DirectoryResponse: [],
 63         RUOKResponse: [],
 64         DetermineRouteResponse: [],
 65         ReverseGeocodeResponse: [],
 66         GeocodeResponse: []
 67     },
 68     sessionId: Math.ceil((new Date().getTime()) * Math.random()),
 69     requestId: null,
 70     /**
 71      * This initializes the JSRequest object and
 72      * Performs the necessary authentication with DDS Web Services.
 73      */
 74     init: function(onInit){
 75         if (typeof onInit !== 'function'){
 76             deCarta.Core.Exception.raise('JSRequest.init needs a callback parameter');
 77         }
 78         var xmlRequest =
 79         "       <xls:Request version='1.0' requestID='1675192' methodName='RuokRequest'>" +
 80         "             <xls:RUOKRequest />" +
 81         "       </xls:Request>";
 82 
 83         var start = new Date().getTime();
 84         var proto = deCarta.Core.Configuration.url.split(":")[0];
 85 
 86         this.send(xmlRequest, function(res){
 87             var time = (new Date().getTime())-start;
 88             try {
 89                 if (typeof res.XLS.ResponseHeader.ErrorList === 'object'){
 90                     onInit( {
 91                         success:false,
 92                         msg:res.XLS.ResponseHeader.ErrorList.Error.message,
 93                         exTime:time
 94                     })
 95                 } else if (typeof res.XLS.Response.RUOKResponse === 'object'){
 96                     if (res.XLS.Response.RUOKResponse.urlFormats){                                                
 97                         if (deCarta.Core.Configuration.urlVersion == null || 
 98                             deCarta.Core.Configuration.urlVersion == 'auto'){
 99 
100                                 var versions = res.XLS.Response.RUOKResponse.urlFormats.split('|');
101                                 var maxV = 1;
102                                 for (var i = 0; i < versions.length; i++){
103                                     maxV = Math.max(maxV,parseFloat(versions[i].match(/\d/g)));                                    
104                                 }
105 
106                                 deCarta.Core.Configuration.urlVersion = maxV;
107                             }
108                     }
109                     
110                     deCarta.Core.Configuration.url = proto + '://' + res.XLS.Response.RUOKResponse.hostName + '/openls/JSON';
111                     var maxHostAliases = res.XLS.Response.RUOKResponse.maxHostAliases ? parseInt(res.XLS.Response.RUOKResponse.maxHostAliases) : 0;
112                     deCarta.Core.Configuration.streetTileHosts=[];
113                     if(maxHostAliases){ 
114                         for( var i=1;i<=maxHostAliases; i++){
115                             var alias = proto + '://' +res.XLS.Response.RUOKResponse.hostName.replace(/\./,"-0"+i+".");
116                             deCarta.Core.Configuration.streetTileHosts.push(alias);
117                         }
118                     } else {
119                         deCarta.Core.Configuration.streetTileHosts.push('http://' +res.XLS.Response.RUOKResponse.hostName);
120                     }
121                     // all is good!  party time...
122                     onInit( {
123                         success:true,
124                         msg : "ok",
125                         exTime: time
126                     } );
127                 } 
128             } catch (e) {
129                 onInit({
130                     success:false,
131                     msg:e.message,
132                     exTime : time
133                 })
134             }
135         }.bind(this),
136             function()  {
137                 var time = (new Date().getTime())-start;
138                 onInit({
139                     success:false,
140                     msg:"connection timeout",
141                     exTime : time
142                 })
143             }.bind(this));
144 
145     },

147     /**
148      * Send an XML request to DDS Web Services
149      * @param {string} xml the XML to be sent to the server
150      * @param {function} onSuccess the callback function that will receive the
151      * results of the xml request
152      * @param {function} onFailure the callback function that will be invoked in
153      * the event of a failure.
154      * @param {bool} getRaw: skips all interceptors. Good for back compat. 
155      * @param {function} requestInterceptor: function that has access to the full request before it is sent. 
156      * if provided, what will be sent to the server will be the return value of the function.
157      */
158     send: function(xml, onSuccess, onFailure, getRaw, requestInterceptor){
159 
160         //generate id
161         var reqId = Math.ceil((new Date().getTime()) * Math.random());
162         var elId = "deCarta-mob-req-" + reqId;
163         var regEx = /requestID=[\'\"]([0-9]+)[\'\"]/;
164         xml = xml.replace(regEx, 'requestID="' + reqId + '"');
165         
166         var completeXML = this.getRequestHeader() + xml + this.getRequestFooter();
167         if (requestInterceptor && typeof requestInterceptor=='function') 
168             completeXML = requestInterceptor(completeXML);
169 
170         if(deCarta.Core.Configuration.consolelogXML){
171             console.log(completeXML);
172         }
173 
174         var data  = encodeURIComponentUTF8(completeXML);
175 
176         this.callbacks[reqId] = function(response){
177             if(deCarta.Core.Configuration.consolelogXML){
178                 try{
179                     console.log(response);
180                 } catch(e){}
181             }
182             var el = document.getElementById(elId);
183             clearTimeout(this.timeouts[reqId]);
184 
185             if (!getRaw){
186 
187                 for (var type in response.response.XLS.Response){
188                     if (this.interceptors[type])
189                     for (var i = 0; i < this.interceptors[type].length; i++){
190                         if (typeof this.interceptors[type][i] === 'function') 
191                             response.response = this.interceptors[type][i](response.response);
192                     }
193                 }
194 
195             }            
196 
197             if(el){
198                 el.parentNode.removeChild(el);
199                 //deCarta.Core.timeLog({type: "jsRequest Complete", requestId: elId});
200                 onSuccess(response.response);
201             }
202         }.bind(this);
203 
204         this.timeouts[reqId] = setTimeout(function(elId){
205             this.cancel(elId);
206             //deCarta.Core.timeLog({type: "jsRequest Timeout", requestId: elId});
207             if (typeof onFailure == 'function') onFailure({
208                 'err': 'timeout'
209             });
210         }.bind(this, elId, onFailure), deCarta.Core.Configuration.requestTimeout);
211 
212         var url = deCarta.Core.Configuration.url +
213         '?reqID=' + reqId +
214         '&chunkNo=1&numChunks=1' +
215         '&callback=window.deCarta.Core.JSRequest.callbacks[' + reqId + ']'+
216         '&data=' + data +
217         '&responseFormat=JSON';
218         var sTag = document.createElement('script');
219         sTag.id  = elId;
220         sTag.src = url;
221 
222         //deCarta.Core.timeLog({type: "jsRequest", url: url, requestId: elId});
223 
224         document.body.appendChild(sTag);
225         return elId;
226     },

252     /**
253      * cancel request by ID
254      * @return {boolean} result of cancelation attempt (true if canceled, false if response already processed)
255      */
256     cancel:function(elId){
257         if(document.getElementById(elId)){
258             var el = document.getElementById(elId);
259             el.src=null;
260             el.parentNode.removeChild(el);
261             return true;
262         } else {
263             return false;
264         }

267     addInterceptor: function(type, interceptor){
268         if (!this.interceptors[type]) deCarta.Core.Exception.raise('Interceptor type '+type+' not supported');
269         var id = this.interceptors[type].length;
270         this.interceptors[type].push(interceptor);
271         return type + "-" + id;
272     },
273 
274     removeInterceptor: function(type, interceptor){
275         if (!this.interceptors[type]) deCarta.Core.Exception.raise('Interceptor type '+type+' not supported');
276         for (var i = 0 ; i < this.interceptors[type].length; i++){
277             if (this.interceptors[type][i] == interceptor){
278                 this.interceptors[type][i] = null;
279                 return;
280             }
281         }
282     },
283 
284     removeInterceptorById: function(id){
285         var id = id.split('-');
286         this.interceptors[id[0]][id[1]] = null;
287         return;
288     }
