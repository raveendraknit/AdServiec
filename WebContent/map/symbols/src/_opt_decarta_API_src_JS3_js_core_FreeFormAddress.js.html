1 /**
  2  * @class Represents a Free Form Address.
  3  * For the purposes of the {@link deCarta.Core.Geocoder}:geocode() function,
  4  * you can either create an instance of a FreeFormAddress, or you can create an
  5  * inline object with the following structure:
  6  * <pre>
  7  *   object: {
  8  *       address: string, //free form address
  9  *       locale: deCarta.Core.Locale, //a locale object
 10  *       type: string = 'freeForm'
 11  *   }
 12  * </pre>
 13  *
 14  * @description Free Form Address for use with Geocoder
 15  *
 16  * @constructor
 17  * @param {String} address The free-form address, containing a partial or complete 
 18  * address
 19  * @param {deCarta.Core.Locale} locale A Locale object
 20  * 
 21  * @see deCarta.Core.Geocoder
 22  * @see deCarta.Core.StructuredAddress
 23  */
 24 
 25 
 26 deCarta.Core.FreeFormAddress = function(address, locale){
 27     this.address = address;
 28     this.locale = locale;
 29     this.type = 'freeForm';
 30 }
 31 
 32 deCarta.Core.FreeFormAddress.prototype = {
 33 
 34     /**
 35 	 * @private
 36 	 */
 37     toString: function(){
 38         return this.address;
 39           
 40     },
 41 
 42     /**
 43      * Returns an XML Representation of the FreeFormAddress
 44      * @return {string} XML formatted address
 45      */
 46     toXML: function(){
 47 
 48 
 49         var xml =
 50         '<xls:Address countryCode="'+this.locale.getCountry()+'" language="'+this.locale.getLanguage()+'">' +
 51             '<xls:freeFormAddress>' + this.address + '</xls:freeFormAddress>' +
 52         '</xls:Address>';
 53 
 54         return xml;
 55     }
 56 }
