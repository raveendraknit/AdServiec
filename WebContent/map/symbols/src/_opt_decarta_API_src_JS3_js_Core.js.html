 18     /**
 19      * @private
 20      */
 21     loadModule: function(module, onLoad){
 22         
 23         if (this.modules[module]) onLoad();
 24         var sTag = document.createElement('script');
 25         sTag.onload = this.moduleLoaded.bind(this, module, onLoad);
 26         sTag.src = 'modules/' + module + '.js';
 27         document.body.appendChild(sTag);
 28     },
 29 
 30     /**
 31      * @private
 32      */
 33     moduleLoaded: function(module, onLoad){
 34         
 35         this.modules[module] = true;
 36         onLoad();
 37     }
