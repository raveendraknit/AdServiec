116                   </xls:Position>

119       </xls:Request>

121  */
122         preference = preference || 'StreetAddress';
123         var xml = 
124          "<xls:Request maximumResponses='10' version='1.0' requestID='0123456789' methodName='ReverseGeocodeRequest'>" +
125                 "<xls:ReverseGeocodeRequest>" +
126                       "<xls:Position>"+
127                             "<gml:Point>"+
128                                   "<gml:pos>" + pos.getLat()  + " " + pos.getLon() + "</gml:pos>"+
129                             "</gml:Point>"+
130                       "</xls:Position>"+
131                       "<xls:ReverseGeocodePreference>"+preference+"</xls:ReverseGeocodePreference>"+
132                 "</xls:ReverseGeocodeRequest>"+
133           "</xls:Request>";
134

139     /**
140      * Translate an address into a list of Positions (latitude and
141 	 * longitude coordinates).
142      * Since user input may often be incomplete, it is important to be able to
143      * accept the multiple matches that can be returned from a geocode request.
144      * Geocoding is an asynchronous request made to the DDS Web Services, and thus
145      * requires a callBack function to catch the returned results.
146      * The application should provide a callBack function that has the following prototype:
147      * <pre>
148      * function geocodeCallBack(results)
149      * </pre>
150      * Where <pre>results</pre> is an array, with each element of the array
151 	 * having the following structure:
152 	 * 
153      * <pre>
154 {
155   "Address": deCarta.Core.StructuredAddress,
156   "Position": deCarta.Core.Position,
157   "matchType": "string",
158   "accuracy": float
159 }
160      * </pre>
161      *
162      * @param {Address} address StructuredAddress or FreeFormAddress to be geocoded
163 	 * <p><strong>Note:</strong> This can be EITHER an instance of a
164 	 * {@link deCarta.Core.StructuredAddress} or
165 	 * {@link deCarta.Core.FreeFormAddress} class OR an equivalent structure
166 	 * created as an inline object.</p>
167 	 * <p>An inline StructuredAddress can be implemented by defining an inline object with one or more of these fields:</p>
168 	 * <ul>
169      * <li>buildingNumber: (optional) Address number for this location.</li>
170 	 *   <li>countryCode: Country, formatted as ISO 3166-1 alpha-2 (2 digit) country code. 
171 	 *       See http://developer.decarta.com/docs/read/bada_api/Features for supported countries.</li>
172      *   <li>countrySecondarySubdivision: (optional) County (or equivalent)
173      *       for this location.</li>
174      *   <li>countrySubdivision: (optional) Sub-country administrative
175      *       division (ie the state, province, or region) for this location.</li>
176 	 *   <li>countryTertiarySubdivision: (optional) Additional subdivision (when relevant)</li>
177 	 *   <li>landmark</li>
178 	 *   <li>locale: Language, formatted as ISO 639 Language code.
179 	 *       See http://developer.decarta.com/docs/read/bada_api/Features for supported languages.</li>
180      *   <li>municipality: (optional) City, town, village, or equivalent for
181      *       this location.</li>
182      *   <li>municipalitySubdivision: (optional) Recognized neighborhood,
183      *       borough, or equivalent for this location.</li>
184      *   <li>postalCode: (optional) Postal code, postcode, ZIP code, or
185      *        equivalent numerical code for this location.</li>
186 	 *   <li>speedLimit: [Not currently documented]</li>
187      *   <li>street: (optional) Name and designation of the street (ie Main
188      *       St) for this location.</li>
189 	 *   <li>streetNameAndNumber: [Not currently documented]</li>
190 	 *   <li>type: Must be set to 'structured' to create an inline structured address object</li>
191 	 * </ul>
192      *
193 	 * <p>An inline FreeFormAddress can be implemented by defining these fields:</p>
194 	 * <ul>
195 	 *   <li>(string) address: The free-form address</li>
196 	 *   <li>locale: Language, formatted as ISO 639 Language code.
197 	 *       See http://developer.decarta.com/docs/read/bada_api/Features for supported languages.</li>
198 	 *   <li>type: Must be set to 'freeForm' to create an inline Free Form address object</li>
199 	 * </ul>
200      * <br />
201      * @param {function} callback The callback function that will receive the results of the geocode operation
202      * @throws {Exception} Invalid or missing address object passed as parameter
203      * @throws {Exception} Invalid or missing callback function
204      */
205     geocode: function(address, callback){
206         if (!address.type || !(address.type in {'freeForm': true, 'structured': true})) {
207             deCarta.Core.Exception.raise('When using the geocode function, the first parameter must be an address object');
208         }
209         if (typeof callback !== 'function'){
210             deCarta.Core.Exception.raise('When using the geocode function, the second parameter must be a callback function');
211         }
212 
213         deCarta.Core.JSRequest.send(this.getGeocodeXML(address), this.handleResponse.bind(this, callback), this.handleFailure.bind(this, callback));
214 
215     },

218     /**
219      * Translate a Position (latitude, longitude coordinate) into an Address.
220      * Reverse geocodes are always approximated to the best possible address
221      * within the range of addresses available in the map data. Reverse geocoding
222      * is an asynchronous request made to the DDS Web Services,
223      * and thus requires a callBack function to catch the returned results.
224      * The application should provide a callBack function that has the following
225      * prototype:
226      * <pre>
227      *  function reverseGeocodeCallBack(result)
228      * </pre>
229      * If a valid reverse geocode can be made, the resulting position will be
230 	 * passed to the callBack in an result object. The structure of the result object
231 	 * is as follows:
232 {
233   "Address": deCarta.Core.StructuredAddress,
234   "Position": deCarta.Core.Position
235 }
236      * </pre>	 
237      * A failed reverse geocode will return an empty Address (result.Address.toString() == "").
238      * @param {deCarta.Core.Position} position The position to reverse geocode.
239 	 * This can be passed as an instance of a {@link deCarta.Core.Position} class,
240 	 * or as an inline-object with the following fields:
241 	 * <ul>
242 	 *   <li>{float} lat: Latitude of the position</li>
243 	 *   <li>{float} lat: Longitude of the position</li>
244 	 * </ul>
245 	 *
246      * @param {function} callback Function that will handle the result of the reverse geocode.
247      * @throws {Exception} Invalid or missing callback function
248      *
249      */
250     reverseGeocode: function(position, callback, preference){
251         if (typeof callback !== 'function'){
252             deCarta.Core.Exception.raise('When using the reverseGeocode function, the second parameter must be a callback function');
253         }
254 
255         deCarta.Core.JSRequest.send(this.getRevGeocodeXML(position, preference), this.handleRevResponse.bind(this, callback), this.handleFailure.bind(this, callback));
256     },

258     /**
259      *@private
260      */
261     handleRevResponse: function(callback, response){
262         try {
263             var t = response.XLS.Response.ReverseGeocodeResponse.ReverseGeocodedLocation;
264             var res = {};
265             res.Address = new deCarta.Core.StructuredAddress();
266             res.Address.fromWSResponse(t.Address);
267 
268             res.Position = new deCarta.Core.Position(t.Point.pos);
269 
270             if (t.SearchCentreDistance){
271                 res.searchCenterDistance = new deCarta.Core.Distance(t.SearchCentreDistance.value, t.SearchCentreDistance.uom);
272             }
273             callback(res, t);
274         } catch (e){            
275             callback({Address: {}, Position: {}}, null);
276         }
277     },

279     /**
280      * @private
281      */
282     handleResponse: function(callback, response){
283         
284         var err = 'Unknown';
285 
286         try {
287             if (response.XLS.Response.ErrorList.Error.message){
288                 callback({error: response.XLS.Response.ErrorList.Error.message});
289                 return;
290             }
291         } catch (e) {
292             //there is no such field forget it.
293         }
294 		try {
295             var res = deCarta.Utilities.makeArrayFix(response.XLS.Response.GeocodeResponse.GeocodeResponseList.GeocodedAddress);
296     		for (var i =0; i < res.length; i++){
297 				var add = new deCarta.Core.StructuredAddress();
298 				add.fromWSResponse(res[i].Address);
299 				res[i].Address = add;
300 				res[i].Position = new deCarta.Core.Position(res[i].Point.pos);
301                 res[i].Point = null;
302                 res[i].matchType = res[i].GeocodeMatchCode.matchType;
303                 res[i].accuracy = res[i].GeocodeMatchCode.accuracy;
304                 res[i].GeocodeMatchCode = null
305                 delete res[i].GeocodeMatchCode;
306                 delete res[i].Point;
307 			}
308 			callback(res, response);
309 		} catch (e){
310             try {
311                 err = response.XLS.Response.ErrorList.Error.message;
312             } catch (e){}
313             deCarta.Core.Exception.raise('Error handling response : ' + err);
314 		}
315
