1 /**
  2  * @class
  3  * A Locale object represents a specific geographical, political, or
  4  * cultural region. An operation that requires a Locale to perform its task is
  5  * called locale-sensitive and uses the Locale to tailor information for the
  6  * user. Currently, Locales can be attached to a {@link deCarta.Core.FreeFormAddress} object
  7  * to aid in producing better results with a {@link deCarta.Core.Geocoder}:geocode().
  8  * <p>These pairs of language can be used to create a valid locale for use with
  9  * the Geocoder.</p>
 10  *
 11  * @description A language, country pair for localization
 12  *
 13  * @constructor
 14  * @param {String} language required A valid ISO Language Code. It is always an
 15  * upper case, two letter string.
 16  * @param {String} country required A valid ISO Country Code. It is always an
 17  * upper case, two letter string.
 18  *
 19  * For a list of country codes, please see: http://www.iso.org/iso/country_codes/iso_3166_code_lists/country_names_and_code_elements.htm
 20  * For a list of language codes, please see: http://www.loc.gov/standards/iso639-2/php/English_list.php
 21  *
 22  * @see deCarta.Core.FreeFormAddress
 23  * @see deCarta.Core.Geocoder
 24  */  
 25 deCarta.Core.Locale = function( language, country ) {  
 26     /**
 27      * A valid ISO Language Code. It is always an upper case, two letter string.
 28      * See the class description for valid language and country code pairs.
 29      * @private
 30      * @type String
 31      */
 32     this.language = language.toUpperCase();
 33     /**
 34      * A valid ISO Country Code. It is always an upper case, two letter string.
 35      * @private
 36      * @type String
 37      */
 38     this.country = country.toUpperCase();
 39 } // End Locale constructor
 40 
 41 
 42 
 43 deCarta.Core.Locale.prototype = {
 44     /**
 45      * Retrieve the ISO Language Code for this locale
 46      * @return {String} language field of the Locale object
 47      */
 48     getLanguage: function() {
 49         return this.language;
 50     },
 51 
 52     /**
 53      * Set the ISO Language Code for this locale.
 54      * @param {String} language required A valid ISO Language Code. It is always an
 55      * upper case, two letter string. 
 56      */
 57     setLanguage: function(language) {
 58         this.language = language.toUpperCase();
 59     },
 60 
 61     /**
 62      * Retrieve the ISO Country Code for this locale
 63      * @return {String} country field of the Locale object
 64      */
 65     getCountry: function() {
 66         return this.country;
 67     },
 68 
 69     /**
 70      * Set the ISO Country Code for this locale.
 71      * @param {String} country required country required A valid ISO Country Code.
 72      * It is always an upper case, two letter string. 
 73      */
 74     setCountry: function(country) {
 75         this.country = country.toUpperCase();
 76     },
 77 
 78     /**
 79      * Concatenates the country and language fields, in that order and returns
 80      * that string.
 81      * @return {String} The Locale object as a single, underbar delimited string, ex: "en_US".
 82      */
 83     toString: function() {
 84         return  this.language  + "_" +  this.country;
 85     }
 86 };
