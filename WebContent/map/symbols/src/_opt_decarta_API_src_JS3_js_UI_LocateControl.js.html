1 /**
  2  * @class
  3  * Add a Geolocation button to the map.
  4  * This class inherits from {@link deCarta.UI.MapControl}.
  5  *
  6  * @description Locate Map Control
  7  *
  8  * @constructor
  9  * @param opt Options A list of options with which to initialize the LocateControl.
 10  * Valid options are:
 11  * <ul>
 12  *   <li>(string) position: (Inherited from {@link deCarta.UI.MapControl}), which should
 13  *       be set to one of: 'topLeft', 'topRight', 'bottomLeft', 'bottomRight' (default='topLeft')</li> 
 14  *   <li>(function) newLocationCallback: a function that will be invoked when a new location is found.</li>
 15  *   <li>(function) newLocationErrorCallback: a function that will invoked when an error is returned from the geolocation api. </li>
 16  *   <li>(bool) enableHighAccuracy: enebles the highest accuracy supported by the device. </li>
 17  * </ul>
 18  * To style this control, edit the corresponding CSS file in the css/UI dir.
 19  * @see deCarta.UI.MapControl
 20  * @see deCarta.Core.Map
 21  */
 22 deCarta.UI.LocateControl = function(opt){	 
 23     this.options = {
 24         newLocationCallback : undefined,
 25         newLocationErrorCallback : undefined,
 26         enableHighAccuracy : false,
 27         cssClass : "deCarta-control-locate",
 28         zoomAndCenter: true,
 29         layout: function(width, height, existingControls){
 30             return {top: height - this.height, left:  0};
 31         }
 32     }
 33     
 34     this.options = deCarta.Utilities.extendObject(this.options, opt);
 35     
 36     deCarta.UI.MapControl.call(this, this.options);  
 37 
 38     this.locating = false;
 39 
 40     this.location;
 41     this.geoLocation;
 42 }
 43 
 44 
 45 //Define methods to extend LocateControl
 46 deCarta.UI.LocateControl.prototype = {
 47     /**
 48      * @private
 49      */
 50     ready: false,
 51     tracking: false,
 52     /**
 53      * @private
 54      */
 55     initElements: function(){   
 56 
 57         this.domElement = document.createElement('div');
 58         this.contentElement = document.createElement('div');
 59         this.domElement.style.position = 'absolute';
 60         this.contentElement.className = 'deCarta-control-locate deCarta-control-locate-off';
 61         if(this.options.cssClass !== this.contentElement.className){
 62             this.contentElement.className = this.contentElement.className + ' ' + this.options.cssClass;
 63         }
 64 
 65         this.domElement.className = "domElement";
 66 
 67         deCarta.Touch.attachListener('tap', this.contentElement, this.geoLocate.bind(this), true);
 68 
 69         this.gpsPinOverlay = new deCarta.Core.MapOverlay({
 70             name: 'GPSPin',
 71             zIndex: 3
 72         });
 73 
 74         this.options.map.addLayer(this.gpsPinOverlay);
 75 
 76         this.height = (this.options.height || 20);
 77 
 78         this.ready = true; 
 79 
 80     },
 81 
 82     /**
 83      * @private
 84      */
 85     stopTracking: function(){
 86         navigator.geolocation.clearWatch(this.watchId);
 87         this.contentElement.className = 'deCarta-control-locate deCarta-control-locate-off';
 88         if(this.options.cssClass !== this.contentElement.className){
 89             this.contentElement.className = this.contentElement.className + ' ' + this.options.cssClass;
 90         }
 91         this.gpsPinOverlay.hide();
 92         this.tracking = false;        
 93     },
 94 
 95 
 96     /**
 97      * @private
 98      */
 99     geoLocate: function(callback){
100 
101         if (!navigator.geolocation) return;
102 
103         if (this.tracking){
104             this.stopTracking();
105             return;
106         }
107 
108         this.tracking=true;
109         var map = this.options.map;
110         var base = this.contentElement.className;
111         base=base.substring(0,base.indexOf(" "));
112         this.contentElement.className =  base + ' deCarta-control-locate-on';
113 
114         var watchId = this.watchId = navigator.geolocation.watchPosition(
115             function(geoLocationPosition){
116                 var pos = new deCarta.Core.Position(geoLocationPosition.coords.latitude, geoLocationPosition.coords.longitude);
117                 this.location = pos;
118                 this.geoLocation=geoLocationPosition;        
119 
120                 if( !this.gpsPin ){  // first time to lazy instantiate
121 
122                     this.gpsPin = new deCarta.Core.Pin({
123                         imageSrc : "img/gps.png",
124                         position:pos.clone(),
125                         xOffset: 7,
126                         yOffset: 7,
127                         text: 'accuracy '+geoLocationPosition.coords.accuracy+'M'
128                     });
129 
130                     this.gpsRadius = new deCarta.Core.Circle({
131                         position:pos.clone(),
132                         radius: geoLocationPosition.coords.accuracy //start with zero since its invisible
133                     });
134                     this.gpsPinOverlay.addObject(this.gpsRadius);
135                     this.gpsPinOverlay.addObject(this.gpsPin);
136 
137                 } else {
138                     this.gpsPin.position = pos.clone();
139                     this.gpsRadius.position = pos.clone();
140                     this.gpsRadius.radius = geoLocationPosition.coords.accuracy-10;
141                     this.gpsPinOverlay.show();
142                 }
143 
144                 var ret = true;
145                 if(this.options.newLocationCallback){
146                     //console.log("foo")
147                     ret = this.options.newLocationCallback(geoLocationPosition,pos); // hey, pass both, be generous
148                 }
149                 if(typeof callback === 'function'){
150                     ret = callback(geoLocationPosition, pos, watchId); // hey, pass both, be generous
151                 }
152 
153                 if (ret === false) {
154                     //console.log('I have fooled you');
155                     return;
156                 }
157 
158                 map.center = pos;
159                 map.zoomTo(new deCarta.Core.Radius(geoLocationPosition.coords.accuracy,"M").getZoom(pos,map.width,map.height))
160                 map.render();
161 
162             }.bind(this),
163             function(error){
164                 if(this.options.newLocationErrorCallback)
165                     this.options.newLocationErrorCallback(error);
166                 navigator.geolocation.clearWatch(watchId);
167                 var base = this.contentElement.className;
168                 base=base.substring(0,base.indexOf(" "));
169                 this.contentElement.className =  base + ' deCarta-control-locate-off';
170 
171                 deCarta.Core.EventManager.stopListeningByIdx('tap',eventIndex);
172                 this.tracking=false;
173 
174             }.bind(this),
175             {
176                 enableHighAccuracy:this.options.enableHighAccuracy
177             }
178         );
179             
180         var eventIndex = deCarta.Core.EventManager.listen('tap', function(e){    
181             navigator.geolocation.clearWatch(watchId);
182                 var base = this.contentElement.className;
183                 base=base.substring(0,base.indexOf(" "));
184                 this.contentElement.className =  base + ' deCarta-control-locate-off';
185 
186             deCarta.Core.EventManager.stopListeningByIdx('tap',eventIndex);
187             this.tracking=false;
188         }.bind(this));
189         // console.log("registered watchId",watchId);
190         // console.log("registered eventIndex",eventIndex);
191     },
192     /**
193      * This render method implements the render method from the 
194      * {@link deCarta.UI.MapControl} base class.
195      * It is responsible for rendering this control on the map,
196      * and produces a single HTML Dom Element containing the whole
197      * GUI for the control.
198      * @param {string} container The DOM element within which the control is rendered. 
199      */
200     render: function(container){
201         if (!this.ready) this.initElements();
202         this.domElement.appendChild( this.contentElement );
203         container.appendChild( this.domElement );
204         this.width = this.domElement.offsetWidth;
205         this.height = this.domElement.offsetHeight;        
206     }
207 }; 
208 //Extend the MapControl with the additional methods for LocateControl
209 deCarta.UI.LocateControl.prototype = deCarta.Utilities.inherit(deCarta.UI.LocateControl.prototype, deCarta.UI.MapControl.prototype);
