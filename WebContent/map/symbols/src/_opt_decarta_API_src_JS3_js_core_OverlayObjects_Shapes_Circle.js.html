1 /**
  2  * @class
  3  * Circle is used to display a circle on the map.
  4  * It extends {@link deCarta.Core.OverlayObject}.
  5  * Use the {@link deCarta.Core.MapOverlay}:addObject() method to add
  6  * this object to a map overlay.
  7  *
  8  * @description A map overlay object used to draw a polygon
  9  *
 10  * @constructor
 11  * @param {object} options Options. May contain one or more of the following:
 12  * <ul>
 13  *  <li>{@link deCarta.Core.Position} position: The position of the center of the circle - required</li>
 14  *  <li>radius: The radius of the circle in meters - optional, default 500</li>
 15  *  <li>strokeColor: Color of the circle border - optional, default '#0077D2'</li>
 16  *  <li>strokeWidth: Thickness of the circle border, in pixels - optional, default 2</li>
 17  *  <li>fillColor: Fill color of the circle - optional, default '#0077D2'</li>
 18  *  <li>opacity: Opacity of the circle (1.0 = opaque, 0.0 = transparent) - optional, default 0.3</li>
 19  * </ul>
 20  *
 21  * @see deCarta.Core.OverlayObject
 22  * @see deCarta.Core.MapOverlay
 23  */
 24 
 25 deCarta.Core.Circle = function(options){
 26     deCarta.Core.Shape.call(this, options);
 27     opts = {
 28         position: null,
 29         radius: 500 //radius is in meters     
 30     }
 31 
 32     this.options = deCarta.Utilities.extendObject(opts, this.options);
 33 
 34     if (!this.options.position)
 35         deCarta.Core.Exception.raise('Instantiating a circle without a position. You can see how this would not work very well. Please pass the center position as one of the options, like this: new deCarta.Core.Circle({position: new Position(37, -122)});');
 36 
 37     if ((typeof this.options.position).toLowerCase() === 'string'){
 38         this.options.position = new deCarta.Core.Position(this.options.position);
 39     } 
 40 
 41     this.setBB();
 42 }
 43 
 44 deCarta.Core.Circle.prototype = {
 45     /**
 46      * Dynamically set the radius of the circle. It will readjust accordingly.
 47      * @param {float} r the radius, in meters.
 48      */
 49     setRadius: function(r){
 50         this.options.radius = r;
 51         this.setBB();
 52     },
 53 
 54     setBB: function(){
 55         var r = this.getPixelRadius(21);
 56         var x = this.options.position.getX(21);
 57         var y = this.options.position.getY(21);
 58 
 59         var tl = new deCarta.Core.Position(0,0);
 60         var br = new deCarta.Core.Position(0,0);
 61 
 62         tl.setXY(x - r, y + r, 21);
 63         br.setXY(x + r, y - r, 21)
 64 
 65         this.boundingBox= new deCarta.Core.BoundingBox([tl, br]);
 66     },
 67 
 68     getBoundingBox: function(){
 69         return this.boundingBox;
 70     },
 71     
 72     /**
 73     * Returns true if the specified position is within the circle. 
 74     * @param pos: the position to check. 
 75     * NOTE: You can also pass an OverlayObject to this function, and the position will be extracted automatically
 76     */
 77     contains: function(pos){
 78         if (!pos.getLon) {
 79             if (pos.getCenter){
 80                 pos = pos.getCenter();
 81             } else {
 82                 throw new deCarta.Core.Exception('The position passed to the "contains" function is not a position.');
 83             }
 84         }
 85         var d = deCarta.Utilities.getPointDistance(this.options.position, pos) * 1000;        
 86         return (d <= this.options.radius);
 87     },
 88 
 89     shouldProcessEvent: function(position){
 90         return this.contains(position);
 91     },
 92 	/** Retrieve the radius of the circle.
 93 	 * @return {float} Radius, in meters
 94 	 */
 95     getRadius: function(){
 96         return this.options.radius;
 97     },
 98 
 99 
100     getGML:function(){
101         var xml = '';
102         xml += "<gml:CircleByCenterPoint numArc='1'>";
103         xml += "<gml:pos>"+this.options.position+"</gml:pos>";
104         xml += "<gml:radius uom='m'>"+this.options.radius+"</gml:radius>";
105         xml += "</gml:CircleByCenterPoint>";
106 
107         return xml;
108     },
109 
110     toJSON: function(){
111         var obj = {
112             type: "CIRCLE",
113             radius: this.options.radius,
114             position: this.getPosition()
115         }
116         
117         return obj;
118     },    
119 
120     /** Gets the radius of the circle, converted to pixels
121 	 * @param z {int} Zoom level for which we want the pixel dimension
122 	 * @return {int} Radius, in pixels
123 	 */
124     getPixelRadius: function(z){
125         return 1 / (deCarta.Utilities.metersPerPixelAtZoom(this.options.position, z) * (1 / this.options.radius));
126     },
127 
128     /**
129      * @private
130      */
131     renderCanvas: function(x, y, z, tiles){
132         
133         if (!this.canvasElement){            
134             this.canvasElement = document.createElement('canvas');
135 
136             this.canvasElement.setAttribute("id", ("deCarta-shape-" + Math.random()).replace(".", ""));
137             this.canvasElement.className = 'deCarta-circle';
138             this.canvasElement.style.zIndex = this.zIndex;
139             this.canvasElement.style.position = 'absolute';
140             return this.canvasElement;
141         }
142 
143         if (!this.canvasElement.getContext){
144             try {
145                 G_vmlCanvasManager.initElement(this.canvasElement);
146             } catch (e){
147                 deCarta.Core.Exception.raise('Error in drawing the shape - have you added exCanvas to your page?');
148             }
149         }
150 
151         //let's figure out if this is in view first
152         var viewport = deCarta.Window.getViewport();
153         var pxRadius = this.getPixelRadius(z);
154         var specs = this.elementSpecs(x -pxRadius, y - pxRadius,z);
155     
156         if (!this.lastSpecs) this.lastSpecs = {}
157         this.domPosition((specs.x), (specs.y), this.canvasElement);
158 
159         if (x + pxRadius < 0 || y + pxRadius < 0 || x - pxRadius > viewport.width || y - pxRadius > viewport.height){
160             return false;
161         }
162 
163         var newWidth = specs.width;//(pxRadius * 2) + (this.padding * 2);
164         var newHeight = specs.height; //(pxRadius * 2) + (this.padding * 2);
165 
166         if (this.canvasElement.width != newWidth || 
167             this.canvasElement.height != newHeight ||
168             this.lastSpecs.xOfs != specs.xOfs ||
169             this.lastSpecs.yOfs != specs.yOfs){
170             //if we are here, we are in view. So... what do we do
171             this.canvasElement.width = specs.width;//(pxRadius * 2) + (this.padding * 2);
172             this.canvasElement.height = specs.height;//(pxRadius * 2) + (this.padding * 2);
173 
174             var ctx = this.canvasElement.getContext("2d");
175 
176             //draw the circle
177             ctx.beginPath();
178             ctx.arc(pxRadius - specs.xOfs + this.padding, pxRadius - specs.yOfs + this.padding, pxRadius, 0, Math.PI*2, true);
179             ctx.closePath();
180 
181             ctx.strokeStyle = deCarta.Utilities.makeRGBA(this.options.strokeColor, this.options.strokeOpacity);
182             ctx.lineWidth = this.options.strokeWidth;
183             ctx.fillStyle = deCarta.Utilities.makeRGBA(this.options.fillColor, this.options.fillOpacity);
184             
185             ctx.fill();
186             ctx.stroke();
187         }
188         this.lastSpecs = specs;
189 
190         return this.canvasElement;
191     },
192 
193     /**
194      * @private
195      */
196     renderSVG: function(x, y, z, tiles){
197         if (!this.svgElement) {        
198             
199             this.container = document.createElement('div');
200             this.container.id = ("deCarta-circle-container-" + Math.random()).replace(".", "");            
201                         
202             this.svgElement = document.createElementNS("http://www.w3.org/2000/svg", "svg");
203             this.svgElement.setAttribute("version", "1.1");                
204 
205             this.svgElement.setAttribute("id", ("deCarta-circle-svg-" + Math.random()).replace(".", ""));
206            
207             this.container.style.zIndex = this.zIndex;            
208             this.container.style.position = 'absolute';
209             
210             this.container.appendChild(this.svgElement); 
211             return this.container;
212         }
213         if (!this.circleElement) {
214             
215             this.circleElement = document.createElementNS("http://www.w3.org/2000/svg", "circle");            
216             this.svgElement.appendChild(this.circleElement);            
217         } 
218         var viewport = deCarta.Window.getViewport();
219         var pxRadius = this.getPixelRadius(z);
220         var specs = this.elementSpecs(x -pxRadius, y - pxRadius,z);
221         
222         if (x + pxRadius < 0 || y + pxRadius < 0 || x - pxRadius > viewport.width || y - pxRadius > viewport.height){
223             return false;
224         }
225         var newWidth = specs.width;//(pxRadius * 2) + (this.padding * 2);
226         var newHeight = specs.height; //(pxRadius * 2) + (this.padding * 2);
227         
228         this.container.style.width = newWidth + 'px';
229         this.container.style.height = newHeight + 'px';
230         
231         this.circleElement.setAttribute('cx', pxRadius - specs.xOfs + this.padding);
232         this.circleElement.setAttribute('cy', pxRadius - specs.yOfs + this.padding);
233         this.circleElement.setAttribute('r', pxRadius);
234         this.circleElement.setAttribute('fill-opacity', this.options.fillOpacity);
235         this.circleElement.setAttribute('stroke-opacity', this.options.strokeOpacity);
236         this.circleElement.setAttribute('style', "fill:"+this.options.fillColor+";stroke:"+this.options.strokeColor+";stroke-width:"+this.options.strokeWidth+";position: absolute;");
237 
238         this.domPosition((specs.x), (specs.y), this.container);
239         
240         return this.container;
241     },
242     
243     renderVML: function(x, y, z){
244         
245         if (!this.container) {        
246             
247             this.container = document.createElement('div');
248             this.container.id = ("deCarta-circle-container-" + Math.random()).replace(".", "");            
249             this.shapeId = ("deCarta-circle-" + Math.random()).replace(".", "");            
250               
251             this.container.unrender = function(){
252                 this.container = null;
253             }.bind(this)
254 
255             this.container.style.zIndex = this.zIndex;            
256             this.container.style.position = 'absolute';
257             
258             return this.container;
259         }
260         
261         var viewport = deCarta.Window.getViewport();
262         var pxRadius = this.getPixelRadius(z);
263         
264         if (x + pxRadius < 0 || y + pxRadius < 0 || x - pxRadius > viewport.width || y - pxRadius > viewport.height){
265             return false;
266         }
267         var newWidth = Math.round((pxRadius * 2) + (this.padding * 2));
268         var newHeight = Math.round((pxRadius * 2) + (this.padding * 2));
269         
270         this.container.style.width = newWidth + 'px';
271         this.container.style.height = newHeight + 'px';
272         
273         var fillStr = '<v:fill opacity="' + Math.round(this.options.fillOpacity * 100) + '%" color="'+this.options.fillColor+'"/>'
274         var strokeStr = '<v:stroke opacity="' + Math.round(this.options.strokeOpacity * 100) + '%" weight="'+ this.options.strokeWidth+'px" color="'+this.options.strokeColor+'" />';        
275         var vmlStr = '<v:oval id="'+this.shapeId +'" style="position: absolute;top:0;left:0;width:'+newWidth+'px;height:'+newHeight+'px"> ' + fillStr + strokeStr + '</v:oval>';
276         
277         this.container.innerHTML = vmlStr;
278         
279         this.domPosition((x - this.padding - pxRadius), (y - this.padding - pxRadius), this.container);
280         var elid = this.shapeId ;
281         /*setTimeout(function(){*/
282             var el = document.getElementById(elid);
283             if (el) el.outerHTML = el.outerHTML;
284         /*}, 1);*/
285 
286         return this.container;
287         
288     }
