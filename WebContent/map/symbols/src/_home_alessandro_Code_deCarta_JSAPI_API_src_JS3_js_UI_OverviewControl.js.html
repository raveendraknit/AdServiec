1 /**
  2  * @class
  3  * Add a map overview window to the map.
  4  * This class inherits from {@link deCarta.UI.MapControl}.
  5  *
  6  * @description Overview Map Control
  7  *
  8  * The OverviewControl map control provides a small window with a zoomed-out overview map as an overlay on the main map.
  9  * This class inherits from {@link deCarta.UI.MapControl}.
 10  * @param opt Options A list of options with which to initialize the OverviewControl.
 11  * Valid options are:
 12  * <ul>
 13  *   <li>(string) position: (Inherited from {@link deCarta.UI.MapControl}), which should
 14  *       be set to one of: 'topLeft', 'topRight', 'bottomLeft', 'bottomRight' (default='topLeft')</li>
 15  * </ul>
 16  * To style this control, edit the corresponding CSS file in the css/UI dir.
 17  * @see deCarta.UI.MapControl
 18  * @see deCarta.Core.Map 
 19  */
 20 deCarta.UI.OverviewControl = function(opt){
 21     
 22     this.options = {
 23         cssClass : 'deCarta-control-overview',
 24         layout: function(width, height, existingControls){
 25             return {top: height - 124, left:  width - 124}
 26         }
 27     }
 28     
 29     this.options = deCarta.Utilities.extendObject(this.options, opt);
 30     
 31     deCarta.UI.MapControl.call(this, this.options);  
 32 }
 33 
 34 
 35 //Define methods to extend OverviewControl
 36 deCarta.UI.OverviewControl.prototype = {
 37 
 38     /**
 39 	 * @private
 40 	 */
 41     ready: false,
 42 	
 43     /**
 44 	 * @private
 45 	 */
 46     ZOOM_DIFF: 5,
 47 
 48     /**
 49 	 * @private
 50 	 */
 51     initElements: function(){   
 52         
 53         this.domElement = document.createElement('div');
 54         this.contentElement = document.createElement('div');
 55         this.domElement.style.position = 'absolute';
 56         this.contentElement.className = 'deCarta-control-overview';
 57         if(this.options.cssClass !== this.contentElement.className){
 58             this.contentElement.className = this.contentElement.className + ' ' + this.options.cssClass;
 59         }
 60         this.domElement.className = "domElement";
 61                 
 62         // get the honest to goodness styles
 63         var w = parseInt(deCarta.Utilities.getComputedStyle(this.contentElement,"width"));
 64         var h = parseInt(deCarta.Utilities.getComputedStyle(this.contentElement,"height"));
 65         this.width = w ? w : 124;
 66         this.height = h ? h : 124;
 67 
 68         this.contentElement.style.width = this.width+'px';
 69         this.contentElement.style.height =  this.height+'px';
 70 
 71         this.mapControl = document.createElement('div');
 72         this.mapControl.id = 'mapOverviewWindow';
 73         this.mapControl.style.position="relative";
 74         this.mapControl.style.width = this.width+'px';
 75         this.mapControl.style.height = this.height+'px';
 76         this.contentElement.appendChild(this.mapControl);
 77 
 78         this.mapControlHider = document.createElement('div');
 79         this.mapControlHider.className = 'deCarta-control-overview-hide';
 80         this.contentElement.appendChild(this.mapControlHider);
 81 
 82         /*var hiderHeight = parseInt(deCarta.Utilities.getComputedStyle(this.mapControlHider,"height"));
 83         var hiderWidth = parseInt(deCarta.Utilities.getComputedStyle(this.mapControlHider,"width"));*/
 84 
 85         this.domElement.open=true;
 86         deCarta.Touch.attachListener('tap', this.contentElement, 
 87             function(){
 88                 var t = parseInt(this.domElement.style.top);
 89                 var l = parseInt(this.domElement.style.left);
 90                 //moved to here because of IE 
 91                 var hiderHeight = parseInt(deCarta.Utilities.getComputedStyle(this.mapControlHider,"height"));
 92                 var hiderWidth = parseInt(deCarta.Utilities.getComputedStyle(this.mapControlHider,"width"));
 93                 var w = parseInt(deCarta.Utilities.getComputedStyle(this.contentElement,"width"));
 94                 var h = parseInt(deCarta.Utilities.getComputedStyle(this.contentElement,"height"));             
 95                 if(this.domElement.open){              
 96                     this.height = hiderHeight;      
 97                     this.width = hiderWidth;      
 98                     this.domElement.style.top = (t+h-hiderHeight)+'px';
 99                     this.domElement.style.left = (l+w-hiderWidth)+'px';
100                     this.domElement.open=false;
101                     this.mapControlHider.className = this.mapControlHider.className+' deCarta-control-overview-show';
102                 } else {
103                     this.height = h;      
104                     this.width = w;      
105                     this.domElement.style.top = (t-h+hiderHeight)+'px';
106                     this.domElement.style.left = (l-w+hiderWidth)+'px';
107                     this.domElement.open=true;
108                     this.mapControlHider.className = 'deCarta-control-overview-hide';
109                 }
110             }.bind(this), true);
111         this.ready = true;         
112     },
113     
114     /**
115 	 * This render method implements the render method from the 
116 	 * {@link deCarta.UI.MapControl} base class.
117 	 * It is responsible for rendering this control on the map,
118 	 * and produces a single HTML Dom Element containing the whole
119 	 * GUI for the control.
120 	 * @param {HTMLDOMElement} container The container which holds this control
121 	 */
122     render: function(container){
123         //console.log("render ovrMap",this.ready, this.tinyMap)
124         if (!this.ready) {
125             this.initElements();    
126             this.domElement.appendChild( this.contentElement );
127             container.appendChild( this.domElement );                
128         }
129         
130         if (!this.tinyMap) {
131             var p = new deCarta.Core.Position(this.options.map.center.getLat(), this.options.map.center.getLon());
132             
133             this.tinyMap = new deCarta.Core.Map({
134                 id: this.mapControl.id,
135                 skipBindingToServer: true,
136                 skipResources: true,
137                 resizeable: false,
138                 draggable: this.options.style == 'mobile',
139                 doubleTapZoom: false,
140                 digitalZoom: false,
141                 easing: false,                
142                 center:  p,
143                 zoom: Math.max(1, this.options.map.zoom - this.ZOOM_DIFF),
144                 
145                 onReady: function(){
146                     
147                     this.tinyMap.addOverlay(this.areaOverlay);
148                     this.areaOverlay.addObject(this.area);  
149 
150                     var listener = this.sync.bind(this);        
151 
152                     this.options.map.onmoveend(listener);
153                     this.options.map.onmove(listener);
154                     this.options.map.onzoomend(listener);                    
155 
156                 }.bind(this)
157             });
158             
159             this.areaOverlay = new deCarta.Core.MapOverlay({name : 'view'});
160             this.area = new deCarta.Core.Polygon({
161                 vertices: this.options.map.getVisibleRect(),
162                 draggable: !(this.options.style == 'mobile'),
163                 scroll: this.tinyMap,
164                 onDrop: function(p){
165                     this.preventRecenter = true;
166                     this.options.map.centerOn(p);
167                     this.tinyMap.centerOn(p);
168                 }.bind(this)
169             });            
170 
171         }        
172     },
173     
174     /**
175 	 * @private
176 	 */
177     sync: function(param){
178                 
179         if (param.map != this.tinyMap){
180             this.area.setVertices(this.options.map.getVisibleRect());
181             
182             if (this.recenterFromMain) {
183                 this.recenterFromMain = false;
184                 return;
185             }                        
186             if (param.zoom) this.tinyMap.zoomTo(Math.max(1, param.zoom - this.ZOOM_DIFF),null,true);
187             if (param.center) this.tinyMap.centerOn(param.center, {animated: false});
188         
189         }
190         this.tinyMap.render();
191     }
192 	
193 	
194 }; //end OverviewControl prototype
195 
196 //Extend the MapControl with the additional methods for OverviewControl
197 deCarta.UI.OverviewControl.prototype = deCarta.Utilities.inherit(deCarta.UI.OverviewControl.prototype, deCarta.UI.MapControl.prototype);
