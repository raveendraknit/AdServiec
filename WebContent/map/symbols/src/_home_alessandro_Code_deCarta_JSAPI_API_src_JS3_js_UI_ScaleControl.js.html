1 /**
  2  * @class
  3  * Add a scale control to the map.
  4  * This class inherits from {@link deCarta.UI.MapControl}.
  5  *
  6  * @description Scale Map Control
  7  *
  8  * The ScaleControl provides a control to indicate measure of distance.
  9  * This class inherits from {@link deCarta.UI.MapControl}.
 10  * @param opt Options A list of options with which to initialize the ScaleControl.
 11  * Valid options are:
 12  * <ul>
 13  *   <li>(string) position: (Inherited from {@link deCarta.UI.MapControl}), which should
 14  *       be set to one of: 'topLeft', 'topRight', 'bottomLeft', 'bottomRight'
 15  *       (default='bottomRight')</li>
 16  * </ul>
 17  * To style this control, edit the corresponding CSS file in the css/UI dir.
 18  * @see deCarta.UI.MapControl
 19  * @see deCarta.Core.Map
 20  */
 21 deCarta.UI.ScaleControl = function(opt){
 22     this.options = {
 23         layout: function(width, height, existingControls){
 24             return {top: height - this.height, left:  width - this.width};
 25         },
 26         position: "bottomRight"
 27     }
 28     
 29     this.options = deCarta.Utilities.extendObject(this.options, opt);
 30     deCarta.UI.MapControl.call(this, this.options);  
 31 };
 32 
 33 //Define methods to extend CopyrightControl
 34 deCarta.UI.ScaleControl.prototype = {
 35 	
 36     /**
 37 	 * @private
 38 	 */
 39     ready: false,
 40 
 41     /**
 42 	 * @private
 43      */
 44     initElements: function(){   
 45 
 46         this.domElement = document.createElement('div');
 47         this.contentElement = document.createElement('div');
 48         this.domElement.style.position = 'absolute';
 49         this.contentElement.className = 'deCarta-control-scale';
 50         if(this.options.cssClass !== this.contentElement.className){
 51             this.contentElement.className = this.contentElement.className + ' ' + this.options.cssClass;
 52         }
 53             this.domElement.className = "domElement";
 54 
 55         this.textImperial = document.createElement("div");
 56         this.scaleImperial = document.createElement("div");
 57         this.scaleImperial.className = "scaleImperial";
 58     
 59         this.textMetric = document.createElement("div");
 60         this.scaleMetric = document.createElement("div");
 61         this.scaleMetric.className = "scaleMetric";
 62         
 63         this.contentElement.appendChild(this.textImperial);
 64         this.contentElement.appendChild(this.scaleImperial);
 65         this.contentElement.appendChild(this.scaleMetric);
 66         this.contentElement.appendChild(this.textMetric);
 67 
 68         this.setScale({
 69             center: this.options.map.center,
 70             zoom: this.options.map.zoom
 71         });
 72 
 73         this.ready = true; 
 74         
 75         this.options.map.onzoomend(this.setScale.bind(this));
 76         this.options.map.onmoveend(this.setScale.bind(this));
 77         
 78     },
 79 
 80     setScale: function(params){
 81         if(!this.domElement) return;
 82         var scale = this.calculate(params.center, params.zoom);
 83         this.textImperial.innerHTML=scale.feetDisp;
 84         this.scaleImperial.style.width=scale.widthFeet + "px";
 85         this.scaleMetric.style.width=scale.widthMeter + "px";
 86         this.textMetric.innerHTML=scale.meterDisp;
 87 
 88     },
 89 
 90     /**
 91 	 * This render method implements the render method from the 
 92 	 * {@link deCarta.UI.MapControl} base class.
 93 	 * It is responsible for rendering this control on the map,
 94 	 * and produces a single HTML Dom Element containing the whole
 95 	 * GUI for the control.
 96 	 * @param {string} container The DOM element within which the control is rendered. 
 97 	 */
 98     render: function(container){
 99         if (!this.ready) this.initElements();
100         this.domElement.appendChild( this.contentElement );
101         container.appendChild( this.domElement );
102         this.width = this.domElement.offsetWidth;
103         this.height = this.domElement.offsetHeight;
104     },
105     
106     calculate: function(center, z){
107         var normz = 20-deCarta.Utilities.normalizeZoom(z);
108         var metersPerPixel = deCarta.Utilities.metersPerPixelAtZoom(center,z);
109         
110         var meterDisp=deCarta.ScaleControl_Params.Scale[normz].meterDisp;
111         var feetDisp=deCarta.ScaleControl_Params.Scale[normz].feetDisp;
112         var widthMeter=deCarta.ScaleControl_Params.Scale[normz].meter/metersPerPixel;
113         var widthFeet=deCarta.ScaleControl_Params.Scale[normz].feet*deCarta.ScaleControl_Params.meterPerFeet/metersPerPixel;
114 
115         for(; normz > 0 && (widthMeter > this.width || widthFeet > this.width); ){
116             normz--;
117             meterDisp=deCarta.ScaleControl_Params.Scale[normz].meterDisp;
118             feetDisp=deCarta.ScaleControl_Params.Scale[normz].feetDisp;
119             widthMeter=deCarta.ScaleControl_Params.Scale[normz].meter/metersPerPixel;
120             widthFeet=deCarta.ScaleControl_Params.Scale[normz].feet*deCarta.ScaleControl_Params.meterPerFeet/metersPerPixel;
121         }
122 
123         if(widthMeter > this.width || widthFeet > this.width){
124             var meters = metersPerPixel * this.width;
125             var feet = meters / deCarta.ScaleControl_Params.meterPerFeet;
126             feetDisp=  Math.round(feet*1000)/1000 + " ft";
127             meterDisp= Math.round(meters*1000)/1000 + " m";
128             widthFeet= this.width;
129             widthMeter= this.width;
130         }
131 
132         return {
133             feetDisp: feetDisp,
134             meterDisp: meterDisp,
135             widthFeet: widthFeet,
136             widthMeter: widthMeter
137         };
138       
139     }
140 
141 }; //end CopyrightControl prototype
