1  /**
  2  * @class
  3  * Swath is a map overlay object used to draw a complex line
  4  * (such as a route) on the map.
  5  * It extends {@link deCarta.Core.Polyline}.
  6  * Use the {@link deCarta.Core.MapOverlay}:addObject() method to add 
  7  * this object to a map overlay.
  8  * (for example a route).
  9  *
 10  * @constructor
 11  * @param {object} opts Options. May contain one or more of the following
 12  * <ul>
 13  *  <li>lineGeometry: Array of deCarta.Mobile.Position</li>
 14  *  <li>fillColor: A Swath has thickness, and therefore has a fill color
 15  *      - optional, default '#000'</li>
 16  *  <li>radius: A Swath has thickness specified by the radius option. </li> 
 17  *  <li>strokeColor: Color hat will be used to stroke the line
 18  *      - optional, default '#0000FF'</li>
 19  *  <li>strokeSize: Size of the line, in pixels - optional, default 6</li>
 20  *  <li>strokeOpacity: Opacity of the line (1.0 = opaque, 0.0 = transparent)
 21  *      - optional, default 0.6</li>
 22  *  <li>strokeLineCap: <b>NOTE: SVG Rendering only</b> Style for the line endings. Can be one of : "butt", "round", "square" (see http://www.w3.org/TR/SVGTiny12/painting.html)</li>
 23  *  <li>strokeLineJoin: <b>NOTE: SVG Rendering only</b> Shape which shall be used at the corners of shapes when they are stroked. Can be one of: "miter", "round, "bevel". (see http://www.w3.org/TR/SVGTiny12/painting.html)</li> 
 24  * </ul>
 25  *
 26  * @see deCarta.Core.OverlayObject
 27  * @see deCarta.Core.MapOverlay
 28  *
 29  * @description An OverlayObject used to display a Swath on a MapOverlay
 30  *
 31 */
 32 
 33 deCarta.Core.Swath = function(options){
 34 
 35     var opts = {
 36         lineGeometry: null,        
 37         radius: 100,
 38         strokeLineCap: "round",
 39         strokeWidth: 8,
 40         strokeLineJoin: "round",
 41         strokeColor: "#0000FF",
 42         strokeOpacity: 0.15,    
 43         granularity: 7 //not documented
 44     }
 45 
 46     var options = this.options = deCarta.Utilities.extendObject(opts, options);
 47 
 48     deCarta.Core.Shape.call(this, options);
 49 
 50     if (!options.lineGeometry || options.lineGeometry.length == 0)
 51         deCarta.Core.Exception.raise('You need to provide a lineGeometry to instantiate a Swath');
 52 
 53     this.pointIndex = [];
 54     this.genGeom = []; //array of generalized geometry, per zoom level 
 55 
 56     this.positions = [];
 57 
 58     this.lastRenderedPointSet = null;
 59     
 60     if (typeof this.options.lineGeometry[0] == 'object' && this.options.lineGeometry[0].getLat){
 61         this.positions = this.options.lineGeometry.slice();        
 62     } else {
 63         for(var i = 0; i< this.options.lineGeometry.length; i++){
 64             var ll = this.options.lineGeometry[i].split(' ');
 65             var tPos = new deCarta.Core.Position(parseFloat(ll[0]), parseFloat(ll[1]));
 66             this.positions.push(tPos);
 67         }
 68     }
 69 
 70     this.type = 'shape';
 71 
 72     this.findBounds();
 73     this.padding=5;
 74 
 75     this.boundingBox = new deCarta.Core.BoundingBox(this.positions);
 76 }
 77 
 78 //Define methods to extend Swath
 79 deCarta.Core.Swath.prototype = {
 80   
 81     /**
 82      * Dynamically set the radius of the circle. It will readjust accordingly.
 83      * @param {float} r the radius, in meters.
 84      */
 85     setSwath: function(radius,geometry){
 86         this.options.lineGeometry = geometry;
 87         this.options.radius = radius;
 88         this.hide()
 89         this.show()
 90     },
 91 
 92     /** Gets the radius of the circle, converted to pixels
 93      * @param z {int} Zoom level for which we want the pixel dimension
 94      * @return {int} Radius, in pixels
 95      */
 96     getPixelRadius: function(z){
 97         return 1 / (deCarta.Utilities.metersPerPixelAtZoom(this.positions[0], z) * (1 / this.options.radius));
 98     },
 99 
100     /**
101      * @private
102      */
103     renderVML: function(x, y, z, tiles){
104         if (!this.container){
105             this.findBounds();
106             this.container = document.createElement('div');
107             this.container.id = ("deCarta-Swath-container-" + Math.random()).replace(".", "");            
108                                         
109             this.container.style.position = 'absolute';
110             this.container.unrender = function(){
111                 this.container = null;
112             }.bind(this)             
113             return this.container;
114         }
115 
116         this.findBounds();  
117         this.indexAndThinPoints(z);
118 
119         var nz = deCarta.Utilities.normalizeZoom(z);
120         
121         var pStr = '';
122         var dX = this.topLeftPoint.getX(z);
123         var dY = this.topLeftPoint.getY(z);
124         
125         var points = this.preparePoints(z, dX, dY, tiles);       
126 
127         for (var i = 0; i < points.length; i++){
128             var p = points[i];
129             pStr += Math.round(p.x) + ',' + Math.round(p.y) + ' ';
130         }        
131 
132         var fillStr = '<v:fill opacity="0%" color="'+this.options.fillColor+'"/>'
133         var strokeStr = '<v:stroke opacity="' + Math.round(this.options.strokeOpacity * 100) + '%" weight="'+ this.options.strokeWidth+'px" color="'+this.options.strokeColor+'" />';
134         var vmlStr = '<v:polyline points="' + pStr + '">'+strokeStr+ fillStr +'</v:polyline>';
135         this.container.style.zIndex = this.zIndex;   
136         this.container.innerHTML = vmlStr;
137 
138         this.clickableElement = this.container.firstChild;
139 
140         this.domPosition((x - 5),(y - 5), this.container);
141         return this.container;
142     },
143 
144     /**
145      * @private
146      */
147     renderSVG: function(x, y, z, tiles){
148 
149         if (!this.svgElement) {
150 
151             this.findBounds();
152             this.svgElement = document.createElementNS("http://www.w3.org/2000/svg", "svg");
153             this.svgElement.setAttribute("version", "1.1");
154             this.svgElement.setAttribute("id", ("deCarta-shape-svg-" + Math.random()).replace(".", ""));            
155         }
156 
157         if (!this.lineElement){
158             this.lineElement = document.createElementNS("http://www.w3.org/2000/svg", "polyline");
159             this.svgElement.appendChild(this.lineElement);
160         }
161         this.indexAndThinPoints(z);
162         var radialPix = Math.round(this.getPixelRadius(z));
163         this.padding = radialPix;
164         var nz = deCarta.Utilities.normalizeZoom(z);        
165         var pStr = '';
166         var dX = this.topLeftPoint.getX(z);
167         var dY = this.topLeftPoint.getY(z);                
168         var points = this.preparePoints(z, dX, dY, tiles);
169         for (var i = 0; i < points.length; i++){
170             var p = points[i];
171             pStr += p.x + ',' + p.y + ' ';
172         }
173         this.lineElement.setAttribute('points', pStr);
174         this.lineElement.setAttribute('style', "stroke-linecap:"+this.options.strokeLineCap+
175             ";stroke-linejoin:"+this.options.strokeLineJoin+
176             ";fill:none;stroke:"+this.options.strokeColor+
177             ";stroke-width:"+radialPix*2+
178             ";stroke-opacity:"+this.options.strokeOpacity);
179 
180         var dim = this.getSize(z);
181 
182         this.svgElement.setAttribute('style', "position: absolute; width: " +
183             (dim.width + radialPix*2) + "px; height: " +
184             (dim.height + radialPix*2) + "px;");
185 
186         this.svgElement.style.zIndex = this.zIndex;
187         this.domPosition((x - radialPix),(y - radialPix), this.svgElement);
188         this.clickableElement = this.lineElement;
189         return this.svgElement;
190 
191     },
192     /**
193      * @private
194      */
195     renderCanvas: function(x, y, z, tiles){
196 
197         var radialPix = this.getPixelRadius(z);
198         if (!this.canvasElement){
199             this.findBounds();
200             this.canvasElement = document.createElement('canvas');
201             this.canvasElement.setAttribute("id", ("deCarta-shape-" + Math.random()).replace(".", ""));
202             this.canvasElement.className = 'deCarta-Swath';        
203             this.canvasElement.style.zIndex = this.zIndex;
204             this.canvasElement.style.position = 'absolute';            
205            
206             return this.canvasElement;
207         }
208 
209         if (!this.canvasElement.getContext){
210             try {
211                 G_vmlCanvasManager.initElement(this.canvasElement);
212             } catch (e){
213                 deCarta.Core.Exception.raise('Error in drawing the shape - have you added exCanvas to your page?');
214             }
215         }        
216 
217         this.indexAndThinPoints(z);
218                 
219         var dim = this.getSize(z);
220         var topClip = 0;
221         var leftClip = 0;
222 
223         //clipping .. otherwise this is terrible.
224 
225         if (x < 0) {
226             leftClip = x;
227             x = 0;
228         }
229         if (y < 0) {
230             topClip = y;
231             y = 0;
232         }
233         if (x + dim.width > window.innerWidth) dim.width = window.innerWidth - x;
234         if (y + dim.height > window.innerHeight) dim.height = window.innerHeight - y;
235 
236         //this.canvasElement.setAttribute('style', "position: absolute;"); //why?
237         this.domPosition((x - radialPix), (y - radialPix), this.canvasElement);
238 
239         //maybe we can skip this rendering? depends.
240         if (z == this.lastZ &&
241             leftClip == this.lastLeftClip &&
242             topClip == this.lastTopClip &&
243             dim.width == this.lastWidth &&
244             dim.height == this.lastHeight){
245             return this.canvasElement;
246         }
247         this.lastZ = z;
248         this.lastLeftClip = leftClip;
249         this.lastTopClip = topClip;
250         this.lastWidth = dim.width;
251         this.lastHeight = dim.height;
252 
253         this.canvasElement.width = dim.width + (radialPix*2);
254         this.canvasElement.height = dim.height+ (radialPix*2);
255         
256         var dX = this.topLeftPoint.getX(z) - leftClip - radialPix;
257         var dY = this.topLeftPoint.getY(z) + topClip + radialPix;
258 
259         var ctx = this.canvasElement.getContext('2d');
260 
261         ctx.beginPath();        
262 
263         var points = this.preparePoints(z, dX, dY, tiles);
264 
265         var moved = false;
266         
267         for (var i = 0; i < points.length; i++){
268             p = points[i];
269             if (!moved) {
270                 ctx.moveTo(p.x, p.y);
271                 moved = true;
272             }
273             else ctx.lineTo(p.x, p.y);
274         }
275 
276         ctx.strokeStyle = this.options.strokeColor;
277         ctx.lineWidth = radialPix*2; //this.options.strokeWidth;
278         ctx.globalAlpha = this.options.strokeOpacity;
279         ctx.lineCap = 'round';
280 
281         ctx.stroke();
282         
283         return this.canvasElement;
284 
285     }
286 }; //end Swath prototype
