1 /**
  2  * @private
  3  * The TileGrid class calculates the required tiles to cover an area of the map
  4  * given the center (a Position) the zoom level (float) and using the current
  5  * width and height of the map. 
  6  * The tiles are arranged in an array, ordered by the distance of each tile
  7  * from the center. 
  8  * The grid object is used by the Map and passed along to the Layers and Overlays
  9  * when a rendering is required.
 10  *
 11  * @description Class for calculating required tiles
 12  *
 13  */
 14 deCarta.Core.TileGrid = function(){
 15 
 16     //the array of tiles
 17     this.tiles = [];
 18 
 19     this.zoom = 0;
 20 
 21     //width and eight of the grid
 22     this.width = this.height = 0;
 23 
 24 }
 25 
 26 
 27 deCarta.Core.TileGrid.prototype = {
 28 
 29     /**
 30      * Prepare a tile grid from the current width, height and the provided zoom and centerpoint.
 31      * @param {deCarta.Core.Position}center  the position around which we want to center the map
 32      * @param {float} zoom the current zoom level
 33      */
 34     prepare: function (center, zoom){
 35 
 36         this.zoom = zoom;
 37 
 38         var width = this.width;
 39         var height= this.height;        
 40 
 41         var centerX = Math.round(center.getX(zoom));
 42         var centerY = Math.round(center.getY(zoom));
 43 
 44         //how big are the tiles at this zoom level
 45         var tileSize = deCarta.Utilities.tileSizeForZoom(zoom);
 46         var worldSize = Math.pow(deCarta.Utilities.normalizeZoom(zoom), 2);        
 47 
 48         //N and E of center tile on this grid
 49         var cE = Math.floor(centerX / tileSize);
 50         var cN = Math.floor(centerY / tileSize);
 51 
 52         //offset of center in tile
 53         var ofsX = centerX % tileSize;
 54         var ofsY = centerY % tileSize;
 55 
 56         /* This is necessary because : 
 57             - when the map is in the negative, say at -256px 
 58             - the ofs is correct at 0
 59             - but the tile is wrong, at Floor(-256 / 256) = -1        
 60         */
 61         if (ofsX == 0 && cE < 0){
 62             cE -=1;
 63         }
 64         //not sure why it not happen in N i ask the question the suggestion to the him
 65 
 66         this.centerTile = {
 67             E: cE,
 68             N: cN,
 69             Z: zoom,
 70             lat: center.getLat(),
 71             lon: center.getLon()
 72         } 
 73 
 74         this.xTiles = Math.floor((width + (2 * tileSize)) / tileSize);
 75         this.yTiles = Math.floor((height + (2 * tileSize)) / tileSize);
 76 
 77         var centerTileGridX = Math.ceil(((width / 2) - ofsX - tileSize) / tileSize);
 78         var centerTileGridY = Math.ceil(((height / 2) + ofsY- tileSize) / tileSize);
 79 
 80         this.gridX = ((width / 2) - ofsX - tileSize) - (tileSize * centerTileGridX);
 81         this.gridY = ((height / 2) + ofsY- tileSize) - (tileSize * centerTileGridY);
 82 
 83         this.tiles = [];
 84 
 85         for (var x = 0; x < this.xTiles; x++){
 86             for (var y = 0; y < this.yTiles; y++){
 87                 
 88                 var tile = {}
 89 
 90                 tile.size = tileSize;
 91 
 92                 tile.E = (cE - (centerTileGridX - x)) ;
 93                 tile.N = (cN + (centerTileGridY - y)) ;    
 94 
 95                 tile.Z = zoom;
 96                 
 97                 if (cN < 0) tile.N += 1;
 98                 if (cE >= 0) tile.E -= 1;
 99 
100                 tile.dX = this.gridX + (tileSize * x);
101                 tile.dY = this.gridY + (tileSize * y);
102 
103                 if (y == 0 && x == 0){
104                     //get the position for top left corner of grid
105                     this.gridPosition = deCarta.Utilities.getTilePosition(tile.E, tile.N, tile.Z);
106                 }
107 
108 
109                 this.tiles.push(tile);
110             }
111         }
112 
113         //sort tiles by distance from ctr
114         this.tiles.sort(function(a, b){
115             var distAfromCtr = Math.sqrt(Math.pow(a.E - cE, 2)  + Math.pow(a.N - cN, 2));
116             var distBfromCtr = Math.sqrt(Math.pow(b.E - cE, 2)  + Math.pow(b.N - cN, 2));
117 
118             return distAfromCtr - distBfromCtr;
119         });
120 
121         return;
122     },
123     
124     /**
125      * Get all tiles in the grid as an array 
126      * @return {object} The files are returned in an object with the
127 	 * following structure:
128 	 * <pre>
129      * {
130      *      <b>tiles</b>: array of tiles
131      *      <b>x</b>: x offset of the grid from the top left corner of the map
132      *      <b>y</b>: y offest of the grid from the top left corner of the map
133      *      <b>position</b>: the grids position
134      *      <b>centerTile</b>: the tile on which the center is located
135      *      <b>zoom</b>: current zoom
136      *      <b>width</b>: width of the grid (in # of tiles)
137      *      <b>height</b>: height of the grid. 
138      * }</pre>
139      */
140     getGrid: function(){
141         return {
142             tiles: this.tiles,
143             x: this.gridX,
144             y: this.gridY,
145             position: this.gridPosition,
146             centerTile: this.centerTile,
147             zoom: this.zoom,
148             width: this.xTiles,
149             height: this.yTiles
150             };
151     },
152 
153     /**
154 	 * @private
155      * Called by the map when it is resized.
156      * This should not be used directly. 
157      * @param {float} width new width
158      * @param {float} height new height
159      */
160     resize: function(width, height){
161         this.width = width;
162         this.height = height;
163     }
164 
165 }
