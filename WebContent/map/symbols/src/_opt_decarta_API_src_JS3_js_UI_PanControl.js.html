1 /**
  2  * @class
  3  * Add a pan control to the map.
  4  * This class inherits from {@link deCarta.UI.MapControl}.
  5  *
  6  * @description Pan Map Control
  7  *
  8  * The PanControl provides a control consisting of 4 arrows 
  9  * over the map that let you pan the map, and provide a GPS locate button in
 10  * the center of the arrows.
 11  * This class inherits from {@link deCarta.UI.MapControl}.
 12  * @param opt Options A list of options with which to initialize the PanControl.
 13  * Valid options are:
 14  * <ul>
 15  *   <li>(string) position: (Inherited from {@link deCarta.UI.MapControl}), which should
 16  *       be set to one of: 'topLeft', 'topRight', 'bottomLeft', 'bottomRight'
 17  *       (default='topLeft')</li>
 18  *   <li>(int) panDistance: 
 19  *       distance (in pixels) of each pan event (defualt = 50)</li>
 20  * </ul>
 21  * To style this control, edit the corresponding CSS file in the css/UI dir.
 22  * @see deCarta.UI.MapControl
 23  * @see deCarta.Core.Map 
 24  */
 25 deCarta.UI.PanControl = function(opt){
 26     this.options = {
 27         cssClass : 'deCarta-control-pan',
 28         panDistance: 50,
 29         layout: function(width, height, existingControls){
 30             return {top: 10, left: 10}
 31         }
 32     }
 33     
 34     this.options = deCarta.Utilities.extendObject(this.options, opt);
 35     
 36     deCarta.UI.MapControl.call(this, this.options);
 37 /*    if (!this.options.style){
 38         if (deCarta.Window.isMobile()) this.options.style = 'mobile';
 39         else this.options.style = 'desktop';
 40     }*/
 41     this.locateTimeout = 10000;
 42     this.locateHighAccuracy = false;
 43     this.locateMaxAge = 1000 * 60 * 60 * 24;
 44 	
 45 }
 46 
 47 
 48 //Define methods to extend PanControl
 49 deCarta.UI.PanControl.prototype = {
 50         
 51         /**
 52 		 * @private
 53 	     */
 54         initElements: function(){
 55 
 56             this.domElement = document.createElement('div');
 57             this.contentElement = document.createElement('div');
 58             this.domElement.style.position = 'absolute';
 59             this.contentElement.style.position = 'relative';
 60             this.contentElement.className = 'deCarta-control-pan';
 61             if(this.options.cssClass !== this.contentElement.className){
 62                 this.contentElement.className = this.contentElement.className + ' ' + this.options.cssClass;
 63             }
 64             this.domElement.className = "domElement";
 65 
 66             var commonStyle = {
 67                 height: '16px',
 68                 width: '16px',
 69                 position:'absolute',
 70                 cursor: 'pointer'
 71             }
 72             
 73             this.panNorthBtn = document.createElement('div');
 74             deCarta.Utilities.extendStyle(this.panNorthBtn.style, commonStyle);
 75             this.panNorthBtn.style.top = 0;
 76             this.panNorthBtn.style.left = '16px';
 77             //this.panNorthBtn.style.backgroundColor = '#0F0';
 78             deCarta.Touch.attachListener('tap', this.panNorthBtn, this.options.map.pan.bind(this.options.map, 'north', this.options.panDistance), true);
 79             
 80             this.panSouthBtn = document.createElement('div');
 81             deCarta.Utilities.extendStyle(this.panSouthBtn.style, commonStyle);
 82             this.panSouthBtn.style.top = '32px';
 83             this.panSouthBtn.style.left = '16px';
 84             //this.panSouthBtn.style.backgroundColor = '#FF0';
 85             deCarta.Touch.attachListener('tap', this.panSouthBtn, this.options.map.pan.bind(this.options.map, 'south', this.options.panDistance), true);
 86             
 87             this.panWestBtn  = document.createElement('div');
 88             deCarta.Utilities.extendStyle(this.panWestBtn.style, commonStyle);
 89             this.panWestBtn.style.top = '16px';
 90             this.panWestBtn.style.left = '0px';
 91             //this.panWestBtn.style.backgroundColor = '#0FF';
 92             deCarta.Touch.attachListener('tap', this.panWestBtn, this.options.map.pan.bind(this.options.map, 'west', this.options.panDistance), true);
 93             
 94             this.panEastBtn  = document.createElement('div');
 95             deCarta.Utilities.extendStyle(this.panEastBtn.style, commonStyle);
 96             this.panEastBtn.style.top = '16px';
 97             this.panEastBtn.style.left = '32px';
 98             //this.panEastBtn.style.backgroundColor = '#00F';
 99             deCarta.Touch.attachListener('tap', this.panEastBtn, this.options.map.pan.bind(this.options.map, 'east', this.options.panDistance), true);
100                         
101             this.locateBtn  = document.createElement('div');
102             deCarta.Utilities.extendStyle(this.locateBtn.style, commonStyle);
103             this.locateBtn.style.top = '16px';
104             this.locateBtn.style.left = '16px';  
105             deCarta.Touch.attachListener('tap', this.locateBtn, this.geoLocate.bind(this), true)
106             
107             this.contentElement.appendChild(this.panNorthBtn);            
108             this.contentElement.appendChild(this.panWestBtn);
109             this.contentElement.appendChild(this.locateBtn);
110             this.contentElement.appendChild(this.panEastBtn);
111             this.contentElement.appendChild(this.panSouthBtn);          
112             
113             this.ready = true;                        
114             
115         },
116         
117         /**
118 		 * @private
119 	     */
120         geoLocate: function(){
121             if (!navigator.geolocation) return;
122             var map = this.options.map;
123             var locateBtn = this.locateBtn;
124             locateBtn.src= 'img/pan_spinner.gif';
125             navigator.geolocation.getCurrentPosition(
126                 function(position){
127                     var pos = new deCarta.Core.Position(position.coords.latitude, position.coords.longitude);
128                     map.centerOn(pos);
129                     locateBtn.src= 'img/pan_center.png';
130                 },
131                 function(error){
132                     //error error oh
133                 },
134                 {
135                     timeout:this.locateTimeout,
136                     maximumAge:this.locateMaxAge,
137                     enableHighAccuracy:this.locateHighAccuracy
138                 }
139             );
140         },
141         
142        /**
143 	    * This render method implements the render method from the 
144 	    * {@link deCarta.UI.MapControl} base class.
145 	    * It is responsible for rendering this control on the map,
146 	    * and produces a single HTML Dom Element containing the whole
147 	    * GUI for the control.
148 	    * @param {HTMLDOMElement} container The container which holds this control
149 	    */
150         render: function(container){
151             if (!this.ready) this.initElements();
152             this.domElement.appendChild( this.contentElement );
153             container.appendChild( this.domElement );            
154             this.width = this.domElement.offsetWidth;
155             this.height = this.domElement.offsetHeight;        
156         }
157 
158 }; //end PanControl prototype
159 
160 //Extend the MapControl with the additional methods for PanControl
161 deCarta.UI.PanControl.prototype = deCarta.Utilities.inherit(deCarta.UI.PanControl.prototype, deCarta.UI.MapControl.prototype);
