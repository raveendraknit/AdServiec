1 /**
  2  * @private
  3  * Provides a layer slider control. This class is not currently supported by the deCarta Mobile JavaScript API.
  4  */
  5 deCarta.UI.LayerControl = function(opt){    
  6     deCarta.UI.MapControl.call(this, opt);
  7 }
  8 deCarta.UI.LayerControl.prototype = {
  9 
 10     ready: false,
 11 
 12     initElements: function(){
 13 
 14         this.domElement = document.createElement('div');
 15         this.domElement.style.position = 'absolute';
 16         this.contentElement = document.createElement('div');
 17         this.contentElement.className = 'deCarta-control-layer';
 18         if(this.options.cssClass && this.options.cssClass !== this.contentElement.className){
 19             this.contentElement.className = this.contentElement.className + ' ' + this.options.cssClass;
 20         }        
 21         this.domElement.className = "domElement";
 22 
 23         this.streetBtn = document.createElement('div'); 
 24         this.streetBtn.className = 'button street selected';
 25         deCarta.Touch.attachListener('tap', this.streetBtn, function(){
 26                 this.options.map.setStreetView() 
 27         }.bind(this), true);
 28 
 29         this.satelliteBtn = document.createElement('div');
 30         deCarta.Touch.attachListener('tap', this.satelliteBtn, function(){
 31             if(document.getElementById("hybridCheck").checked)
 32                 this.options.map.setHybridView() 
 33             else
 34                 this.options.map.setSatelliteView() 
 35         }.bind(this), true);
 36         this.satelliteBtn.className = 'button satellite';
 37 
 38         this.hybridDiv = document.createElement('div');
 39         this.hybridCheck = document.createElement('input');
 40         this.hybridCheck.type="checkbox"
 41         this.hybridCheck.id="hybridCheck"
 42         this.hybridDiv.appendChild(this.hybridCheck)
 43         this.hybridDiv.innerHTML+=" Labels"
 44 
 45         var hybridMode = false;
 46 
 47         this.hybridDiv.className = 'hybrid';
 48 
 49         var clear = document.createElement('div');
 50         clear.style.clear = 'both';
 51 
 52         this.contentElement.appendChild(this.streetBtn);
 53         this.contentElement.appendChild(this.satelliteBtn);
 54 
 55         this.contentElement.appendChild(this.hybridDiv);
 56         this.contentElement.appendChild(clear);
 57 
 58         this.domElement.appendChild(this.contentElement);
 59       
 60         this.options.map.onviewchange(function(ev){
 61             switch (ev.view){
 62                 case 'street':
 63                     this.streetBtn.className = 'button street selected';
 64                     this.satelliteBtn.className = 'button satellite';
 65                     this.hybridDiv.style.display="none"
 66                 break;
 67                 case 'satellite':
 68                     this.streetBtn.className = 'button street';
 69                     this.satelliteBtn.className = 'button satellite selected';
 70                     this.hybridDiv.style.display="block"
 71                 break;
 72                 case 'hybrid':
 73                     this.streetBtn.className = 'button street';
 74                     this.satelliteBtn.className = 'button satellite selected';
 75                     this.hybridDiv.style.display="block"
 76                 break;
 77             }
 78         }.bind(this));
 79 
 80         this.ready = true;
 81     },
 82 
 83     render: function(container){        
 84         if (!this.ready) this.initElements();
 85         container.appendChild( this.domElement );
 86 
 87         // added this after DOM insertion
 88         document.getElementById("hybridCheck").onchange=function(){
 89             if(document.getElementById("hybridCheck").checked){
 90                 this.options.map.setHybridView() 
 91             }else{
 92                 this.options.map.setSatelliteView() 
 93             }
 94         }.bind(this)
 95         this.width = this.domElement.offsetWidth;
 96         this.height = this.domElement.offsetHeight;          
 97     }
 98 	
 99 	
100 }; //end LayerControl prototype
101 
102 //Extend the MapControl with the additional methods for LayerControl
103 deCarta.UI.LayerControl.prototype = deCarta.Utilities.inherit(deCarta.UI.LayerControl.prototype, deCarta.UI.MapControl.prototype);
