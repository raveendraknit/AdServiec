1 /**
  2  * @class
  3  * The Map Class provides the necessary objects for a mobile online
  4  * mapping application: it provides the map display and interaction. In its
  5  * simplest form, Map objects provide a draggable map to the end user.
  6  * 
  7  * @description The master map object.
  8  *
  9  * @constructor
 10  * @param {object} options. An object containing some (or all) of the following
 11  * properties:
 12  * <ul>
 13  *   <li>{HTMLDOMElement} id: DOM id of the map element *required*</li>
 14  *   <li>(int) zoom: the starting zoom level, 20=maxzoom, 1=minzoom, default=3</li>
 15  *   <li>{@link deCarta.Core.Position} center: Geographic position where the map is initially centered</li>
 16  *   <li>(bool) easing: whether easing (smooth transitions between zoom levels) is enabled (default: true)</li>
 17  *   <li>(bool) digitalZoom: <em>true</em> enables animated zoom between levels, optional, default=true</li>
 18  *   <li>(bool) doubleTapZoom: <em>true</em> enables double click (tap) to zoom on position, optional, default=true</li>
 19  *   <li>(array of {@link deCarta.Core.MapLayer}) layers: set of tile layers which will be initialized with the map</li>
 20  *   <li>(array of {@link deCarta.UI.MapControl}) controls: set of map control objects to add to the map</li>
 21  *   <li>onReady: a callback function which will be invoked when the map has loaded</li>
 22  *   <li>(bool) draggable: If <em>true</em> allows map to be draggable. Optional, default="true"</li>
 23  *   <li>(int) controlZ: css Z-index for map controls. Each new control created
 24  *       will have an increasing css z-index starting from this value, allowing
 25  *       the application to insert page elements above or below the map controls
 26  *       </li>
 27  *   <li>(bool) resizeable: If <em>true</em>, map is resizeable. Optional,
 28  *       default=true</li>
 29  *   <li>(bool) autoResize: If <em>true</em>, the map will resize automatically when the container is resized. Optional,
 30  *       default=true</li>
 31  *   <li>(bool) scrollWheelEnabled: If <em>true</em>, the scroll wheel will serve as a zoom control. Optional,
 32  *       default=true</li> 
 33  *   <li>(int) maxZoom: Maximum zoom level (tighest zoom), in the range 1
 34  *       (min possible zoom) to 20 (max possible zoom). Optional, default=19.</li>
 35  *   <li>(int) minZoom: Minimum zoom level (widest zoom), in the range 1
 36  *       (min possible zoom) to 20 (max possible zoom). Optional, default=2.</li>
 37  *   <li>skipResources: Set to true to skip loading of resources. Optional, 
 38  *       default=false</li>
 39  *   <li>boundaries: Defines the latitude and longitude limits for the map. The
 40  *       map will not display tiles when outside of these bounds. This object has these fields:
 41  *     <ul>
 42  *       <li>(float) top: maximum latitude (up to 90), default: 87</li>
 43  *       <li>(float) bottom: minimum latitude (down to -90) , default: -87</li>
 44  *       <li>(float) left: leftmost longitude (-180 to 180), default: null</li>
 45  *       <li>(float) right: right longitude (-180 to 180), default: null</li>
 46  *     </ul>
 47  *   </li>
 48  *   <li>(bool) avoidEvents: If <em>true</em>, the Map does not respond to any input events. optional, default=false</li>
 49  * </ul>
 50  * 
 51  * Map Events
 52  * 
 53  * 
 54  * Your application can add listeners for specific map events. This can be done through the EventManager,
 55  * or through proxy methods available on the Map object. For example, to install a callback for a zoomStart event,
 56  * you can proceed in two ways:
 57  * 
 58  * map.zoomstart(function(){//your code});
 59  * 
 60  * or
 61  * 
 62  * deCarta.Core.EventManager.listen('zoomstart', function(){//your code});
 63  * 
 64  * 
 65  *
 66  * @throw {Exception} if the id option is not set, or invalid.

357     /**
358      * Resize the map according to the new size of the containing element.
359      * The map will adapt to fill 100% of the containing element when this function
360      * is called
361      */        
362     resize: function(){
363         
364         if (this._lastResize && !this.options.autoResize) {            
365             return;
366         }
367         if (this._lastResize && this.options.autoResize && this.options.minResizeInterval){
368             
369             var now = new Date().getTime();
370             if (now < (this._lastResize + this.options.minResizeInterval)){
371                 
372                 if (this._resizeTimeout) clearTimeout(this._resizeTimeout);
373                 this._resizeTimeout = setTimeout(this.resize.bind(this),this.options.minResizeInterval );
374                 return;
375             }
376         }        
377         //if (!this.options.resizeable) return;
378         //fix this
379         var i = 0;
380         
381         /*if (this.options.mode == 'canvas'){
382             this.width = this.canvas.width = parseFloat(this.containerElement.style.width);
383             this.height = this.canvas.height = parseFloat(this.containerElement.style.height);
384         } else {*/
385         this.width = parseFloat(this.containerElement.style.width);
386         this.height = parseFloat(this.containerElement.style.height);
387         
388         if (!this.width || !this.height){
389             this.width = parseFloat(this.containerElement.clientWidth);
390             this.height = parseFloat(this.containerElement.clientHeight);
391         }           
392 
393         this.canvas.style.width = this.width + 'px';            
394         this.canvas.style.height = this.height + 'px';
395               
396         //}                
397 
398         if (deCarta.Window.isIe()){
399             window.attachEvent('onresize', this.resize.bind(this));
400         } else {
401             window.addEventListener('resize', this.resize.bind(this));
402         }
403 
404         this.tileGrid.resize(this.width, this.height);
405 
406         for (i = 0; i < this.layers.length; i++){
407             this.layers[i].resize(this.width, this.height);
408         }
409 
410         this.triggerEvent('resize');
411         this._lastResize = new Date().getTime();
412         this.drawControls();
413         this.render();
414     },

416     /**
417      * Returns the highest css z-index used by the map and it's control and overlay
418 	 * layers. This allows the user to place page elements above the map and all
419 	 * its layers.
420      * @return {int} The highest z-index used by the map
421      */
422     getHighestZ: function(){
423 
424         var max = 0;
425         try {
426             //find highest layer z
427             for (var i = 0; i < this.layers.length; i++){
428                 if (this.layers[i].options.zIndex) {
429                     if (this.layers[i].options.zIndex > max) max = this.layers[i].options.zIndex;
430                 }
431             }
432         }catch (e) {
433         //error
434         }
435 
436         return deCarta.Core.Configuration.baseOvlZ + max;
437     },

538     /**
539      * Add a layer (tile set) to the map. {@see deCarta.Core.MapLayer}
540      * @param {deCarta.Core.MapLayer} layer The MapLayer object you wish to add.
541      */
542     addLayer: function(layer){
543 
544         if (!layer.render || !layer.resize){
545             deCarta.Core.Exception.raise ('The layer you are trying to add does not support required methods "render" or "resize".');
546         }
547 
548         var dest = (layer.type == 'layer')  ? this.layerContainer : this.ovlContainer;
549 
550         layer.setOwner({
551             owner: this,
552             canvas: dest
553         });
554         this.layers.push(layer);
555         return layer;
556
