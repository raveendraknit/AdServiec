1 /**
  2 *	@class
  3 *	The Time class defines a structure for easily manipualting time values,
  4 *	for instance those returned by a routing request, or a corridor search.
  5 *
  6 * @constructor
  7 * @param {float} time value
  8 * @param {string} uom Unit of measure. valid values are : S (Seconds), M (Minutes), H (Hours), D (days)
  9 */
 10 deCarta.Core.Time = function(value, uom){	
 11 
 12 	if (typeof value == 'object'){
 13 		if (value.value && value.uom){
 14 			this.value = value.value;
 15 			this.uom = value.uom;
 16 			return this;
 17 		}
 18 		throw ('Unrecognized format for time value')
 19 	}
 20 
 21 	this.value = value || 0;
 22 	this.uom = uom || 'S';
 23 	return this;
 24 }
 25 
 26 deCarta.Core.Time.prototype = {
 27 
 28 	toString: function(){
 29 		return this.value + ' ' + this.uom;
 30 	},
 31 
 32 	toJSON: function(){
 33 		return JSON.stringify({
 34 			value: this.value,
 35 			uom: this.uom
 36 		});
 37 	}
 38 
 39 }
