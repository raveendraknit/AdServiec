1 /**
  2  *  @class
  3  *  A MapLayer represents a tile set that can be displayed on the map.
  4  *  Tiles are addressed by their North, East and Zoom parameters.
  5  *  Map layers can be customized by setting a specific tile store to get the
  6  *  tiles from (a street view tile, a satellite tile, whatever you come up with)
  7  *
  8  *  There is no limit, other than performance, to the number of layers that can be
  9  *  added to a map.
 10  *
 11  *  @description A tile set that can be displayed on the map
 12  *
 13  *  @constructor
 14  *  @param {object} options
 15  *  Each layer is initilized with a set of options.
 16  *
 17  *  options:
 18  *  <pre>
 19  *  {
 20  *      tileStore : the tile store that will provide tiles for this layer. defaults to street.
 21  *      name: the layer's name
 22  *      visible: visibility attribute for the layer
 23  *  }
 24  *  </pre>
 25  *
 26  *  @see deCarta.Core.MapOverlay
 27  */
 28 deCarta.Core.MapLayer = function MapLayer(options){
 29     
 30     this.options = {
 31         canvas: null,
 32         owner: null,
 33         name: 'Unnamed Layer',
 34         id: 'deCarta-layer-' + Math.random().toString().replace('.',''),
 35         visible: true,
 36         tileStore: new deCarta.Core.StreetTileStore(),
 37         opacity: 100
 38     }
 39     this.options = deCarta.Utilities.extendObject(this.options, options);    
 40     this.type ='layer';
 41     this.ready = false;
 42 }
 43 
 44 deCarta.Core.MapLayer.prototype = {
 45 
 46     /**
 47      *  @private. Completes initialization.
 48      *
 49      */
 50     _initialize: function(){
 51         if (!this.options.canvas || !this.options.owner){
 52             deCarta.Core.Exception.raise('MapLayer requires a canvas and an owner for instantiation.');
 53         }
 54         
 55         this.visibleTiles = {};
 56 
 57         this.owner = this.options.owner;
 58         this.canvas = this.options.canvas;
 59         this.name = this.options.name;
 60         this.visible = this.options.visible;
 61         
 62         this.options.tileStore.setOwner(this.owner);
 63         this.options.tileStore.startPrecaching();
 64         //console.log('', this.owner.options.mode)
 65         /*if (this.owner.options.mode == 'WebGL' && deCarta.Core.WebGLRenderer && THREE){
 66             this.renderer = new deCarta.Core.WebGLRenderer(this.canvas);
 67         } else */
 68         if (this.owner.options.mode == 'canvas' && deCarta.Core.CanvasRenderer){
 69             this.renderer = new deCarta.Core.CanvasRenderer(this.canvas);
 70         } else if (deCarta.Core.Configuration.useHardwareAcceleration && deCarta.Window.hasCSSTransforms()){             
 71             this.renderer = new deCarta.Core.CSS3Renderer(this.canvas);
 72         } else {            
 73             this.renderer = new deCarta.Core.HTMLRenderer(this.canvas);
 74         }        
 75 
 76         this.resize();
 77 
 78         this.setOpacity(this.options.opacity);
 79 
 80         this.tileStore = this.options.tileStore;
 81         this.ready = true;
 82     },
 83     
 84     /**
 85     * Renders the map.
 86     * Should not be called directly. Renderings are controlled by the Map object.
 87     * @param: (deCarta.Core.TileGrid) grid
 88     */
 89     render: function(tileGrid){
 90         
 91         if (!this.visible) return;               
 92         
 93         var renderingQueue = [];
 94         var tileSet = {};
 95 
 96         var grid = tileGrid.getGrid();
 97         var gridTiles = grid.tiles;
 98         
 99         var time = new Date().getTime();
100 
101         for (var x = 0; x < gridTiles.length; x++){
102             
103             var tile = gridTiles[x];
104             
105             var tileImg = this.tileStore.getTile(tile.E,tile.N,tile.Z);
106             
107             if (!tileSet[tileImg.name]){
108 
109                 var drawEntry = {
110                     img : tileImg.img,
111                     x: tile.dX + (tileImg.ofsX * tile.size),
112                     y: tile.dY + (tileImg.ofsY * tile.size),
113                     size: tile.size * tileImg.scale,
114                     time: (this.visibleTiles[tileImg.name]) ? this.visibleTiles[tileImg.name] : time,
115                     scale: tileImg.scale
116                 }
117 
118                 if (!(drawEntry.x >= this.width || drawEntry.y  >= this.height || drawEntry.x + drawEntry.size < 0 || drawEntry.y + drawEntry.size < 0)) {
119                     renderingQueue.push(drawEntry);
120                 }
121                 tileSet[tileImg.name] = true;                
122             }
123 			
124         }
125         
126         //now merge tileSet to the visibleTiles
127         for (var prop in this.visibleTiles){
128             if (!tileSet[prop]) delete this.visibleTiles[prop];
129         }
130         for (var prop in tileSet){
131             if (!this.visibleTiles[prop]) this.visibleTiles[prop] = new Date().getTime();
132         }
133 
134         this.tileStore.purgeLoadRequests(grid.centerTile, grid.zoom, Math.max(grid.width, grid.height));
135 
136         renderingQueue.sort(function (a, b){
137             return b.size - a.size;
138         });
139 
140         this.renderer.render(renderingQueue);
141 
142     },
143 
144     /**
145      * @private
146      * Registers an owner for this layer.
147      * It should be a map. 
148      */
149     setOwner: function(ownerOpts){
150         this.options.owner = ownerOpts.owner;
151         this.options.canvas = ownerOpts.canvas;
152         this._initialize();
153     },
154 
155     /**
156 	 * @private
157      * Notifies the layer of a map resize.
158      * Should not be called directly. This is handled by the Map object. 
159      * @param width (float)
160      * @param height (float)
161      */
162     resize: function(width, height){
163                    
164         this.width = width;
165         this.height = height;
166 
167         this.renderer.resize(width, height);
168     },
169 
170     /**
171      * Sets the layer as visible.
172      */
173     show: function(){
174         this.visible = true;
175         this.renderer.show();
176     },
177 
178     /**
179      * Hides the layer from view. 
180      */
181     hide: function(){
182         this.visible = false;
183         this.renderer.hide();
184     },
185 
186     toggle: function(){
187         
188         if (this.visible){
189             this.hide();
190         } else {
191             this.show();
192         }
193     },
194 
195     setOpacity: function(opacity){
196         this.options.opacity = opacity;
197         this.renderer.setOpacity(opacity);
198     },
199     /**
200 	 * @private
201 	 */
202     toJSON: function(){
203         var layerObj = {
204             name: this.options.name,
205             visible: this.options.visible,
206             tileStore: this.options.tileStore
207         };        
208 
209         return layerObj;
210     }
211     
212 }
