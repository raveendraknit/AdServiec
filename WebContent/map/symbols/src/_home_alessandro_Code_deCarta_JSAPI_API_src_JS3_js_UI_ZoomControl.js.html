1 /**
  2  * @class ZoomControl - Add a zoom control to the map to present the user
  3  * with zoom in / zoom out buttons.
  4  * This class inherits from {@link deCarta.UI.MapControl}.
  5  *
  6  * @description Zoom Map Control
  7  *
  8  * @constructor
  9  * @param opt Options A list of options with which to initialize the ZoomControl.
 10  * Valid options are:
 11  * <ul>
 12  *   <li>(string) position: (Inherited from {@link deCarta.UI.MapControl}), which should
 13  *       be set to one of: 'topLeft', 'topRight', 'bottomLeft', 'bottomRight' (default='topLeft')</li>
 14  *   <li>(bool) continuousZoom: in case of desktop style, zoom will be continuous if the zoom handle is dragged.
 15  *   Performance intensive. Use appropriately. </li>
 16  *   <li>(bool) autoResize: If this options is set, the zoom controller will collapse to the zoomin / out 
 17  *   buttons only if there isn't sufficient space for the full element. Default: true </li> 
 18  * </ul>
 19  * To style this control, edit the corresponding CSS file in the css/UI dir.
 20  *
 21  * @see deCarta.UI.MapControl
 22  * @see deCarta.Core.Map
 23  */
 24 
 25 deCarta.UI.ZoomControl = function(opt){
 26     this.options = {
 27         autoResize: true,
 28         continuousZoom : true,
 29         cssClass : 'deCarta-control-zoom',
 30         layout: function(width, height, existingControls, allControls){
 31             var top = 10;
 32             if (allControls)
 33                 for (var i = 0; i < allControls.length; i++){
 34                     var c = allControls[i];
 35                     if (c instanceof deCarta.UI.PanControl){
 36                         var p = c.options.layout(width, height, existingControls);
 37                         if (p.top == 10 && p.left == 10){
 38                             var top = 70;
 39                         }
 40                     }
 41                 }
 42             return {
 43                 top: top,
 44                 left: 20
 45             }
 46         }
 47     }
 48 
 49     this.options = deCarta.Utilities.extendObject(this.options, opt);
 50 
 51     deCarta.UI.MapControl.call(this, this.options);
 52 
 53 }
 54 
 55 
 56 //Define methods to extend ZoomControl
 57 deCarta.UI.ZoomControl.prototype = {
 58 
 59     /**
 60      * @private
 61      */
 62     ready: false,
 63 
 64     /**
 65      * @private
 66      */
 67     zoomLevels: 21,
 68 
 69     /**
 70      * @private
 71      */
 72     zoomOfs: 0,
 73 
 74     /**
 75      * @private
 76      */
 77     initElements: function(){
 78         if (this.domElement) return;
 79 
 80         this.domElement = document.createElement('div');
 81         this.contentElement = document.createElement('div');
 82         this.domElement.style.position = 'absolute';
 83         this.contentElement.className = 'deCarta-control-zoom';
 84         if(this.options.cssClass !== this.contentElement.className){
 85             this.contentElement.className = this.contentElement.className + ' ' + this.options.cssClass;
 86         }
 87         this.domElement.className = "domElement";
 88 
 89         this.zoomIn = document.createElement('div');
 90         this.zoomOut = document.createElement('div');
 91 
 92         this.zoomLevels = this.options.map.maxZoom - this.options.map.minZoom + 1; //[]
 93         this.zoomOfs = this.options.map.minZoom;
 94 
 95         this.makeControl();
 96 
 97         this.zoomIn.onclick = function(){
 98             this.options.map.zoomIn();
 99         }.bind(this);
100 
101         this.zoomOut.onclick = function(){
102             this.options.map.zoomOut();
103         }.bind(this);
104 
105         this.ready = true;
106     },
107 
108     /**
109      * @private
110      * renders the control with a desktop look.
111      */
112     makeControl: function(){
113         this.positionCursor = function(){
114 
115             if (this.cursor)
116                 this.cursor.style.top = ((this.zoomLevels - this.options.map.zoom + this.zoomOfs -1) * 10 ) + 'px';
117 
118         }.bind(this);
119 
120         this.options.map.onzoomend(this.positionCursor);
121 
122         this.zoomIn.className = "deCarta-control-zoomInBtn"
123 
124         this.zoomOut.className = "deCarta-control-zoomOutBtn"
125 
126         var self = this;
127         function zoomFromY(y){
128             return (self.zoomLevels + self.zoomOfs - (y / 10));
129         }
130 
131         this.pillar = document.createElement('div');
132         this.pillar.className = "deCarta-control-zoomPillar";
133         this.pillar.unselectable = 'on';
134 
135         this.pillar.style.height = (10 * this.zoomLevels)  + 'px'
136 
137         this.cursor = document.createElement('div');
138         this.cursor.unselectable = 'on';
139         this.cursor.className = "deCarta-control-zoomCursor";
140 
141         deCarta.Touch.attachListener('touchstart', this.pillar, function(ev, oev){
142             this.cursorDragging = true;
143             this.pillar.style.cursor = 'url("img/closedhand.cur"), auto';
144             this.cursor.style.cursor = 'url("img/closedhand.cur"), auto';
145             //prevent the defaults 
146             if (oev && oev.preventDefault) oev.preventDefault();
147             if (oev && oev.stopPropagation) oev.stopPropagation();
148             if (oev && oev.stopImmediatePropagation) oev.stopImmediatePropagation();            
149         }.bind(this), false);
150 
151         deCarta.Touch.attachListener('touchmove', document.body, function(ev, oev){
152 
153             if (!this.cursorDragging) 
154                 return false;
155             var y = ev.pageY - deCarta.Utilities.domPosition(this.pillar).top;
156             //console.log(y, oev.pageY, deCarta.Utilities.domPosition(this.pillar).top, ev.pageY, oev);
157             if (this.options.continuousZoom) {
158                 this.options.map.zoomTo(zoomFromY(y), null, true);
159             }
160 
161             //prevent the defaults 
162             if (oev && oev.preventDefault) oev.preventDefault();
163             if (oev && oev.stopPropagation) oev.stopPropagation();
164             if (oev && oev.stopImmediatePropagation) oev.stopImmediatePropagation();
165 
166         }.bind(this), false);
167 
168         deCarta.Touch.attachListener('touchend', document.body, function (ev, oev){
169             if (this.cursorDragging){
170 
171                 this.pillar.style.cursor = 'url("img/openhand.cur"), auto';
172                 this.cursor.style.cursor = 'url("img/openhand.cur"), auto';
173 
174                 this.cursorDragging = false;
175                 var y = ev.pageY - deCarta.Utilities.domPosition(this.pillar).top;
176                 var level = zoomFromY(y);
177 
178                 this.options.map.zoomTo(Math.round(level), null, true);
179 
180             }
181         }.bind(this), false);
182 
183         var out = function(e){
184             if (!this.cursorDragging) return;
185             if (!e.relatedTarget) e.relatedTarget = e.toElement;
186             if (!(e.relatedTarget == this.cursor || e.relatedTarget == this.pillar)){
187 
188                 this.cursorDragging = false;
189                 this.pillar.style.cursor = 'url("img/openhand.cur"), auto';
190                 this.cursor.style.cursor = 'url("img/openhand.cur"), auto';
191                 var y = parseInt(this.cursor.style.top) + 10;
192                 this.options.map.zoomTo(Math.round(zoomFromY(y)));
193             }
194 
195         }.bind(this);
196 
197         if (deCarta.Window.isIe())
198             this.options.map.canvas.attachEvent('onmouseout', out);
199         else
200             this.options.map.canvas.addEventListener('mouseout', out, true);
201 
202         this.pillar.appendChild(this.cursor);
203 
204         this.contentElement.appendChild(this.zoomIn);
205         this.contentElement.appendChild(this.pillar);
206         this.contentElement.appendChild(this.zoomOut);
207 
208         this.positionCursor();
209 
210         this.unbridledHeight = this.height = parseFloat(this.pillar.style.height) + (27 * 2);
211 
212     },
213 
214     /**
215     * This render method implements the render method from the
216     * {@link deCarta.UI.MapControl} base class.
217     * It is responsible for rendering this control on the map,
218     * and produces a single HTML Dom Element containing the whole
219     * GUI for the control.
220     * @param {string} container The DOM element within which the control is rendered.
221     */
222     render: function(container){
223         if (!this.ready) this.initElements();
224         this.domElement.appendChild( this.contentElement );
225         container.appendChild( this.domElement );
226         this.width = this.domElement.offsetWidth;
227         this.height = this.domElement.offsetHeight;
228     },
229 
230     resize: function(event){
231 
232         var height = this.calculateAvailableHeight();
233 
234         if(!this.options.autoResize || !this.ready) return;
235 
236         if(this.pillar && this.pillar.tagName=='DIV'){
237             if(this.unbridledHeight > height){
238                 this.pillar.style.display='none';
239             }else{
240                 this.pillar.style.display='block';
241             }
242         }
243 
244         this.height = this.domElement.offsetHeight;
245     },
246 
247     calculateAvailableHeight: function(){
248         var allHeight = this.options.map.height;
249         var myPlace = this.getPlacement();
250         for (var i = 0; i < this.options.map.controls.length; i++){
251             if (this.options.map.controls[i] == this) continue;
252             var place = this.options.map.controls[i].getPlacement();
253             //check if c is above / below
254             var overlap =
255             (place.left <= myPlace.left && place.right >= myPlace.right) ||
256             (place.right >= myPlace.left && place.right <= myPlace.right) ||
257             (place.left >= myPlace.left && place.left <= myPlace.right);
258 
259 
260             if (overlap) allHeight -= place.height;
261         }
262         return allHeight;
263     }
264 
265 
266 
267 }; //end ZoomControl prototype
