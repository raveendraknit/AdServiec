 57     lat2pix_EPSG3395: function(lat, zoom){
 58         
 59         var radPerPix = 2*Math.PI / (deCarta.Utilities.tileSizeForZoom(zoom) * Math.pow(2, Math.floor(zoom)));
 60 
 61         var radLat = (parseFloat(lat) * (2 * Math.PI))/360;
 62 
 63         var ecc = 0.08181919084262157;
 64 
 65         var sinPhi = Math.sin(radLat);
 66         
 67         var eSinPhi = ecc * sinPhi;
 68         
 69         var retVal = Math.log(((1.0 + sinPhi) / (1.0 - sinPhi)) *
 70             Math.pow((1.0 - eSinPhi) / (1.0 + eSinPhi), ecc)) / 2.0;
 71         
 72         return (retVal / radPerPix);

 84     forwardMercator: function(lat, lon, zoom) {       
 85         var wSize = (deCarta.Utilities.tileSizeForZoom(zoom) * Math.pow(2, Math.floor(zoom - 1))); 
 86          
 87         var x = lon * wSize / 180;
 88         var y = Math.log(Math.tan((90 + lat) * Math.PI / 360)) / (Math.PI / 180);
 89 
 90         y = y * wSize / 180;                       
 91         
 92         return {x: x, y:y};
 93     },

 95     inverseMercator: function(x, y, zoom) {
 96 
 97         var wSize = (deCarta.Utilities.tileSizeForZoom(zoom) * Math.pow(2, Math.floor(zoom - 1)));
 98 
 99         var lon = (x / wSize) * 180;
100         var lat = (y / wSize) * 180;
101 
102         lat = 180/Math.PI * (2 * Math.atan(Math.exp(lat * Math.PI / 180)) - Math.PI / 2);
103         
104         return new deCarta.Core.Position(lat, lon);
105     },

140     pix2lat_EPSG3395 : function(y, zoom){        
141         
142         var radPerPix = 2*Math.PI / (deCarta.Utilities.tileSizeForZoom(zoom) * Math.pow(2, Math.floor(zoom)));
143 
144         var phiEpsilon = 1E-7;
145         var phiMaxIter = 12;
146         var t = Math.pow(Math.E, -y * radPerPix);
147         var prevPhi = deCarta.Utilities.mercatorUnproject(t);
148         var newPhi = deCarta.Utilities.findRadPhi(prevPhi, t);
149         var iterCount = 0;
150         while ( iterCount < phiMaxIter &&
151             Math.abs(prevPhi - newPhi) > phiEpsilon) {
152             prevPhi = newPhi;
153             newPhi = deCarta.Utilities.findRadPhi(prevPhi, t);
154             iterCount++;
155         }
156         return newPhi*180/Math.PI;
157

248     /**
249      * Get distance between two positions
250      * @param {Position} Point1
251      * @param {Position} Point2
252      * @return distance in kilometers
253      */
254     getPointDistance: function(Point1, Point2){
255 
256         var lat1 = Point1.getLat();
257         var lat2 = Point2.getLat();
258         var lon1 = Point1.getLon();
259         var lon2 = Point2.getLon();
260 
261         var R = 6378.137; // km - THIS IS THE RADIUS OF THE CURRENT PLANET! (Earth).
262 
263         var dLat = (lat2 - lat1).toRad();
264         var dLon = (lon2 - lon1).toRad();
265         var a = Math.sin(dLat/2) * Math.sin(dLat/2) +
266                 Math.cos(lat1.toRad()) * Math.cos(lat2.toRad()) *
267                 Math.sin(dLon/2) * Math.sin(dLon/2);
268         var c = 2 * Math.atan2(Math.sqrt(a), Math.sqrt(1-a));
269         var d = R * c;
270 
271         return d;
272     },

406     /**
407 	 * @private
408 	 */
409     urlParse: function (urlString){
410 
411         var result = {};
412         var split = urlString.split('#')[0].split('?');
413         result.baseUrl = split[0];
414         result.queryParameters = {};
415 
416         var params = {};
417         try {
418             var queryParameters = split[1].split('&');
419             for (var i = 0; i < queryParameters.length; i++){
420                 var val = queryParameters[i].split('=');
421                 params[val[0]] = val[1];
422             }
423         } catch (e) {}
424         result.queryParameters = params;
425         
426         try {
427             split = result.baseUrl.split('://');
428             result.proto = split[0];
429             var hp = split[1].split('/');
430             
431             result.host = hp[0];
432             hp[0]='';
433             result.path = hp.join('/');
434         } catch (e) {
435             result.proto = 'unknown';
436         }
437         
438         return result;
439     },
